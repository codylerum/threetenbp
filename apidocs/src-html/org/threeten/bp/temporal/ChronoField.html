057/**
058 * A standard set of fields.
059 * <p>
060 * This set of fields provide field-based access to manipulate a date, time or date-time.
061 * The standard set of fields can be extended by implementing {@link TemporalField}.
062 * <p>
063 * These fields are intended to be applicable in multiple calendar systems.
064 * For example, most non-ISO calendar systems define dates as a year, month and day,
065 * just with slightly different rules.
066 * The documentation of each field explains how it operates.
067 *
068 * <h3>Specification for implementors</h3>
069 * This is a final, immutable and thread-safe enum.
070 */
071public enum ChronoField implements TemporalField {
072
073    /**
074     * The nano-of-second.
075     * <p>
076     * This counts the nanosecond within the second, from 0 to 999,999,999.
077     * This field has the same meaning for all calendar systems.
078     * <p>
079     * This field is used to represent the nano-of-second handling any fraction of the second.
080     * Implementations of {@code TemporalAccessor} should provide a value for this field if
081     * they can return a value for {@link #SECOND_OF_MINUTE}, {@link #SECOND_OF_DAY} or
082     * {@link #INSTANT_SECONDS} filling unknown precision with zero.
083     * <p>
084     * When this field is used for setting a value, it should set as much precision as the
085     * object stores, using integer division to remove excess precision.
086     * For example, if the {@code TemporalAccessor} stores time to millisecond precision,
087     * then the nano-of-second must be divided by 1,000,000 before replacing the milli-of-second.
088     */

099     */

121     * <p>
122     * This field is used to represent the micro-of-day handling any fraction of the second.

131     * The milli-of-second.
132     * <p>
133     * This counts the millisecond within the second, from 0 to 999.
134     * This field has the same meaning for all calendar systems.
135     * <p>
136     * This field is used to represent the milli-of-second handling any fraction of the second.
137     * Implementations of {@code TemporalAccessor} should provide a value for this field if
138     * they can return a value for {@link #SECOND_OF_MINUTE}, {@link #SECOND_OF_DAY} or
139     * {@link #INSTANT_SECONDS} filling unknown precision with zero.
140     * <p>
141     * When this field is used for setting a value, it should behave in the same way as
142     * setting {@link #NANO_OF_SECOND} with the value multiplied by 1,000,000.
143     */
144    MILLI_OF_SECOND("MilliOfSecond", MILLIS, SECONDS, ValueRange.of(0, 999)),
145    /**
146     * The milli-of-day.
147     * <p>
148     * This counts the millisecond within the day, from 0 to (24 * 60 * 60 * 1,000) - 1.
149     * This field has the same meaning for all calendar systems.
150     * <p>
151     * This field is used to represent the milli-of-day handling any fraction of the second.
152     * Implementations of {@code TemporalAccessor} should provide a value for this field if
153     * they can return a value for {@link #SECOND_OF_DAY} filling unknown precision with zero.
154     * <p>

157     */

160     * The second-of-minute.
161     * <p>
162     * This counts the second within the minute, from 0 to 59.
163     * This field has the same meaning for all calendar systems.
164     */
165    SECOND_OF_MINUTE("SecondOfMinute", SECONDS, MINUTES, ValueRange.of(0, 59)),
166    /**
167     * The second-of-day.
168     * <p>
169     * This counts the second within the day, from 0 to (24 * 60 * 60) - 1.
170     * This field has the same meaning for all calendar systems.
171     */
172    SECOND_OF_DAY("SecondOfDay", SECONDS, DAYS, ValueRange.of(0, 86400L - 1)),

175     * <p>
176     * This counts the minute within the hour, from 0 to 59.
177     * This field has the same meaning for all calendar systems.
178     */

181     * The minute-of-day.
182     * <p>
183     * This counts the minute within the day, from 0 to (24 * 60) - 1.
184     * This field has the same meaning for all calendar systems.
185     */
186    MINUTE_OF_DAY("MinuteOfDay", MINUTES, DAYS, ValueRange.of(0, (24 * 60) - 1)),

188     * The hour-of-am-pm.
189     * <p>
190     * This counts the hour within the AM/PM, from 0 to 11.
191     * This is the hour that would be observed on a standard 12-hour digital clock.
192     * This field has the same meaning for all calendar systems.
193     */
194    HOUR_OF_AMPM("HourOfAmPm", HOURS, HALF_DAYS, ValueRange.of(0, 11)),
195    /**
196     * The clock-hour-of-am-pm.
197     * <p>
198     * This counts the hour within the AM/PM, from 1 to 12.
199     * This is the hour that would be observed on a standard 12-hour analog wall clock.
200     * This field has the same meaning for all calendar systems.
201     */
202    CLOCK_HOUR_OF_AMPM("ClockHourOfAmPm", HOURS, HALF_DAYS, ValueRange.of(1, 12)),
203    /**
204     * The hour-of-day.
205     * <p>
206     * This counts the hour within the day, from 0 to 23.
207     * This is the hour that would be observed on a standard 24-hour digital clock.
208     * This field has the same meaning for all calendar systems.
209     */
210    HOUR_OF_DAY("HourOfDay", HOURS, DAYS, ValueRange.of(0, 23)),
211    /**
212     * The clock-hour-of-day.
213     * <p>
214     * This counts the hour within the AM/PM, from 1 to 24.
215     * This is the hour that would be observed on a 24-hour analog wall clock.
216     * This field has the same meaning for all calendar systems.
217     */
218    CLOCK_HOUR_OF_DAY("ClockHourOfDay", HOURS, DAYS, ValueRange.of(1, 24)),
219    /**
220     * The am-pm-of-day.
221     * <p>
222     * This counts the AM/PM within the day, from 0 (AM) to 1 (PM).
223     * This field has the same meaning for all calendar systems.
224     */
225    AMPM_OF_DAY("AmPmOfDay", HALF_DAYS, DAYS, ValueRange.of(0, 1)),
226    /**
227     * The day-of-week, such as Tuesday.
228     * <p>
229     * This represents the standard concept of the day of the week.
230     * In the default ISO calendar system, this has values from Monday (1) to Sunday (7).
231     * The {@link DayOfWeek} class can be used to interpret the result.
232     * <p>
233     * Most non-ISO calendar systems also define a seven day week that aligns with ISO.
234     * Those calendar systems must also use the same numbering system, from Monday (1) to
235     * Sunday (7), which allows {@code DayOfWeek} to be used.
236     * <p>
237     * Calendar systems that do not have a standard seven day week should implement this field
238     * if they have a similar concept of named or numbered days within a period similar
239     * to a week. It is recommended that the numbering starts from 1.
240     */
241    DAY_OF_WEEK("DayOfWeek", DAYS, WEEKS, ValueRange.of(1, 7)),
242    /**
243     * The aligned day-of-week within a month.
244     * <p>
245     * This represents concept of the count of days within the period of a week
246     * where the weeks are aligned to the start of the month.
247     * This field is typically used with {@link #ALIGNED_WEEK_OF_MONTH}.
248     * <p>
249     * For example, in a calendar systems with a seven day week, the first aligned-week-of-month
250     * starts on day-of-month 1, the second aligned-week starts on day-of-month 8, and so on.
251     * Within each of these aligned-weeks, the days are numbered from 1 to 7 and returned
252     * as the value of this field.
253     * As such, day-of-month 1 to 7 will have aligned-day-of-week values from 1 to 7.
254     * And day-of-month 8 to 14 will repeat this with aligned-day-of-week values from 1 to 7.
255     * <p>
256     * Calendar systems that do not have a seven day week should typically implement this
257     * field in the same way, but using the alternate week length.
258     */
259    ALIGNED_DAY_OF_WEEK_IN_MONTH("AlignedDayOfWeekInMonth", DAYS, WEEKS, ValueRange.of(1, 7)),
260    /**
261     * The aligned day-of-week within a year.
262     * <p>
263     * This represents concept of the count of days within the period of a week
264     * where the weeks are aligned to the start of the year.
265     * This field is typically used with {@link #ALIGNED_WEEK_OF_YEAR}.
266     * <p>
267     * For example, in a calendar systems with a seven day week, the first aligned-week-of-year
268     * starts on day-of-year 1, the second aligned-week starts on day-of-year 8, and so on.
269     * Within each of these aligned-weeks, the days are numbered from 1 to 7 and returned
270     * as the value of this field.
271     * As such, day-of-year 1 to 7 will have aligned-day-of-week values from 1 to 7.
272     * And day-of-year 8 to 14 will repeat this with aligned-day-of-week values from 1 to 7.
273     * <p>
274     * Calendar systems that do not have a seven day week should typically implement this
275     * field in the same way, but using the alternate week length.
276     */
277    ALIGNED_DAY_OF_WEEK_IN_YEAR("AlignedDayOfWeekInYear", DAYS, WEEKS, ValueRange.of(1, 7)),
278    /**
279     * The day-of-month.
280     * <p>
281     * This represents the concept of the day within the month.
282     * In the default ISO calendar system, this has values from 1 to 31 in most months.
283     * April, June, September, November have days from 1 to 30, while February has days
284     * from 1 to 28, or 29 in a leap year.
285     * <p>
286     * Non-ISO calendar systems should implement this field using the most recognized
287     * day-of-month values for users of the calendar system.
288     * Normally, this is a count of days from 1 to the length of the month.
289     */
290    DAY_OF_MONTH("DayOfMonth", DAYS, MONTHS, ValueRange.of(1, 28, 31)),
291    /**
292     * The day-of-year.
293     * <p>
294     * This represents the concept of the day within the year.
295     * In the default ISO calendar system, this has values from 1 to 365 in standard
296     * years and 1 to 366 in leap years.
297     * <p>
298     * Non-ISO calendar systems should implement this field using the most recognized
299     * day-of-year values for users of the calendar system.
300     * Normally, this is a count of days from 1 to the length of the year.
301     */
302    DAY_OF_YEAR("DayOfYear", DAYS, YEARS, ValueRange.of(1, 365, 366)),
303    /**
304     * The epoch-day, based on the Java epoch of 1970-01-01 (ISO).
305     * <p>
306     * This field is the sequential count of days where 1970-01-01 (ISO) is zero.
307     * Note that this uses the <i>local</i> time-line, ignoring offset and time-zone.
308     * <p>
309     * This field is strictly defined to have the same meaning in all calendar systems.
310     * This is necessary to ensure interoperation between calendars.
311     */
312    EPOCH_DAY("EpochDay", DAYS, FOREVER, ValueRange.of((long) (Year.MIN_VALUE * 365.25), (long) (Year.MAX_VALUE * 365.25))),
313    /**
314     * The aligned week within a month.
315     * <p>
316     * This represents concept of the count of weeks within the period of a month
317     * where the weeks are aligned to the start of the month.
318     * This field is typically used with {@link #ALIGNED_DAY_OF_WEEK_IN_MONTH}.
319     * <p>
320     * For example, in a calendar systems with a seven day week, the first aligned-week-of-month
321     * starts on day-of-month 1, the second aligned-week starts on day-of-month 8, and so on.
322     * Thus, day-of-month values 1 to 7 are in aligned-week 1, while day-of-month values
323     * 8 to 14 are in aligned-week 2, and so on.
324     * <p>
325     * Calendar systems that do not have a seven day week should typically implement this
326     * field in the same way, but using the alternate week length.
327     */
328    ALIGNED_WEEK_OF_MONTH("AlignedWeekOfMonth", WEEKS, MONTHS, ValueRange.of(1, 4, 5)),
329    /**
330     * The aligned week within a year.
331     * <p>
332     * This represents concept of the count of weeks within the period of a year
333     * where the weeks are aligned to the start of the year.
334     * This field is typically used with {@link #ALIGNED_DAY_OF_WEEK_IN_YEAR}.
335     * <p>
336     * For example, in a calendar systems with a seven day week, the first aligned-week-of-year
337     * starts on day-of-year 1, the second aligned-week starts on day-of-year 8, and so on.
338     * Thus, day-of-year values 1 to 7 are in aligned-week 1, while day-of-year values
339     * 8 to 14 are in aligned-week 2, and so on.
340     * <p>
341     * Calendar systems that do not have a seven day week should typically implement this
342     * field in the same way, but using the alternate week length.
343     */
344    ALIGNED_WEEK_OF_YEAR("AlignedWeekOfYear", WEEKS, YEARS, ValueRange.of(1, 53)),
345    /**
346     * The month-of-year, such as March.
347     * <p>
348     * This represents the concept of the month within the year.
349     * In the default ISO calendar system, this has values from January (1) to December (12).
350     * <p>
351     * Non-ISO calendar systems should implement this field using the most recognized
352     * month-of-year values for users of the calendar system.
353     * Normally, this is a count of months starting from 1.
354     */
355    MONTH_OF_YEAR("MonthOfYear", MONTHS, YEARS, ValueRange.of(1, 12)),
356    /**
357     * The epoch-month based on the Java epoch of 1970-01-01.
358     * <p>
359     * This field is the sequential count of months where January 1970 (ISO) is zero.
360     * Note that this uses the <i>local</i> time-line, ignoring offset and time-zone.
361     * <p>
362     * Non-ISO calendar systems should also implement this field to represent a sequential
363     * count of months. It is recommended to define zero as the month of 1970-01-01 (ISO).
364     */
365    EPOCH_MONTH("EpochMonth", MONTHS, FOREVER, ValueRange.of((Year.MIN_VALUE - 1970L) * 12, (Year.MAX_VALUE - 1970L) * 12L - 1L)),
366    /**
367     * The year within the era.
368     * <p>
369     * This represents the concept of the year within the era.
370     * This field is typically used with {@link #ERA}.
371     * <p>
372     * The standard mental model for a date is based on three concepts - year, month and day.
373     * These map onto the {@code YEAR}, {@code MONTH_OF_YEAR} and {@code DAY_OF_MONTH} fields.
374     * Note that there is no reference to eras.
375     * The full model for a date requires four concepts - era, year, month and day. These map onto
376     * the {@code ERA}, {@code YEAR_OF_ERA}, {@code MONTH_OF_YEAR} and {@code DAY_OF_MONTH} fields.
377     * Whether this field or {@code YEAR} is used depends on which mental model is being used.
378     * See {@link ChronoLocalDate} for more discussion on this topic.
379     * <p>
380     * In the default ISO calendar system, there are two eras defined, 'BCE' and 'CE'.
381     * The era 'CE' is the one currently in use and year-of-era runs from 1 to the maximum value.
382     * The era 'BCE' is the previous era, and the year-of-era runs backwards.
383     * <p>
384     * For example, subtracting a year each time yield the following:<br>
385     * - year-proleptic 2  = 'CE' year-of-era 2<br>
386     * - year-proleptic 1  = 'CE' year-of-era 1<br>
387     * - year-proleptic 0  = 'BCE' year-of-era 1<br>
388     * - year-proleptic -1 = 'BCE' year-of-era 2<br>
389     * <p>
390     * Note that the ISO-8601 standard does not actually define eras.
391     * Note also that the ISO eras do not align with the well-known AD/BC eras due to the
392     * change between the Julian and Gregorian calendar systems.
393     * <p>
394     * Non-ISO calendar systems should implement this field using the most recognized
395     * year-of-era value for users of the calendar system.
396     * Since most calendar systems have only two eras, the year-of-era numbering approach
397     * will typically be the same as that used by the ISO calendar system.
398     * The year-of-era value should typically always be positive, however this is not required.
399     */
400    YEAR_OF_ERA("YearOfEra", YEARS, FOREVER, ValueRange.of(1, Year.MAX_VALUE, Year.MAX_VALUE + 1)),
401    /**
402     * The proleptic year, such as 2012.
403     * <p>
404     * This represents the concept of the year, counting sequentially and using negative numbers.
405     * The proleptic year is not interpreted in terms of the era.
406     * See {@link #YEAR_OF_ERA} for an example showing the mapping from proleptic year to year-of-era.
407     * <p>
408     * The standard mental model for a date is based on three concepts - year, month and day.
409     * These map onto the {@code YEAR}, {@code MONTH_OF_YEAR} and {@code DAY_OF_MONTH} fields.
410     * Note that there is no reference to eras.
411     * The full model for a date requires four concepts - era, year, month and day. These map onto
412     * the {@code ERA}, {@code YEAR_OF_ERA}, {@code MONTH_OF_YEAR} and {@code DAY_OF_MONTH} fields.
413     * Whether this field or {@code YEAR_OF_ERA} is used depends on which mental model is being used.
414     * See {@link ChronoLocalDate} for more discussion on this topic.
415     * <p>
416     * Non-ISO calendar systems should implement this field as follows.
417     * If the calendar system has only two eras, before and after a fixed date, then the
418     * proleptic-year value must be the same as the year-of-era value for the later era,
419     * and increasingly negative for the earlier era.
420     * If the calendar system has more than two eras, then the proleptic-year value may be
421     * defined with any appropriate value, although defining it to be the same as ISO may be
422     * the best option.
423     */
424    YEAR("Year", YEARS, FOREVER, ValueRange.of(Year.MIN_VALUE, Year.MAX_VALUE)),
425    /**
426     * The era.
427     * <p>
428     * This represents the concept of the era, which is the largest division of the time-line.
429     * This field is typically used with {@link #YEAR_OF_ERA}.
430     * <p>
431     * In the default ISO calendar system, there are two eras defined, 'BCE' and 'CE'.
432     * The era 'CE' is the one currently in use and year-of-era runs from 1 to the maximum value.
433     * The era 'BCE' is the previous era, and the year-of-era runs backwards.
434     * See {@link #YEAR_OF_ERA} for a full example.
435     * <p>
436     * Non-ISO calendar systems should implement this field to define eras.
437     * The value of the era that was active on 1970-01-01 (ISO) must be assigned the value 1.
438     * Earlier eras must have sequentially smaller values.
439     * Later eras must have sequentially larger values,
440     */
441    ERA("Era", ERAS, FOREVER, ValueRange.of(0, 1)),
442    /**
443     * The instant epoch-seconds.
444     * <p>
445     * This represents the concept of the sequential count of seconds where
446     * 1970-01-01T00:00Z (ISO) is zero.
447     * This field may be used with {@link #NANO_OF_DAY} to represent the fraction of the day.
448     * <p>
449     * An {@link Instant} represents an instantaneous point on the time-line.
450     * On their own they have no elements which allow a local date-time to be obtained.
451     * Only when paired with an offset or time-zone can the local date or time be found.
452     * This field allows the seconds part of the instant to be queried.
453     * <p>
454     * This field is strictly defined to have the same meaning in all calendar systems.
455     * This is necessary to ensure interoperation between calendars.
456     */
457    INSTANT_SECONDS("InstantSeconds", SECONDS, FOREVER, ValueRange.of(Long.MIN_VALUE, Long.MAX_VALUE)),
458    /**
459     * The offset from UTC/Greenwich.
460     * <p>
461     * This represents the concept of the offset in seconds of local time from UTC/Greenwich.
462     * <p>
463     * A {@link ZoneOffset} represents the period of time that local time differs from UTC/Greenwich.
464     * This is usually a fixed number of hours and minutes.
465     * It is equivalent to the {@link ZoneOffset#getTotalSeconds() total amount} of the offset in seconds.
466     * For example, during the winter Paris has an offset of {@code +01:00}, which is 3600 seconds.
467     * <p>
468     * This field is strictly defined to have the same meaning in all calendar systems.
469     * This is necessary to ensure interoperation between calendars.
470     */
