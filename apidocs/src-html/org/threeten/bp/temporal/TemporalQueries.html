032package org.threeten.bp.temporal;
033
034import static org.threeten.bp.temporal.ChronoField.EPOCH_DAY;
035import static org.threeten.bp.temporal.ChronoField.NANO_OF_DAY;
036import static org.threeten.bp.temporal.ChronoField.OFFSET_SECONDS;
037
038import org.threeten.bp.LocalDate;
039import org.threeten.bp.LocalTime;
040import org.threeten.bp.ZoneId;
041import org.threeten.bp.ZoneOffset;
042import org.threeten.bp.chrono.Chronology;

044/**
045 * Common implementations of {@code TemporalQuery}.
046 * <p>
047 * This class provides common implementations of {@link TemporalQuery}.
048 * These queries are primarily used as optimizations, allowing the internals
049 * of other objects to be extracted effectively. Note that application code
050 * can also use the {@code from(TemporalAccessor)} method on most temporal
051 * objects as a method reference matching the query interface, such as
052 * {@code LocalDate::from} and {@code ZoneId::from}.
053 * <p>
054 * There are two equivalent ways of using a {@code TemporalQuery}.
055 * The first is to invoke the method on the interface directly.
056 * The second is to use {@link TemporalAccessor#query(TemporalQuery)}:
057 * <pre>
058 *   // these two lines are equivalent, but the second approach is recommended
059 *   dateTime = query.queryFrom(dateTime);
060 *   dateTime = dateTime.query(query);
061 * </pre>
062 * It is recommended to use the second approach, {@code query(TemporalQuery)},
063 * as it is a lot clearer to read in code.
064 *
065 * <h3>Specification for implementors</h3>
066 * This is a thread-safe utility class.
067 * All returned adjusters are immutable and thread-safe.
068 */
069public final class TemporalQueries {
070    // note that it is vital that each method supplies a constant, not a
071    // calculated value, as they will be checked for using ==
072    // it is also vital that each constant is different (due to the == checking)
073
074    /**
075     * Private constructor since this is a utility class.
076     */
077    private TemporalQueries() {
078    }
079
080    //-----------------------------------------------------------------------
081    // special constants should be used to extract information from a TemporalAccessor
082    // that cannot be derived in other ways
083    // Javadoc added here, so as to pretend they are more normal than they really are
084
085    /**
086     * A strict query for the {@code ZoneId}.
087     * <p>
088     * This queries a {@code TemporalAccessor} for the zone.
089     * The zone is only returned if the date-time conceptually contains a {@code ZoneId}.
090     * It will not be returned if the date-time only conceptually has an {@code ZoneOffset}.
091     * Thus a {@link ZonedDateTime} will return the result of
092     * {@code getZone()}, but an {@link OffsetDateTime} will
093     * return null.
094     * <p>
095     * In most cases, applications should use {@link #ZONE} as this query is too strict.
096     * <p>
097     * The result from JDK classes implementing {@code TemporalAccessor} is as follows:<br>
098     * {@code LocalDate} returns null<br>
099     * {@code LocalTime} returns null<br>
100     * {@code LocalDateTime} returns null<br>
101     * {@code ZonedDateTime} returns the associated zone<br>
102     * {@code OffsetTime} returns null<br>
103     * {@code OffsetDateTime} returns null<br>
104     * {@code ChronoLocalDate} returns null<br>
105     * {@code ChronoLocalDateTime} returns null<br>
106     * {@code ChronoZonedDateTime} returns the associated zone<br>
107     * {@code Era} returns null<br>
108     * {@code DayOfWeek} returns null<br>
109     * {@code Month} returns null<br>
110     * {@code Year} returns null<br>
111     * {@code YearMonth} returns null<br>
112     * {@code MonthDay} returns null<br>
113     * {@code ZoneOffset} returns null<br>
114     * {@code Instant} returns null<br>
115     *
116     * @return a query that can obtain the zone ID of a temporal, not null
117     */
118    public static final TemporalQuery<ZoneId> zoneId() {
119        return ZONE_ID;
120    }
121    static final TemporalQuery<ZoneId> ZONE_ID = new TemporalQuery<ZoneId>() {
122        @Override
123        public ZoneId queryFrom(TemporalAccessor temporal) {
124            return temporal.query(this);
125        }
126    };
127
128    /**
129     * A query for the {@code Chrono}.
130     * <p>
131     * This queries a {@code TemporalAccessor} for the chronology.
132     * If the target {@code TemporalAccessor} represents a date, or part of a date,
133     * then it should return the chronology that the date is expressed in.
134     * As a result of this definition, objects only representing time, such as
135     * {@code LocalTime}, will return null.
136     * <p>
137     * The result from JDK classes implementing {@code TemporalAccessor} is as follows:<br>
138     * {@code LocalDate} returns {@code IsoChronology.INSTANCE}<br>
139     * {@code LocalTime} returns null (does not represent a date)<br>
140     * {@code LocalDateTime} returns {@code IsoChronology.INSTANCE}<br>
141     * {@code ZonedDateTime} returns {@code IsoChronology.INSTANCE}<br>
142     * {@code OffsetTime} returns null (does not represent a date)<br>
143     * {@code OffsetDateTime} returns {@code IsoChronology.INSTANCE}<br>
144     * {@code ChronoLocalDate} returns the associated chronology<br>
145     * {@code ChronoLocalDateTime} returns the associated chronology<br>
146     * {@code ChronoZonedDateTime} returns the associated chronology<br>
147     * {@code Era} returns the associated chronology<br>
148     * {@code DayOfWeek} returns null (shared across chronologies)<br>
149     * {@code Month} returns {@code IsoChronology.INSTANCE}<br>
150     * {@code Year} returns {@code IsoChronology.INSTANCE}<br>
151     * {@code YearMonth} returns {@code IsoChronology.INSTANCE}<br>
152     * {@code MonthDay} returns null {@code IsoChronology.INSTANCE}<br>
153     * {@code ZoneOffset} returns null (does not represent a date)<br>
154     * {@code Instant} returns null (does not represent a date)<br>
155     * <p>
156     * The method {@link Chronology#from(TemporalAccessor)} can be used as a
157     * {@code TemporalQuery} via a method reference, {@code Chrono::from}.
158     * That method is equivalent to this query, except that it throws an
159     * exception if a chronology cannot be obtained.
160     *
161     * @return a query that can obtain the chronology of a temporal, not null
162     */
163    public static final TemporalQuery<Chronology> chronology() {
164        return CHRONO;
165    }
166    static final TemporalQuery<Chronology> CHRONO = new TemporalQuery<Chronology>() {
167        @Override
168        public Chronology queryFrom(TemporalAccessor temporal) {
169            return temporal.query(this);
170        }
171    };
172
173    /**
174     * A query for the smallest supported unit.
175     * <p>
176     * This queries a {@code TemporalAccessor} for the time precision.
177     * If the target {@code TemporalAccessor} represents a consistent or complete date-time,
178     * date or time then this must return the smallest precision actually supported.
179     * Note that fields such as {@code NANO_OF_DAY} and {@code NANO_OF_SECOND}
180     * are defined to always return ignoring the precision, thus this is the only
181     * way to find the actual smallest supported unit.
182     * For example, were {@code GregorianCalendar} to implement {@code TemporalAccessor}
183     * it would return a precision of {@code MILLIS}.
184     * <p>
185     * The result from JDK classes implementing {@code TemporalAccessor} is as follows:<br>
186     * {@code LocalDate} returns {@code DAYS}<br>
187     * {@code LocalTime} returns {@code NANOS}<br>
188     * {@code LocalDateTime} returns {@code NANOS}<br>
189     * {@code ZonedDateTime} returns {@code NANOS}<br>
190     * {@code OffsetTime} returns {@code NANOS}<br>
191     * {@code OffsetDateTime} returns {@code NANOS}<br>
192     * {@code ChronoLocalDate} returns {@code DAYS}<br>
193     * {@code ChronoLocalDateTime} returns {@code NANOS}<br>
194     * {@code ChronoZonedDateTime} returns {@code NANOS}<br>
195     * {@code Era} returns {@code ERAS}<br>
196     * {@code DayOfWeek} returns {@code DAYS}<br>
197     * {@code Month} returns {@code MONTHS}<br>
198     * {@code Year} returns {@code YEARS}<br>
199     * {@code YearMonth} returns {@code MONTHS}<br>
200     * {@code MonthDay} returns null (does not represent a complete date or time)<br>
201     * {@code ZoneOffset} returns null (does not represent a date or time)<br>
202     * {@code Instant} returns {@code NANOS}<br>
203     *
204     * @return a query that can obtain the precision of a temporal, not null
205     */
206    public static final TemporalQuery<TemporalUnit> precision() {
207        return PRECISION;
208    }
209    static final TemporalQuery<TemporalUnit> PRECISION = new TemporalQuery<TemporalUnit>() {
210        @Override
211        public TemporalUnit queryFrom(TemporalAccessor temporal) {
212            return temporal.query(this);
213        }
214    };
215
216    //-----------------------------------------------------------------------
217    // non-special constants are standard queries that derive information from other information
218    /**
219     * A lenient query for the {@code ZoneId}, falling back to the {@code ZoneOffset}.
220     * <p>
221     * This queries a {@code TemporalAccessor} for the zone.
222     * It first tries to obtain the zone, using {@link #zoneId()}.
223     * If that is not found it tries to obtain the {@link #offset()}.
224     * <p>
225     * In most cases, applications should use this query rather than {@code #zoneId()}.
226     * <p>
227     * This query examines the {@link ChronoField#OFFSET_SECONDS offset-seconds}
228     * field and uses it to create a {@code ZoneOffset}.
229     * <p>
230     * The method {@link ZoneId#from(TemporalAccessor)} can be used as a
231     * {@code TemporalQuery} via a method reference, {@code ZoneId::from}.
232     * That method is equivalent to this query, except that it throws an
233     * exception if a zone cannot be obtained.
234     *
235     * @return a query that can obtain the zone ID or offset of a temporal, not null
236     */
237    public static final TemporalQuery<ZoneId> zone() {
238        return ZONE;
239    }
240    static final TemporalQuery<ZoneId> ZONE = new TemporalQuery<ZoneId>() {
241        @Override
242        public ZoneId queryFrom(TemporalAccessor temporal) {
243            ZoneId zone = temporal.query(ZONE_ID);
244            return (zone != null ? zone : temporal.query(OFFSET));
245        }
246    };
247
248    /**
249     * A query for {@code ZoneOffset} returning null if not found.
250     * <p>
251     * This returns a {@code TemporalQuery} that can be used to query a temporal
252     * object for the offset. The query will return null if the temporal
253     * object cannot supply an offset.
254     * <p>
255     * The query implementation examines the {@link ChronoField#OFFSET_SECONDS OFFSET_SECONDS}
256     * field and uses it to create a {@code ZoneOffset}.
257     *
258     * @return a query that can obtain the offset of a temporal, not null
259     */
260    public static final TemporalQuery<ZoneOffset> offset() {
261        return OFFSET;
262    }
263    static final TemporalQuery<ZoneOffset> OFFSET = new TemporalQuery<ZoneOffset>() {
264        @Override
265        public ZoneOffset queryFrom(TemporalAccessor temporal) {
266            if (temporal.isSupported(OFFSET_SECONDS)) {
267                return ZoneOffset.ofTotalSeconds(temporal.get(OFFSET_SECONDS));
268            }
269            return null;
270        }
271    };
272
273    /**
274     * A query for {@code LocalDate} returning null if not found.
275     * <p>
276     * This returns a {@code TemporalQuery} that can be used to query a temporal
277     * object for the local date. The query will return null if the temporal
278     * object cannot supply a local date.
279     * <p>
280     * The query implementation examines the {@link ChronoField#EPOCH_DAY EPOCH_DAY}
281     * field and uses it to create a {@code LocalDate}.
282     *
283     * @return a query that can obtain the date of a temporal, not null
284     */
285    public static final TemporalQuery<LocalDate> localDate() {
286        return LOCAL_DATE;
287    }
288    static final TemporalQuery<LocalDate> LOCAL_DATE = new TemporalQuery<LocalDate>() {
289        @Override
290        public LocalDate queryFrom(TemporalAccessor temporal) {
291            if (temporal.isSupported(EPOCH_DAY)) {
292                return LocalDate.ofEpochDay(temporal.getLong(EPOCH_DAY));
293            }
294            return null;
295        }
296    };
297
298    /**
299     * A query for {@code LocalTime} returning null if not found.
300     * <p>
301     * This returns a {@code TemporalQuery} that can be used to query a temporal
302     * object for the local time. The query will return null if the temporal
303     * object cannot supply a local time.
304     * <p>
305     * The query implementation examines the {@link ChronoField#NANO_OF_DAY NANO_OF_DAY}
306     * field and uses it to create a {@code LocalTime}.
307     *
308     * @return a query that can obtain the date of a temporal, not null
309     */
310    public static final TemporalQuery<LocalTime> localTime() {
311        return LOCAL_TIME;
312    }
313    static final TemporalQuery<LocalTime> LOCAL_TIME = new TemporalQuery<LocalTime>() {
314        @Override
315        public LocalTime queryFrom(TemporalAccessor temporal) {
316            if (temporal.isSupported(NANO_OF_DAY)) {
317                return LocalTime.ofNanoOfDay(temporal.getLong(NANO_OF_DAY));
318            }
319            return null;
320        }
321    };
