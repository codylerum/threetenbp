032package org.threeten.bp.temporal;
033
034import org.threeten.bp.DateTimeException;
035import org.threeten.bp.Duration;
036import org.threeten.bp.Period;
037
038/**
039 * A unit of date-time, such as Days or Hours.
040 * <p>
041 * Measurement of time is built on units, such as years, months, days, hours, minutes and seconds.
042 * Implementations of this interface represent those units.
043 * <p>
044 * An instance of this interface represents the unit itself, rather than an amount of the unit.
045 * See {@link Period} for a class that represents an amount in terms of the common units.
046 * <p>
047 * The most commonly used units are defined in {@link ChronoUnit}.
048 * Further units are supplied in {@link IsoFields}.
049 * Units can also be written by application code by implementing this interface.
050 * <p>
051 * The unit works using double dispatch. Client code calls methods on a date-time like
052 * {@code LocalDateTime} which check if the unit is a {@code ChronoUnit}.
053 * If it is, then the date-time must handle it.
054 * Otherwise, the method call is re-dispatched to the matching method in this interface.
055 *
056 * <h3>Specification for implementors</h3>
057 * This interface must be implemented with care to ensure other classes operate correctly.
058 * All implementations that can be instantiated must be final, immutable and thread-safe.
059 * It is recommended to use an enum where possible.
060 */
061public interface TemporalUnit {
062
063    /**
064     * Gets a descriptive name for the unit.
065     * <p>
066     * This should be in the plural and upper-first camel case, such as 'Days' or 'Minutes'.
067     *
068     * @return the name, not null
069     */
070    String getName();
071
072    /**
073     * Gets the duration of this unit, which may be an estimate.
074     * <p>
075     * All units return a duration measured in standard nanoseconds from this method.
076     * The duration will be positive and non-zero.
077     * For example, an hour has a duration of {@code 60 * 60 * 1,000,000,000ns}.
078     * <p>
079     * Some units may return an accurate duration while others return an estimate.
080     * For example, days have an estimated duration due to the possibility of
081     * daylight saving time changes.
082     * To determine if the duration is an estimate, use {@link #isDurationEstimated()}.
083     *
084     * @return the duration of this unit, which may be an estimate, not null
085     */
086    Duration getDuration();
087
088    /**
089     * Checks if the duration of the unit is an estimate.
090     * <p>
091     * All units have a duration, however the duration is not always accurate.
092     * For example, days have an estimated duration due to the possibility of
093     * daylight saving time changes.
094     * This method returns true if the duration is an estimate and false if it is
095     * accurate. Note that accurate/estimated ignores leap seconds.
096     *
097     * @return true if the duration is estimated, false if accurate
098     */
099    boolean isDurationEstimated();
100
101    //-----------------------------------------------------------------------
102    /**
103     * Checks if this unit is supported by the specified temporal object.
104     * <p>
105     * This checks that the implementing date-time can add/subtract this unit.
106     * This can be used to avoid throwing an exception.
107     *
108     * @param temporal  the temporal object to check, not null
109     * @return true if the unit is supported
110     */
111    boolean isSupportedBy(Temporal temporal);
112
113    /**
114     * Returns a copy of the specified temporal object with the specified period added.
115     * <p>
116     * The period added is a multiple of this unit. For example, this method
117     * could be used to add "3 days" to a date by calling this method on the
118     * instance representing "days", passing the date and the period "3".
119     * The period to be added may be negative, which is equivalent to subtraction.
120     * <p>
121     * There are two equivalent ways of using this method.
122     * The first is to invoke this method directly.
123     * The second is to use {@link Temporal#plus(long, TemporalUnit)}:
124     * <pre>
125     *   // these two lines are equivalent, but the second approach is recommended
126     *   temporal = thisUnit.doPlus(temporal);
127     *   temporal = temporal.plus(thisUnit);
128     * </pre>
129     * It is recommended to use the second approach, {@code plus(TemporalUnit)},
130     * as it is a lot clearer to read in code.
131     * <p>
132     * Implementations should perform any queries or calculations using the units
133     * available in {@link ChronoUnit} or the fields available in {@link ChronoField}.
134     * If the field is not supported a {@code DateTimeException} must be thrown.
135     * <p>
136     * Implementations must not alter the specified temporal object.
137     * Instead, an adjusted copy of the original must be returned.
138     * This provides equivalent, safe behavior for immutable and mutable implementations.
139     *
140     * @param <R>  the type of the Temporal object
141     * @param dateTime  the temporal object to adjust, not null
142     * @param periodToAdd  the period of this unit to add, positive or negative
143     * @return the adjusted temporal object, not null
144     * @throws DateTimeException if the period cannot be added
145     */
146    <R extends Temporal> R addTo(R dateTime, long periodToAdd);
147
148    //-----------------------------------------------------------------------
149    /**
150     * Calculates the period in terms of this unit between two temporal objects of the same type.
151     * <p>
152     * This calculates the period between two temporals in terms of this unit.
153     * The start and end points are supplied as temporal objects and must be of the same type.
154     * The result will be negative if the end is before the start.
155     * For example, the period in hours between two temporal objects can be calculated
156     * using {@code HOURS.between(startTime, endTime)}.
157     * <p>
158     * The calculation returns a whole number, representing the number of complete units between the two temporals.
159     * For example, the period in hours between the times 11:30 and 13:29 will only b
160     * one hour as it is one minute short of two hours.
161     * <p>
162     * There are two equivalent ways of using this method.
163     * The first is to invoke this method directly.
164     * The second is to use {@link Temporal#periodUntil(Temporal, TemporalUnit)}:
165     * <pre>
166     *   // these two lines are equivalent
167     *   between = thisUnit.between(start, end);
168     *   between = start.periodUntil(end, thisUnit);
169     * </pre>
170     * The choice should be made based on which makes the code more readable. 
171     * <p>
172     * For example, this method allows the number of days between two dates to be calculated:
173     * <pre>
174     *   long daysBetween = DAYS.between(start, end);
175     *   // or alternatively
176     *   long daysBetween = start.periodUntil(end, DAYS);
177     * </pre>
178     * Implementations should perform any queries or calculations using the units available in
179     * {@link ChronoUnit} or the fields available in {@link ChronoField}.
180     * If the unit is not supported a DateTimeException must be thrown.
181     * Implementations must not alter the specified temporal objects.
182     *
183     * @param temporal1  the base temporal object, not null
184     * @param temporal2  the other temporal object, not null
185     * @return the period between temporal1 and temporal2 in terms of this unit;
186     *  positive if temporal2 is later than temporal1, negative if earlier
187     * @throws DateTimeException if the period cannot be calculated
188     * @throws ArithmeticException if numeric overflow occurs
189     */
190    long between(Temporal temporal1, Temporal temporal2);
191
192    //-----------------------------------------------------------------------
193    /**
194     * Outputs this unit as a {@code String} using the name.
195     *
196     * @return the name of this unit, not null
197     */
198    @Override
199    String toString();
200
201}
