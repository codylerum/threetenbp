032package org.threeten.bp.temporal;
033
034import static org.threeten.bp.temporal.ChronoField.EPOCH_DAY;
035import static org.threeten.bp.temporal.ChronoUnit.DAYS;
036import static org.threeten.bp.temporal.ChronoUnit.FOREVER;
037
038import java.util.Collections;
039import java.util.Map;
040
041import org.threeten.bp.DateTimeException;
042import org.threeten.bp.jdk8.Jdk8Methods;
043
044/**
045 * A set of date fields that provide access to Julian Days.
046 * <p>
047 * The Julian Day is a standard way of expressing date and time commonly used in the scientific community.
048 * It is expressed as a decimal number of whole days where days start at midday.
049 * This class represents variations on Julian Days that count whole days from midnight.
050 *
051 * <h3>Specification for implementors</h3>
052 * This is an immutable and thread-safe class.
053 */
054public final class JulianFields {
055
056    /**
057     * Julian Day field.
058     * <p>
059     * This is an integer-based version of the Julian Day Number.
060     * Julian Day is a well-known system that represents the count of whole days since day 0,
061     * which is defined to be January 1, 4713 BCE in the Julian calendar, and -4713-11-24 Gregorian.
062     * The field  has "JulianDay" as 'name', and 'DAYS' as 'baseUnit'.
063     * The field always refers to the local date-time, ignoring the offset or zone.
064     * <p>
065     * For date-times, 'JULIAN_DAY.getFrom()' assumes the same value from
066     * midnight until just before the next midnight.
067     * When 'JULIAN_DAY.adjustInto()' is applied to a date-time, the time of day portion remains unaltered.
068     * 'JULIAN_DAY.adjustInto()' and 'JULIAN_DAY.getFrom()' only apply to {@code Temporal} objects that
069     * can be converted into {@link ChronoField#EPOCH_DAY}.
070     * A {@link DateTimeException} is thrown for any other type of object.
071     * <p>
072     * <h3>Astronomical and Scientific Notes</h3>
073     * The standard astronomical definition uses a fraction to indicate the time-of-day,
074     * thus 3.25 would represent the time 18:00, since days start at midday.
075     * This implementation uses an integer and days starting at midnight.
076     * The integer value for the Julian Day Number is the astronomical Julian Day value at midday
077     * of the date in question.
078     * This amounts to the astronomical Julian Day, rounded to an integer {@code JDN = floor(JD + 0.5)}.
079     * <p>
080     * <pre>
081     *  | ISO date          |  Julian Day Number | Astronomical Julian Day |
082     *  | 1970-01-01T00:00  |         2,440,588  |         2,440,587.5     |
083     *  | 1970-01-01T06:00  |         2,440,588  |         2,440,587.75    |
084     *  | 1970-01-01T12:00  |         2,440,588  |         2,440,588.0     |
085     *  | 1970-01-01T18:00  |         2,440,588  |         2,440,588.25    |
086     *  | 1970-01-02T00:00  |         2,440,589  |         2,440,588.5     |
087     *  | 1970-01-02T06:00  |         2,440,589  |         2,440,588.75    |
088     *  | 1970-01-02T12:00  |         2,440,589  |         2,440,589.0     |
089     * </pre>
090     * <p>
091     * Julian Days are sometimes taken to imply Universal Time or UTC, but this
092     * implementation always uses the Julian Day number for the local date,
093     * regardless of the offset or time-zone.
094     */
095    public static final TemporalField JULIAN_DAY = Field.JULIAN_DAY;
096    /**
097     * Modified Julian Day field.
098     * <p>
099     * This is an integer-based version of the Modified Julian Day Number.
100     * Modified Julian Day (MJD) is a well-known system that counts days continuously.
101     * It is defined relative to astronomical Julian Day as  {@code MJD = JD - 2400000.5}.
102     * Each Modified Julian Day runs from midnight to midnight.
103     * The field always refers to the local date-time, ignoring the offset or zone.
104     * <p>
105     * For date-times, 'MODIFIED_JULIAN_DAY.getFrom()' assumes the same value from
106     * midnight until just before the next midnight.
107     * When 'MODIFIED_JULIAN_DAY.adjustInto()' is applied to a date-time, the time of day portion remains unaltered.
108     * 'MODIFIED_JULIAN_DAY.adjustInto()' and 'MODIFIED_JULIAN_DAY.getFrom()' only apply to {@code Temporal} objects
109     * that can be converted into {@link ChronoField#EPOCH_DAY}.
110     * A {@link DateTimeException} is thrown for any other type of object.
111     * <p>
112     * This implementation is an integer version of MJD with the decimal part rounded to floor.
113     * <p>
114     * <h3>Astronomical and Scientific Notes</h3>
115     * <pre>
116     *  | ISO date          | Modified Julian Day |      Decimal MJD |
117     *  | 1970-01-01T00:00  |             40,587  |       40,587.0   |
118     *  | 1970-01-01T06:00  |             40,587  |       40,587.25  |
119     *  | 1970-01-01T12:00  |             40,587  |       40,587.5   |
120     *  | 1970-01-01T18:00  |             40,587  |       40,587.75  |
121     *  | 1970-01-02T00:00  |             40,588  |       40,588.0   |
122     *  | 1970-01-02T06:00  |             40,588  |       40,588.25  |
123     *  | 1970-01-02T12:00  |             40,588  |       40,588.5   |
124     * </pre>
125     * <p>
126     * Modified Julian Days are sometimes taken to imply Universal Time or UTC, but this
127     * implementation always uses the Modified Julian Day for the local date,
128     * regardless of the offset or time-zone.
129     */
130    public static final TemporalField MODIFIED_JULIAN_DAY = Field.MODIFIED_JULIAN_DAY;
131    /**
132     * Rata Die field.
133     * <p>
134     * Rata Die counts whole days continuously starting day 1 at midnight at the beginning of 0001-01-01 (ISO).
135     * The field always refers to the local date-time, ignoring the offset or zone.
136     * <p>
137     * For date-times, 'RATA_DIE.getFrom()' assumes the same value from
138     * midnight until just before the next midnight.
139     * When 'RATA_DIE.adjustInto()' is applied to a date-time, the time of day portion remains unaltered.
140     * 'MODIFIED_JULIAN_DAY.adjustInto()' and 'RATA_DIE.getFrom()' only apply to {@code Temporal} objects
141     * that can be converted into {@link ChronoField#EPOCH_DAY}.
142     * A {@link DateTimeException} is thrown for any other type of object.
143     */
144    public static final TemporalField RATA_DIE = Field.RATA_DIE;
145
146    /**
147     * Hidden implementation.
148     */
149    private static enum Field implements TemporalField {
150        /**
151         * Julian Day field.
152         */
153        // 719163L + 1721425L = 2440588L
154        JULIAN_DAY("JulianDay", DAYS, FOREVER, 2440588L),
155        /**
156         * Modified Julian Day field.
157         */
158        // 719163L - 678576L = 40587L
159        MODIFIED_JULIAN_DAY("ModifiedJulianDay", DAYS, FOREVER, 40587L),
160        /**
161         * Rata Die field.
162         */
163        RATA_DIE("RataDie", DAYS, FOREVER, 719163L),
164        // lots of others Truncated,Lilian, ANSI COBOL (also dotnet related), Excel?
165        ;
166
167        private final String name;
168        private final TemporalUnit baseUnit;
169        private final TemporalUnit rangeUnit;
170        private final ValueRange range;
171        private final long offset;
172
173        private Field(String name, TemporalUnit baseUnit, TemporalUnit rangeUnit, long offset) {
174            this.name = name;
175            this.baseUnit = baseUnit;
176            this.rangeUnit = rangeUnit;
177            this.range = ValueRange.of(-365243219162L + offset, 365241780471L + offset);
178            this.offset = offset;
179        }
180
181        //-----------------------------------------------------------------------
182        @Override
183        public String getName() {
184            return name;
185        }
186
187        @Override
188        public TemporalUnit getBaseUnit() {
189            return baseUnit;
190        }
191
192        @Override
193        public TemporalUnit getRangeUnit() {
194            return rangeUnit;
195        }
196
197        @Override
198        public ValueRange range() {
199            return range;
200        }
201
202        @Override
203        public int compare(TemporalAccessor temporal1, TemporalAccessor temporal2) {
204            return Long.compare(temporal1.getLong(this), temporal2.getLong(this));
205        }
206
207        //-----------------------------------------------------------------------
208        @Override
209        public boolean isSupportedBy(TemporalAccessor temporal) {
210            return temporal.isSupported(EPOCH_DAY);
211        }
212
213        @Override
214        public ValueRange rangeRefinedBy(TemporalAccessor temporal) {
215            if (isSupportedBy(temporal) == false) {
216                throw new DateTimeException("Unsupported field: " + this);
217            }
218            return range();
219        }
220
221        @Override
222        public long getFrom(TemporalAccessor temporal) {
223            return temporal.getLong(EPOCH_DAY) + offset;
224        }
225
226        @Override
227        public <R extends Temporal> R adjustInto(R dateTime, long newValue) {
228            if (range().isValidValue(newValue) == false) {
229                throw new DateTimeException("Invalid value: " + name + " " + newValue);
230            }
231            return (R) dateTime.with(EPOCH_DAY, Jdk8Methods.safeSubtract(newValue, offset));
232        }
233
234        //-----------------------------------------------------------------------
235        @Override
236        public Map<TemporalField, Long> resolve(TemporalAccessor temporal, long value) {
237            return Collections.<TemporalField, Long>singletonMap(EPOCH_DAY, Jdk8Methods.safeSubtract(value, offset));
238        }
239
240        //-----------------------------------------------------------------------
241        @Override
242        public String toString() {
243            return getName();
244        }
245
246    }
247}
