066/**
067 * A date-time with an offset from UTC/Greenwich in the ISO-8601 calendar system,
068 * such as {@code 2007-12-03T10:15:30+01:00}.
069 * <p>
070 * {@code OffsetDateTime} is an immutable representation of a date-time with an offset.
071 * This class stores all date and time fields, to a precision of nanoseconds,
072 * as well as the offset from UTC/Greenwich. For example, the value
073 * "2nd October 2007 at 13:45.30.123456789 +02:00" can be stored in an {@code OffsetDateTime}.
074 * <p>
075 * {@code OffsetDateTime}, {@link ZonedDateTime} and {@link Instant} all store an instant
076 * on the time-line to nanosecond precision.
077 * {@code Instant} is the simplest, simply representing the instant.
078 * {@code OffsetDateTime} adds to the instant the offset from UTC/Greenwich, which allows
079 * the local date-time to be obtained.
080 * {@code ZonedDateTime} adds full time-zone rules.
081 * <p>
082 * It is intended that {@code ZonedDateTime} or {@code Instant} is used to model data
083 * in simpler applications. This class may be used when modeling date-time concepts in
084 * more detail, or when communicating to a database or in a network protocol.
085 *
086 * <h3>Specification for implementors</h3>
087 * This class is immutable and thread-safe.
088 */
089public final class OffsetDateTime
090        extends DefaultInterfaceTemporal
091        implements Temporal, TemporalAdjuster, Comparable<OffsetDateTime>, Serializable {

094     * The minimum supported {@code OffsetDateTime}, '-999999999-01-01T00:00:00+18:00'.
095     * This is the local date-time of midnight at the start of the minimum date
096     * in the maximum offset (larger offsets are earlier on the time-line).
097     * This combines {@link LocalDateTime#MIN} and {@link ZoneOffset#MAX}.
098     * This could be used by an application as a "far past" date-time.

103     * This is the local date-time just before midnight at the end of the maximum date
104     * in the minimum offset (larger negative offsets are later on the time-line).
105     * This combines {@link LocalDateTime#MAX} and {@link ZoneOffset#MIN}.
106     * This could be used by an application as a "far future" date-time.

110    /**
111     * Comparator for two {@code OffsetDateTime} instances based solely on the instant.
112     * <p>
113     * This method differs from the comparison in {@link #compareTo} in that it
114     * only compares the underlying instant.
115     *
116     * @see #isAfter
117     * @see #isBefore
118     * @see #isEqual
119     */
120    public static final Comparator<OffsetDateTime> INSTANT_COMPARATOR = new Comparator<OffsetDateTime>() {
121        @Override
122        public int compare(OffsetDateTime datetime1, OffsetDateTime datetime2) {
123            int cmp = Long.compare(datetime1.toEpochSecond(), datetime2.toEpochSecond());
124            if (cmp == 0) {
125                cmp = Long.compare(datetime1.toLocalTime().toNanoOfDay(), datetime2.toLocalTime().toNanoOfDay());
126            }
127            return cmp;
128        }
129    };

259     * Obtains an instance of {@code OffsetDateTime} from an {@code Instant} and zone ID.
260     * <p>
261     * This creates an offset date-time with the same instant as that specified.
262     * Finding the offset from UTC/Greenwich is simple as there is only one valid
263     * offset for each instant.
264     *
265     * @param instant  the instant to create the date-time from, not null
266     * @param zone  the time-zone, which may be an offset, not null
267     * @return the offset date-time, not null
268     * @throws DateTimeException if the result exceeds the supported range
269     */
270    public static OffsetDateTime ofInstant(Instant instant, ZoneId zone) {
271        Objects.requireNonNull(instant, "instant");
272        Objects.requireNonNull(zone, "zone");
273        ZoneRules rules = zone.getRules();
274        ZoneOffset offset = rules.getOffset(instant);
275        LocalDateTime ldt = LocalDateTime.ofEpochSecond(instant.getEpochSecond(), instant.getNano(), offset);
276        return new OffsetDateTime(ldt, offset);
277    }

280    /**
281     * Obtains an instance of {@code OffsetDateTime} from a temporal object.
282     * <p>
283     * A {@code TemporalAccessor} represents some form of date and time information.
284     * This factory converts the arbitrary temporal object to an instance of {@code OffsetDateTime}.
285     * <p>
286     * The conversion extracts and combines {@code LocalDateTime} and {@code ZoneOffset}.
287     * If that fails it will try to extract and combine {@code Instant} and {@code ZoneOffset}.
288     * <p>
289     * This method matches the signature of the functional interface {@link TemporalQuery}
290     * allowing it to be used in queries via method reference, {@code OffsetDateTime::from}.
291     *
292     * @param temporal  the temporal object to convert, not null
293     * @return the offset date-time, not null
294     * @throws DateTimeException if unable to convert to an {@code OffsetDateTime}
295     */
296    public static OffsetDateTime from(TemporalAccessor temporal) {
297        if (temporal instanceof OffsetDateTime) {
298            return (OffsetDateTime) temporal;
299        }
300        ZoneOffset offset = ZoneOffset.from(temporal);
301        try {
302            try {
303                LocalDateTime ldt = LocalDateTime.from(temporal);
304                return OffsetDateTime.of(ldt, offset);
305            } catch (DateTimeException ignore) {
306                Instant instant = Instant.from(temporal);
307                return OffsetDateTime.ofInstant(instant, offset);
308            }
309        } catch (DateTimeException ex) {
310            throw new DateTimeException("Unable to obtain OffsetDateTime from TemporalAccessor: " + temporal.getClass(), ex);
311        }
312    }

391     * <li>{@code HOUR_OF_AMPM}
392     * <li>{@code CLOCK_HOUR_OF_AMPM}
393     * <li>{@code HOUR_OF_DAY}
394     * <li>{@code CLOCK_HOUR_OF_DAY}
395     * <li>{@code AMPM_OF_DAY}
396     * <li>{@code DAY_OF_WEEK}
397     * <li>{@code ALIGNED_DAY_OF_WEEK_IN_MONTH}
398     * <li>{@code ALIGNED_DAY_OF_WEEK_IN_YEAR}
399     * <li>{@code DAY_OF_MONTH}
400     * <li>{@code DAY_OF_YEAR}

431     * This date-time is used to enhance the accuracy of the returned range.
432     * If it is not possible to return the range, because the field is not supported

461     * Gets the value of the specified field from this date-time as an {@code int}.
462     * <p>
463     * This queries this date-time for the value for the specified field.
464     * The returned value will always be within the valid range of values for the field.
465     * If it is not possible to return the value, because the field is not supported
466     * or for some other reason, an exception is thrown.
467     * <p>
468     * If the field is a {@link ChronoField} then the query is implemented here.
469     * The {@link #isSupported(TemporalField) supported fields} will return valid
470     * values based on this date-time, except {@code NANO_OF_DAY}, {@code MICRO_OF_DAY},
471     * {@code EPOCH_DAY}, {@code EPOCH_MONTH} and {@code INSTANT_SECONDS} which are too
472     * large to fit in an {@code int} and throw a {@code DateTimeException}.
473     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
474     * <p>
475     * If the field is not a {@code ChronoField}, then the result of this method
476     * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
477     * passing {@code this} as the argument. Whether the value can be obtained,
478     * and what the value represents, is determined by the field.
479     *
480     * @param field  the field to get, not null
481     * @return the value for the field
482     * @throws DateTimeException if a value for the field cannot be obtained
483     * @throws ArithmeticException if numeric overflow occurs
484     */
485    @Override
486    public int get(TemporalField field) {
487        if (field instanceof ChronoField) {
488            switch ((ChronoField) field) {
489                case INSTANT_SECONDS: throw new DateTimeException("Field too large for an int: " + field);
490                case OFFSET_SECONDS: return getOffset().getTotalSeconds();
491            }
492            return dateTime.get(field);
493        }
494        return super.get(field);
495    }

713    /**
714     * Returns an adjusted copy of this date-time.
715     * <p>
716     * This returns a new {@code OffsetDateTime}, based on this one, with the date-time adjusted.
717     * The adjustment takes place using the specified adjuster strategy object.
718     * Read the documentation of the adjuster to understand what adjustment will be made.
719     * <p>
720     * A simple adjuster might simply set the one of the fields, such as the year field.
721     * A more complex adjuster might set the date to the last day of the month.
722     * A selection of common adjustments is provided in {@link TemporalAdjusters}.
723     * These include finding the "last day of the month" and "next Wednesday".
724     * Key date-time classes also implement the {@code TemporalAdjuster} interface,
725     * such as {@link Month} and {@link MonthDay}.
726     * The adjuster is responsible for handling special cases, such as the varying
727     * lengths of month and leap years.
728     * <p>
729     * For example this code returns a date on the last day of July:
730     * <pre>
731     *  import static org.threeten.bp.Month.*;
732     *  import static org.threeten.bp.temporal.Adjusters.*;
733     *
734     *  result = offsetDateTime.with(JULY).with(lastDayOfMonth());
735     * </pre>
736     * <p>
737     * The classes {@link LocalDate}, {@link LocalTime} and {@link ZoneOffset} implement
738     * {@code TemporalAdjuster}, thus this method can be used to change the date, time or offset:
739     * <pre>
740     *  result = offsetDateTime.with(date);
741     *  result = offsetDateTime.with(time);
742     *  result = offsetDateTime.with(offset);
743     * </pre>
744     * <p>
745     * The result of this method is obtained by invoking the
746     * {@link TemporalAdjuster#adjustInto(Temporal)} method on the
747     * specified adjuster passing {@code this} as the argument.
748     * <p>
749     * This instance is immutable and unaffected by this method call.
750     *
751     * @param adjuster the adjuster to use, not null
752     * @return an {@code OffsetDateTime} based on {@code this} with the adjustment made, not null
753     * @throws DateTimeException if the adjustment cannot be made
754     * @throws ArithmeticException if numeric overflow occurs
755     */
756    @Override
757    public OffsetDateTime with(TemporalAdjuster adjuster) {
758        // optimizations
759        if (adjuster instanceof LocalDate || adjuster instanceof LocalTime || adjuster instanceof LocalDateTime) {
760            return with(dateTime.with(adjuster), offset);
761        } else if (adjuster instanceof Instant) {
762            return ofInstant((Instant) adjuster, offset);
763        } else if (adjuster instanceof ZoneOffset) {
764            return with(dateTime, (ZoneOffset) adjuster);
765        } else if (adjuster instanceof OffsetDateTime) {
766            return (OffsetDateTime) adjuster;
767        }
768        return (OffsetDateTime) adjuster.adjustInto(this);
769    }

771    /**
772     * Returns a copy of this date-time with the specified field set to a new value.
773     * <p>
774     * This returns a new {@code OffsetDateTime}, based on this one, with the value
775     * for the specified field changed.
776     * This can be used to change any supported field, such as the year, month or day-of-month.
777     * If it is not possible to set the value, because the field is not supported or for
778     * some other reason, an exception is thrown.
779     * <p>
780     * In some cases, changing the specified field can cause the resulting date-time to become invalid,
781     * such as changing the month from 31st January to February would make the day-of-month invalid.
782     * In cases like this, the field is responsible for resolving the date. Typically it will choose
783     * the previous valid date, which would be the last valid day of February in this example.
784     * <p>
785     * If the field is a {@link ChronoField} then the adjustment is implemented here.
786     * <p>
787     * The {@code INSTANT_SECONDS} field will return a date-time with the specified instant.
788     * The offset and nano-of-second are unchanged.
789     * If the new instant value is outside the valid range then a {@code DateTimeException} will be thrown.
790     * <p>
791     * The {@code OFFSET_SECONDS} field will return a date-time with the specified offset.
792     * The local date-time is unaltered. If the new offset value is outside the valid range
793     * then a {@code DateTimeException} will be thrown.
794     * <p>
795     * The other {@link #isSupported(TemporalField) supported fields} will behave as per
796     * the matching method on {@link LocalDateTime#with(TemporalField, long) LocalDateTime}.
797     * In this case, the offset is not part of the calculation and will be unchanged.
798     * <p>
799     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
800     * <p>
801     * If the field is not a {@code ChronoField}, then the result of this method
802     * is obtained by invoking {@code TemporalField.adjustInto(Temporal, long)}
803     * passing {@code this} as the argument. In this case, the field determines
804     * whether and how to adjust the instant.
805     * <p>
806     * This instance is immutable and unaffected by this method call.
807     *
808     * @param field  the field to set in the result, not null
809     * @param newValue  the new value of the field in the result
810     * @return an {@code OffsetDateTime} based on {@code this} with the specified field set, not null
811     * @throws DateTimeException if the field cannot be set
812     * @throws ArithmeticException if numeric overflow occurs
813     */
814    @Override
815    public OffsetDateTime with(TemporalField field, long newValue) {
816        if (field instanceof ChronoField) {
817            ChronoField f = (ChronoField) field;
818            switch (f) {
819                case INSTANT_SECONDS: return ofInstant(Instant.ofEpochSecond(newValue, getNano()), offset);
820                case OFFSET_SECONDS: {
821                    return with(dateTime, ZoneOffset.ofTotalSeconds(f.checkValidIntValue(newValue)));
822                }
823            }
824            return with(dateTime.with(field, newValue), offset);
825        }
826        return field.adjustInto(this, newValue);
827    }

1412    /**
1413     * Calculates the period between this date-time and another date-time in
1414     * terms of the specified unit.
1415     * <p>
1416     * This calculates the period between two date-times in terms of a single unit.
1417     * The start and end points are {@code this} and the specified date-time.
1418     * The result will be negative if the end is before the start.
1419     * For example, the period in days between two date-times can be calculated
1420     * using {@code startDateTime.periodUntil(endDateTime, DAYS)}.
1421     * <p>
1422     * The {@code Temporal} passed to this method must be an {@code OffsetDateTime}.
1423     * If the offset differs between the two date-times, the specified
1424     * end date-time is normalized to have the same offset as this date-time.
1425     * <p>
1426     * The calculation returns a whole number, representing the number of
1427     * complete units between the two date-times.
1428     * For example, the period in months between 2012-06-15T00:00Z and 2012-08-14T23:59Z
1429     * will only be one month as it is one minute short of two months.
1430     * <p>
1431     * This method operates in association with {@link TemporalUnit#between}.
1432     * The result of this method is a {@code long} representing the amount of
1433     * the specified unit. By contrast, the result of {@code between} is an
1434     * object that can be used directly in addition/subtraction:
1435     * <pre>
1436     *   long period = start.periodUntil(end, MONTHS);   // this method
1437     *   dateTime.plus(MONTHS.between(start, end));      // use in plus/minus
1438     * </pre>
1439     * <p>
1440     * The calculation is implemented in this method for {@link ChronoUnit}.
1441     * The units {@code NANOS}, {@code MICROS}, {@code MILLIS}, {@code SECONDS},
1442     * {@code MINUTES}, {@code HOURS} and {@code HALF_DAYS}, {@code DAYS},
1443     * {@code WEEKS}, {@code MONTHS}, {@code YEARS}, {@code DECADES},
1444     * {@code CENTURIES}, {@code MILLENNIA} and {@code ERAS} are supported.
1445     * Other {@code ChronoUnit} values will throw an exception.
1446     * <p>
1447     * If the unit is not a {@code ChronoUnit}, then the result of this method
1448     * is obtained by invoking {@code TemporalUnit.between(Temporal, Temporal)}
1449     * passing {@code this} as the first argument and the input temporal as
1450     * the second argument.
1451     * <p>
1452     * This instance is immutable and unaffected by this method call.
1453     *
1454     * @param endDateTime  the end date-time, which must be an {@code OffsetDateTime}, not null
1455     * @param unit  the unit to measure the period in, not null
1456     * @return the amount of the period between this date-time and the end date-time
1457     * @throws DateTimeException if the period cannot be calculated
1458     * @throws ArithmeticException if numeric overflow occurs
1459     */
1460    @Override
1461    public long periodUntil(Temporal endDateTime, TemporalUnit unit) {
1462        if (endDateTime instanceof OffsetDateTime == false) {
1463            Objects.requireNonNull(endDateTime, "endDateTime");
1464            throw new DateTimeException("Unable to calculate period between objects of two different types");
1465        }
1466        if (unit instanceof ChronoUnit) {
1467            OffsetDateTime end = (OffsetDateTime) endDateTime;
1468            end = end.withOffsetSameInstant(offset);
1469            return dateTime.periodUntil(end.dateTime, unit);
1470        }
1471        return unit.between(this, endDateTime);
1472    }

1610     * Compares this {@code OffsetDateTime} to another date-time.
1611     * <p>
1612     * The comparison is based on the instant then on the local date-time.
1613     * It is "consistent with equals", as defined by {@link Comparable}.
1614     * <p>
1615     * For example, the following is the comparator order:
1616     * <ol>
1617     * <li>{@code 2008-12-03T10:30+01:00}</li>
1618     * <li>{@code 2008-12-03T11:00+01:00}</li>
1619     * <li>{@code 2008-12-03T12:00+02:00}</li>

1621     * <li>{@code 2008-12-03T12:00+01:00}</li>
1622     * <li>{@code 2008-12-03T12:30+01:00}</li>
1623     * </ol>
1624     * Values #2 and #3 represent the same instant on the time-line.
1625     * When two values represent the same instant, the local date-time is compared
1626     * to distinguish them. This step is needed to make the ordering
1627     * consistent with {@code equals()}.
1628     *
1629     * @param other  the other date-time to compare to, not null
1630     * @return the comparator value, negative if less, positive if greater
1631     */
1632    @Override
1633    public int compareTo(OffsetDateTime other) {
1634        if (getOffset().equals(other.getOffset())) {
1635            return toLocalDateTime().compareTo(other.toLocalDateTime());
1636        }
1637        int cmp = Long.compare(toEpochSecond(), other.toEpochSecond());
1638        if (cmp == 0) {
1639            cmp = toLocalTime().getNano() - other.toLocalTime().getNano();
1640            if (cmp == 0) {
1641                cmp = toLocalDateTime().compareTo(other.toLocalDateTime());
1642            }
1643        }
1644        return cmp;
1645    }
