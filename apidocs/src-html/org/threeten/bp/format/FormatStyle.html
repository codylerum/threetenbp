032package org.threeten.bp.format;
033
034/**
035 * Enumeration of the style of a localized date, time or date-time formatter.
036 * <p>
037 * These styles are used when obtaining a date-time style from configuration.
038 * See {@link DateTimeFormatters} and {@link DateTimeFormatterBuilder} for usage.
039 *
040 * <h3>Specification for implementors</h3>
041 * This is an immutable and thread-safe enum.
042 */
043public enum FormatStyle {
044    // ordered from large to small
045
046    /**
047     * Full text style, with the most detail.
048     * For example, the format might be 'Tuesday, April 12, 1952 AD' or '3:30:42pm PST'.
049     */
050    FULL,
051    /**
052     * Long text style, with lots of detail.
053     * For example, the format might be 'January 12, 1952'.
054     */
055    LONG,
056    /**
057     * Medium text style, with some detail.
058     * For example, the format might be 'Jan 12, 1952'.
059     */
060    MEDIUM,
061    /**
062     * Short text style, typically numeric.
063     * For example, the format might be '12.13.52' or '3:30pm'.
064     */
065    SHORT;
066
067}
