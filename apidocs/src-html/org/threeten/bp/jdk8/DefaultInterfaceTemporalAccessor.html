032package org.threeten.bp.jdk8;
033
034import org.threeten.bp.DateTimeException;
035import org.threeten.bp.temporal.ChronoField;
036import org.threeten.bp.temporal.TemporalAccessor;
037import org.threeten.bp.temporal.TemporalField;
038import org.threeten.bp.temporal.TemporalQueries;
039import org.threeten.bp.temporal.TemporalQuery;
040import org.threeten.bp.temporal.ValueRange;
041
042/**
043 * A temporary class providing implementations that will become default interface
044 * methods once integrated into JDK 8.
045 */
046public abstract class DefaultInterfaceTemporalAccessor implements TemporalAccessor {
047
048    @Override
049    public ValueRange range(TemporalField field) {
050        if (field instanceof ChronoField) {
051            if (isSupported(field)) {
052                return field.range();
053            }
054            throw new DateTimeException("Unsupported field: " + field.getName());
055        }
056        return field.doRange(this);
057    }
058
059    @Override
060    public int get(TemporalField field) {
061        return range(field).checkValidIntValue(getLong(field), field);
062    }
063
064    @Override
065    public <R> R query(TemporalQuery<R> query) {
066        if (query == TemporalQueries.zoneId() || query == TemporalQueries.chrono() || query == TemporalQueries.precision()) {
067            return null;
068        }
069        return query.queryFrom(this);
070    }
071
072}
