056/**
057 * A time-zone offset from Greenwich/UTC, such as {@code +02:00}.
058 * <p>
059 * A time-zone offset is the period of time that a time-zone differs from Greenwich/UTC.
060 * This is usually a fixed number of hours and minutes.
061 * <p>
062 * Different parts of the world have different time-zone offsets.
063 * The rules for how offsets vary by place and time of year are captured in the
064 * {@link ZoneId} class.
065 * <p>
066 * For example, Paris is one hour ahead of Greenwich/UTC in winter and two hours
067 * ahead in summer. The {@code ZoneId} instance for Paris will reference two
068 * {@code ZoneOffset} instances - a {@code +01:00} instance for winter,
069 * and a {@code +02:00} instance for summer.
070 * <p>
071 * In 2008, time-zone offsets around the world extended from -12:00 to +14:00.
072 * To prevent any problems with that range being extended, yet still provide
073 * validation, the range of offsets is restricted to -18:00 to 18:00 inclusive.
074 * <p>
075 * This class is designed for use with the ISO calendar system.
076 * The fields of hours, minutes and seconds make assumptions that are valid for the
077 * standard ISO definitions of those fields. This class may be used with other
078 * calendar systems providing the definition of the time fields matches those
079 * of the ISO calendar system.
080 * <p>
081 * Instances of {@code ZoneOffset} must be compared using {@link #equals}.
082 * Implementations may choose to cache certain common offsets, however
083 * applications must not rely on such caching.
084 *
085 * <h3>Specification for implementors</h3>
086 * This class is immutable and thread-safe.
087 */
088public final class ZoneOffset
089        extends ZoneId
090        implements TemporalAccessor, TemporalAdjuster, Comparable<ZoneOffset>, Serializable {

095    private static final ConcurrentMap<String, ZoneOffset> ID_CACHE = new ConcurrentHashMap<>(16, 0.75f, 4);

141    /**
142     * Obtains an instance of {@code ZoneOffset} using the ID.
143     * <p>
144     * This method parses the string ID of a {@code ZoneOffset} to
145     * return an instance. The parsing accepts all the formats generated by
146     * {@link #getId()}, plus some additional formats:
147     * <p><ul>
148     * <li>{@code Z} - for UTC
149     * <li>{@code +h}
150     * <li>{@code +hh}
151     * <li>{@code +hh:mm}
152     * <li>{@code -hh:mm}
153     * <li>{@code +hhmm}
154     * <li>{@code -hhmm}
155     * <li>{@code +hh:mm:ss}
156     * <li>{@code -hh:mm:ss}
157     * <li>{@code +hhmmss}
158     * <li>{@code -hhmmss}
159     * </ul><p>
160     * Note that &plusmn; means either the plus or minus symbol.
161     * <p>
162     * The ID of the returned offset will be normalized to one of the formats
163     * described by {@link #getId()}.
164     * <p>
165     * The maximum supported range is from +18:00 to -18:00 inclusive.
166     *
167     * @param offsetId  the offset ID, not null
168     * @return the zone-offset, not null
169     * @throws DateTimeException if the offset ID is invalid
170     */
171    public static ZoneOffset of(String offsetId) {
172        Objects.requireNonNull(offsetId, "offsetId");
173        // "Z" is always in the cache
174        ZoneOffset offset = ID_CACHE.get(offsetId);
175        if (offset != null) {
176            return offset;
177        }
178
179        // parse - +h, +hh, +hhmm, +hh:mm, +hhmmss, +hh:mm:ss
180        final int hours, minutes, seconds;
181        switch (offsetId.length()) {
182            case 2:
183                offsetId = offsetId.charAt(0) + "0" + offsetId.charAt(1);  // fallthru
184            case 3:
185                hours = parseNumber(offsetId, 1, false);
186                minutes = 0;
187                seconds = 0;
188                break;
189            case 5:
190                hours = parseNumber(offsetId, 1, false);
191                minutes = parseNumber(offsetId, 3, false);
192                seconds = 0;
193                break;
194            case 6:
195                hours = parseNumber(offsetId, 1, false);
196                minutes = parseNumber(offsetId, 4, true);
197                seconds = 0;
198                break;
199            case 7:
200                hours = parseNumber(offsetId, 1, false);
201                minutes = parseNumber(offsetId, 3, false);
202                seconds = parseNumber(offsetId, 5, false);
203                break;
204            case 9:
205                hours = parseNumber(offsetId, 1, false);
206                minutes = parseNumber(offsetId, 4, true);
207                seconds = parseNumber(offsetId, 7, true);
208                break;
209            default:
210                throw new DateTimeException("Invalid ID for ZoneOffset, invalid format: " + offsetId);
211        }
212        char first = offsetId.charAt(0);
213        if (first != '+' && first != '-') {
214            throw new DateTimeException("Invalid ID for ZoneOffset, plus/minus not found when expected: " + offsetId);
215        }
216        if (first == '-') {
217            return ofHoursMinutesSeconds(-hours, -minutes, -seconds);
218        } else {
219            return ofHoursMinutesSeconds(hours, minutes, seconds);
220        }
221    }
222
223    /**
224     * Parse a two digit zero-prefixed number.
225     *
226     * @param offsetId  the offset ID, not null
227     * @param pos  the position to parse, valid
228     * @param precededByColon  should this number be prefixed by a precededByColon
229     * @return the parsed number, from 0 to 99
230     */
231    private static int parseNumber(CharSequence offsetId, int pos, boolean precededByColon) {
232        if (precededByColon && offsetId.charAt(pos - 1) != ':') {
233            throw new DateTimeException("Invalid ID for ZoneOffset, colon not found when expected: " + offsetId);
234        }
235        char ch1 = offsetId.charAt(pos);
236        char ch2 = offsetId.charAt(pos + 1);
237        if (ch1 < '0' || ch1 > '9' || ch2 < '0' || ch2 > '9') {
238            throw new DateTimeException("Invalid ID for ZoneOffset, non numeric characters found: " + offsetId);
239        }
240        return (ch1 - 48) * 10 + (ch2 - 48);
241    }

294     * <p>
295     * A {@code TemporalAccessor} represents some form of date and time information.
296     * This factory converts the arbitrary temporal object to an instance of {@code ZoneOffset}.
297     * <p>

299     * on extracting the {@link ChronoField#OFFSET_SECONDS OFFSET_SECONDS} field.
300     * <p>
301     * This method matches the signature of the functional interface {@link TemporalQuery}
302     * allowing it to be used in queries via method reference, {@code ZoneOffset::from}.
303     *
304     * @param temporal  the temporal object to convert, not null
305     * @return the zone-offset, not null
306     * @throws DateTimeException if unable to convert to an {@code ZoneOffset}
307     */
308    public static ZoneOffset from(TemporalAccessor temporal) {
309        ZoneOffset offset = temporal.query(TemporalQueries.offset());
310        if (offset == null) {
311            throw new DateTimeException("Unable to obtain ZoneOffset from TemporalAccessor: " + temporal.getClass());
312        }
313        return offset;
314    }

317    /**
318     * Validates the offset fields.
319     *
320     * @param hours  the time-zone offset in hours, from -18 to +18
321     * @param minutes  the time-zone offset in minutes, from 0 to &plusmn;59
322     * @param seconds  the time-zone offset in seconds, from 0 to &plusmn;59
323     * @throws DateTimeException if the offset is not in the required range
324     */
325    private static void validate(int hours, int minutes, int seconds) {
326        if (hours < -18 || hours > 18) {
327            throw new DateTimeException("Zone offset hours not in valid range: value " + hours +
328                    " is not in the range -18 to 18");
329        }
330        if (hours > 0) {
331            if (minutes < 0 || seconds < 0) {
332                throw new DateTimeException("Zone offset minutes and seconds must be positive because hours is positive");
333            }
334        } else if (hours < 0) {
335            if (minutes > 0 || seconds > 0) {
336                throw new DateTimeException("Zone offset minutes and seconds must be negative because hours is negative");
337            }
338        } else if ((minutes > 0 && seconds < 0) || (minutes < 0 && seconds > 0)) {
339            throw new DateTimeException("Zone offset minutes and seconds must have the same sign");
340        }
341        if (Math.abs(minutes) > 59) {
342            throw new DateTimeException("Zone offset minutes not in valid range: abs(value) " +
343                    Math.abs(minutes) + " is not in the range 0 to 59");
344        }
345        if (Math.abs(seconds) > 59) {
346            throw new DateTimeException("Zone offset seconds not in valid range: abs(value) " +
347                    Math.abs(seconds) + " is not in the range 0 to 59");
348        }
349        if (Math.abs(hours) == 18 && (Math.abs(minutes) > 0 || Math.abs(seconds) > 0)) {
350            throw new DateTimeException("Zone offset not in valid range: -18:00 to +18:00");
351        }
352    }

354    /**
355     * Calculates the total offset in seconds.
356     *
357     * @param hours  the time-zone offset in hours, from -18 to +18
358     * @param minutes  the time-zone offset in minutes, from 0 to &plusmn;59, sign matches hours and seconds
359     * @param seconds  the time-zone offset in seconds, from 0 to &plusmn;59, sign matches hours and minutes
360     * @return the total in seconds
361     */
362    private static int totalSeconds(int hours, int minutes, int seconds) {
363        return hours * SECONDS_PER_HOUR + minutes * SECONDS_PER_MINUTE + seconds;
364    }

367    /**
368     * Obtains an instance of {@code ZoneOffset} specifying the total offset in seconds
369     * <p>
370     * The offset must be in the range {@code -18:00} to {@code +18:00}, which corresponds to -64800 to +64800.
371     *
372     * @param totalSeconds  the total time-zone offset in seconds, from -64800 to +64800
373     * @return the ZoneOffset, not null
374     * @throws DateTimeException if the offset is not in the required range
375     */
376    public static ZoneOffset ofTotalSeconds(int totalSeconds) {
377        if (Math.abs(totalSeconds) > MAX_SECONDS) {
378            throw new DateTimeException("Zone offset not in valid range: -18:00 to +18:00");
379        }
380        if (totalSeconds % (15 * SECONDS_PER_MINUTE) == 0) {
381            Integer totalSecs = totalSeconds;
382            ZoneOffset result = SECONDS_CACHE.get(totalSecs);
383            if (result == null) {
384                result = new ZoneOffset(totalSeconds);
385                SECONDS_CACHE.putIfAbsent(totalSecs, result);
386                result = SECONDS_CACHE.get(totalSecs);
387                ID_CACHE.putIfAbsent(result.getId(), result);
388            }
389            return result;
390        } else {
391            return new ZoneOffset(totalSeconds);
392        }
393    }

397     * Constructor.
398     *
399     * @param totalSeconds  the total time-zone offset in seconds, from -64800 to +64800

401    private ZoneOffset(int totalSeconds) {
402        super();
403        this.totalSeconds = totalSeconds;
404        id = buildId(totalSeconds);
405    }

407    private static String buildId(int totalSeconds) {
408        if (totalSeconds == 0) {
409            return "Z";
410        } else {
411            int absTotalSeconds = Math.abs(totalSeconds);
412            StringBuilder buf = new StringBuilder();
413            int absHours = absTotalSeconds / SECONDS_PER_HOUR;
414            int absMinutes = (absTotalSeconds / SECONDS_PER_MINUTE) % MINUTES_PER_HOUR;
415            buf.append(totalSeconds < 0 ? "-" : "+")
416                .append(absHours < 10 ? "0" : "").append(absHours)
417                .append(absMinutes < 10 ? ":0" : ":").append(absMinutes);
418            int absSeconds = absTotalSeconds % SECONDS_PER_MINUTE;
419            if (absSeconds != 0) {
420                buf.append(absSeconds < 10 ? ":0" : ":").append(absSeconds);
421            }
422            return buf.toString();
423        }
424    }

613    @SuppressWarnings("unchecked")
614    @Override
615    public <R> R query(TemporalQuery<R> query) {
616        if (query == TemporalQueries.offset() || query == TemporalQueries.zone()) {
617            return (R) this;
618        } else if (query == TemporalQueries.precision() || query == TemporalQueries.chronology() || query == TemporalQueries.zoneId()) {
619            return null;
620        }
621        return query.queryFrom(this);
622    }

624    /**
625     * Adjusts the specified temporal object to have the same offset as this object.
626     * <p>
627     * This returns a temporal object of the same observable type as the input
628     * with the offset changed to be the same as this.
629     * <p>
630     * The adjustment is equivalent to using {@link Temporal#with(TemporalField, long)}
631     * passing {@link ChronoField#OFFSET_SECONDS} as the field.
632     * <p>
633     * In most cases, it is clearer to reverse the calling pattern by using
634     * {@link Temporal#with(TemporalAdjuster)}:
635     * <pre>
636     *   // these two lines are equivalent, but the second approach is recommended
637     *   temporal = thisOffset.adjustInto(temporal);
638     *   temporal = temporal.with(thisOffset);
639     * </pre>
640     * <p>
641     * This instance is immutable and unaffected by this method call.
642     *
643     * @param temporal  the target object to be adjusted, not null
644     * @return the adjusted object, not null
645     * @throws DateTimeException if unable to make the adjustment
646     * @throws ArithmeticException if numeric overflow occurs
647     */
648    @Override
649    public Temporal adjustInto(Temporal temporal) {
650        return temporal.with(OFFSET_SECONDS, totalSeconds);
651    }

656     * <p>
657     * The offsets are compared in the order that they occur for the same time
658     * of day around the world. Thus, an offset of {@code +10:00} comes before an
659     * offset of {@code +09:00} and so on down to {@code -18:00}.
660     * <p>
661     * The comparison is "consistent with equals", as defined by {@link Comparable}.
662     *
663     * @param other  the other date to compare to, not null
664     * @return the comparator value, negative if less, postive if greater
665     * @throws NullPointerException if {@code other} is null
666     */

734    void writeExternal(DataOutput out) throws IOException {
735        final int offsetSecs = totalSeconds;
736        int offsetByte = offsetSecs % 900 == 0 ? offsetSecs / 900 : 127;  // compress to -72 to +72
737        out.writeByte(offsetByte);
738        if (offsetByte == 127) {
739            out.writeInt(offsetSecs);
740        }
741    }

743    static ZoneOffset readExternal(DataInput in) throws IOException {
744        int offsetByte = in.readByte();
745        return (offsetByte == 127 ? ZoneOffset.ofTotalSeconds(in.readInt()) : ZoneOffset.ofTotalSeconds(offsetByte * 900));
746    }
