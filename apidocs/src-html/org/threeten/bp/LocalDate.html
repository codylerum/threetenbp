077/**
078 * A date without a time-zone in the ISO-8601 calendar system,
079 * such as {@code 2007-12-03}.
080 * <p>
081 * {@code LocalDate} is an immutable date-time object that represents a date,
082 * often viewed as year-month-day. Other date fields, such as day-of-year,
083 * day-of-week and week-of-year, can also be accessed.
084 * For example, the value "2nd October 2007" can be stored in a {@code LocalDate}.
085 * <p>
086 * This class does not store or represent a time or time-zone.
087 * Instead, it is a description of the date, as used for birthdays.
088 * It cannot represent an instant on the time-line without additional information
089 * such as an offset or time-zone.
090 * <p>
091 * The ISO-8601 calendar system is the modern civil calendar system used today
092 * in most of the world. It is equivalent to the proleptic Gregorian calendar
093 * system, in which today's rules for leap years are applied for all time.
094 * For most applications written today, the ISO-8601 rules are entirely suitable.
095 * However, any application that makes use of historical dates, and requires them
096 * to be accurate will find the ISO-8601 approach unsuitable.
097 *
098 * <h3>Specification for implementors</h3>
099 * This class is immutable and thread-safe.
100 */
101public final class LocalDate
102        extends DefaultInterfaceChronoLocalDate<LocalDate>
103        implements Temporal, TemporalAdjuster, ChronoLocalDate<LocalDate>, Serializable {

177     * Obtains the current date from the specified clock.
178     * <p>
179     * This will query the specified clock to obtain the current date - today.
180     * Using this method allows the use of an alternate clock for testing.
181     * The alternate clock may be introduced using {@link Clock dependency injection}.
182     *
183     * @param clock  the clock to use, not null
184     * @return the current date, not null

186    public static LocalDate now(Clock clock) {
187        Objects.requireNonNull(clock, "clock");
188        final Instant now = clock.instant();  // called once
189        ZoneOffset offset = clock.getZone().getRules().getOffset(now);
190        long epochSec = now.getEpochSecond() + offset.getTotalSeconds();  // overflow caught later
191        long epochDay = Jdk8Methods.floorDiv(epochSec, SECONDS_PER_DAY);
192        return LocalDate.ofEpochDay(epochDay);
193    }

235    /**
236     * Obtains an instance of {@code LocalDate} from a year and day-of-year.
237     * <p>
238     * The day-of-year must be valid for the year, otherwise an exception will be thrown.
239     *
240     * @param year  the year to represent, from MIN_YEAR to MAX_YEAR
241     * @param dayOfYear  the day-of-year to represent, from 1 to 366
242     * @return the local date, not null
243     * @throws DateTimeException if the value of any field is out of range
244     * @throws DateTimeException if the day-of-year is invalid for the month-year
245     */
246    public static LocalDate ofYearDay(int year, int dayOfYear) {
247        YEAR.checkValidValue(year);
248        DAY_OF_YEAR.checkValidValue(dayOfYear);
249        boolean leap = IsoChronology.INSTANCE.isLeapYear(year);
250        if (dayOfYear == 366 && leap == false) {
251            throw new DateTimeException("Invalid date 'DayOfYear 366' as '" + year + "' is not a leap year");
252        }
253        Month moy = Month.of((dayOfYear - 1) / 31 + 1);
254        int monthEnd = moy.firstDayOfYear(leap) + moy.length(leap) - 1;
255        if (dayOfYear > monthEnd) {
256            moy = moy.plus(1);
257        }
258        int dom = dayOfYear - moy.firstDayOfYear(leap) + 1;
259        return create(year, moy, dom);
260    }

263    /**
264     * Obtains an instance of {@code LocalDate} from the epoch day count.
265     * <p>
266     * The Epoch Day count is a simple incrementing count of days
267     * where day 0 is 1970-01-01. Negative numbers represent earlier days.
268     *
269     * @param epochDay  the Epoch Day to convert, based on the epoch 1970-01-01
270     * @return the local date, not null
271     * @throws DateTimeException if the epoch days exceeds the supported date range
272     */
273    public static LocalDate ofEpochDay(long epochDay) {
274        long zeroDay = epochDay + DAYS_0000_TO_1970;
275        // find the march-based year
276        zeroDay -= 60;  // adjust to 0000-03-01 so leap day is at end of four year cycle
277        long adjust = 0;
278        if (zeroDay < 0) {
279            // adjust negative years to positive for calculation
280            long adjustCycles = (zeroDay + 1) / DAYS_PER_CYCLE - 1;
281            adjust = adjustCycles * 400;
282            zeroDay += -adjustCycles * DAYS_PER_CYCLE;
283        }
284        long yearEst = (400 * zeroDay + 591) / DAYS_PER_CYCLE;
285        long doyEst = zeroDay - (365 * yearEst + yearEst / 4 - yearEst / 100 + yearEst / 400);
286        if (doyEst < 0) {
287            // fix estimate
288            yearEst--;
289            doyEst = zeroDay - (365 * yearEst + yearEst / 4 - yearEst / 100 + yearEst / 400);
290        }
291        yearEst += adjust;  // reset any negative year
292        int marchDoy0 = (int) doyEst;
293
294        // convert march-based values back to january-based
295        int marchMonth0 = (marchDoy0 * 5 + 2) / 153;
296        int month = (marchMonth0 + 2) % 12 + 1;
297        int dom = marchDoy0 - (marchMonth0 * 306 + 5) / 10 + 1;
298        yearEst += marchMonth0 / 10;
299
300        // check year now we are certain it is correct
301        int year = YEAR.checkValidIntValue(yearEst);
302        return new LocalDate(year, month, dom);
303    }

362     * Creates a local date from the year, month and day fields.
363     *
364     * @param year  the year to represent, validated from MIN_YEAR to MAX_YEAR
365     * @param month  the month-of-year to represent, validated not null
366     * @param dayOfMonth  the day-of-month to represent, validated from 1 to 31
367     * @return the local date, not null
368     * @throws DateTimeException if the day-of-month is invalid for the month-year
369     */
370    private static LocalDate create(int year, Month month, int dayOfMonth) {
371        if (dayOfMonth > 28 && dayOfMonth > month.length(IsoChronology.INSTANCE.isLeapYear(year))) {
372            if (dayOfMonth == 29) {
373                throw new DateTimeException("Invalid date 'February 29' as '" + year + "' is not a leap year");
374            } else {
375                throw new DateTimeException("Invalid date '" + month.name() + " " + dayOfMonth + "'");
376            }
377        }
378        return new LocalDate(year, month.getValue(), dayOfMonth);
379    }

381    /**
382     * Resolves the date, resolving days past the end of month.
383     *
384     * @param year  the year to represent, validated from MIN_YEAR to MAX_YEAR
385     * @param month  the month-of-year to represent, validated from 1 to 12
386     * @param day  the day-of-month to represent, validated from 1 to 31
387     * @return the resolved date, not null
388     */
389    private static LocalDate resolvePreviousValid(int year, int month, int day) {
390        switch (month) {
391            case 2:
392                day = Math.min(day, IsoChronology.INSTANCE.isLeapYear(year) ? 29 : 28);
393                break;
394            case 4:
395            case 6:
396            case 9:
397            case 11:
398                day = Math.min(day, 30);
399                break;
400        }
401        return LocalDate.of(year, month, day);
402    }

459    /**
460     * Gets the range of valid values for the specified field.
461     * <p>
462     * The range object expresses the minimum and maximum valid values for a field.
463     * This date is used to enhance the accuracy of the returned range.
464     * If it is not possible to return the range, because the field is not supported
465     * or for some other reason, an exception is thrown.
466     * <p>
467     * If the field is a {@link ChronoField} then the query is implemented here.
468     * The {@link #isSupported(TemporalField) supported fields} will return
469     * appropriate range instances.
470     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
471     * <p>
472     * If the field is not a {@code ChronoField}, then the result of this method
473     * is obtained by invoking {@code TemporalField.rangeRefinedBy(TemporalAccessor)}
474     * passing {@code this} as the argument.
475     * Whether the range can be obtained is determined by the field.
476     *
477     * @param field  the field to query the range for, not null
478     * @return the range of valid values for the field, not null
479     * @throws DateTimeException if the range for the field cannot be obtained
480     */
481    @Override
482    public ValueRange range(TemporalField field) {
483        if (field instanceof ChronoField) {
484            ChronoField f = (ChronoField) field;
485            if (f.isDateField()) {
486                switch (f) {
487                    case DAY_OF_MONTH: return ValueRange.of(1, lengthOfMonth());
488                    case DAY_OF_YEAR: return ValueRange.of(1, lengthOfYear());
489                    case ALIGNED_WEEK_OF_MONTH: return ValueRange.of(1, getMonth() == Month.FEBRUARY && isLeapYear() == false ? 4 : 5);
490                    case YEAR_OF_ERA:
491                        return (getYear() <= 0 ? ValueRange.of(1, Year.MAX_VALUE + 1) : ValueRange.of(1, Year.MAX_VALUE));
492                }
493                return field.range();
494            }
495            throw new DateTimeException("Unsupported field: " + field.getName());
496        }
497        return field.rangeRefinedBy(this);
498    }

568    private int get0(TemporalField field) {
569        switch ((ChronoField) field) {
570            case DAY_OF_WEEK: return getDayOfWeek().getValue();
571            case ALIGNED_DAY_OF_WEEK_IN_MONTH: return ((day - 1) % 7) + 1;
572            case ALIGNED_DAY_OF_WEEK_IN_YEAR: return ((getDayOfYear() - 1) % 7) + 1;
573            case DAY_OF_MONTH: return day;
574            case DAY_OF_YEAR: return getDayOfYear();
575            case EPOCH_DAY: throw new DateTimeException("Field too large for an int: " + field);
576            case ALIGNED_WEEK_OF_MONTH: return ((day - 1) / 7) + 1;
577            case ALIGNED_WEEK_OF_YEAR: return ((getDayOfYear() - 1) / 7) + 1;
578            case MONTH_OF_YEAR: return month;
579            case EPOCH_MONTH: throw new DateTimeException("Field too large for an int: " + field);
580            case YEAR_OF_ERA: return (year >= 1 ? year : 1 - year);
581            case YEAR: return year;
582            case ERA: return (year >= 1 ? 1 : 0);
583        }
584        throw new DateTimeException("Unsupported field: " + field.getName());
585    }

806     * @throws ArithmeticException if numeric overflow occurs
807     */

817    /**
818     * Returns a copy of this date with the specified field set to a new value.
819     * <p>
820     * This returns a new {@code LocalDate}, based on this one, with the value
821     * for the specified field changed.
822     * This can be used to change any supported field, such as the year, month or day-of-month.
823     * If it is not possible to set the value, because the field is not supported or for
824     * some other reason, an exception is thrown.
825     * <p>
826     * In some cases, changing the specified field can cause the resulting date to become invalid,
827     * such as changing the month from 31st January to February would make the day-of-month invalid.
828     * In cases like this, the field is responsible for resolving the date. Typically it will choose
829     * the previous valid date, which would be the last valid day of February in this example.
830     * <p>
831     * If the field is a {@link ChronoField} then the adjustment is implemented here.
832     * The supported fields behave as follows:
833     * <ul>
834     * <li>{@code DAY_OF_WEEK} -
835     *  Returns a {@code LocalDate} with the specified day-of-week.
836     *  The date is adjusted up to 6 days forward or backward within the boundary
837     *  of a Monday to Sunday week.
838     * <li>{@code ALIGNED_DAY_OF_WEEK_IN_MONTH} -
839     *  Returns a {@code LocalDate} with the specified aligned-day-of-week.
840     *  The date is adjusted to the specified month-based aligned-day-of-week.
841     *  Aligned weeks are counted such that the first week of a given month starts
842     *  on the first day of that month.
843     *  This may cause the date to be moved up to 6 days into the following month.
844     * <li>{@code ALIGNED_DAY_OF_WEEK_IN_YEAR} -
845     *  Returns a {@code LocalDate} with the specified aligned-day-of-week.
846     *  The date is adjusted to the specified year-based aligned-day-of-week.
847     *  Aligned weeks are counted such that the first week of a given year starts
848     *  on the first day of that year.
849     *  This may cause the date to be moved up to 6 days into the following year.
850     * <li>{@code DAY_OF_MONTH} -
851     *  Returns a {@code LocalDate} with the specified day-of-month.
852     *  The month and year will be unchanged. If the day-of-month is invalid for the
853     *  year and month, then a {@code DateTimeException} is thrown.
854     * <li>{@code DAY_OF_YEAR} -
855     *  Returns a {@code LocalDate} with the specified day-of-year.
856     *  The year will be unchanged. If the day-of-year is invalid for the
857     *  year, then a {@code DateTimeException} is thrown.
858     * <li>{@code EPOCH_DAY} -
859     *  Returns a {@code LocalDate} with the specified epoch-day.
860     *  This completely replaces the date and is equivalent to {@link #ofEpochDay(long)}.
861     * <li>{@code ALIGNED_WEEK_OF_MONTH} -
862     *  Returns a {@code LocalDate} with the specified aligned-week-of-month.
863     *  Aligned weeks are counted such that the first week of a given month starts
864     *  on the first day of that month.
865     *  This adjustment moves the date in whole week chunks to match the specified week.
866     *  The result will have the same day-of-week as this date.
867     *  This may cause the date to be moved into the following month.
868     * <li>{@code ALIGNED_WEEK_OF_YEAR} -
869     *  Returns a {@code LocalDate} with the specified aligned-week-of-year.
870     *  Aligned weeks are counted such that the first week of a given year starts
871     *  on the first day of that year.
872     *  This adjustment moves the date in whole week chunks to match the specified week.
873     *  The result will have the same day-of-week as this date.
874     *  This may cause the date to be moved into the following year.
875     * <li>{@code MONTH_OF_YEAR} -
876     *  Returns a {@code LocalDate} with the specified month-of-year.
877     *  The year will be unchanged. The day-of-month will also be unchanged,
878     *  unless it would be invalid for the new month and year. In that case, the
879     *  day-of-month is adjusted to the maximum valid value for the new month and year.
880     * <li>{@code EPOCH_MONTH} -
881     *  Returns a {@code LocalDate} with the specified epoch-month.
882     *  The day-of-month will be unchanged, unless it would be invalid for the new month
883     *  and year. In that case, the day-of-month is adjusted to the maximum valid value
884     *  for the new month and year.
885     * <li>{@code YEAR_OF_ERA} -
886     *  Returns a {@code LocalDate} with the specified year-of-era.
887     *  The era and month will be unchanged. The day-of-month will also be unchanged,
888     *  unless it would be invalid for the new month and year. In that case, the
889     *  day-of-month is adjusted to the maximum valid value for the new month and year.
890     * <li>{@code YEAR} -
891     *  Returns a {@code LocalDate} with the specified year.
892     *  The month will be unchanged. The day-of-month will also be unchanged,
893     *  unless it would be invalid for the new month and year. In that case, the
894     *  day-of-month is adjusted to the maximum valid value for the new month and year.
895     * <li>{@code ERA} -
896     *  Returns a {@code LocalDate} with the specified era.
897     *  The year-of-era and month will be unchanged. The day-of-month will also be unchanged,
898     *  unless it would be invalid for the new month and year. In that case, the
899     *  day-of-month is adjusted to the maximum valid value for the new month and year.
900     * </ul>
901     * <p>
902     * In all cases, if the new value is outside the valid range of values for the field
903     * then a {@code DateTimeException} will be thrown.
904     * <p>
905     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
906     * <p>
907     * If the field is not a {@code ChronoField}, then the result of this method
908     * is obtained by invoking {@code TemporalField.adjustInto(Temporal, long)}
909     * passing {@code this} as the argument. In this case, the field determines
910     * whether and how to adjust the instant.
911     * <p>
912     * This instance is immutable and unaffected by this method call.
913     *
914     * @param field  the field to set in the result, not null
915     * @param newValue  the new value of the field in the result
916     * @return a {@code LocalDate} based on {@code this} with the specified field set, not null
917     * @throws DateTimeException if the field cannot be set
918     * @throws ArithmeticException if numeric overflow occurs
919     */
920    @Override
921    public LocalDate with(TemporalField field, long newValue) {
922        if (field instanceof ChronoField) {
923            ChronoField f = (ChronoField) field;
924            f.checkValidValue(newValue);
925            switch (f) {
926                case DAY_OF_WEEK: return plusDays(newValue - getDayOfWeek().getValue());
927                case ALIGNED_DAY_OF_WEEK_IN_MONTH: return plusDays(newValue - getLong(ALIGNED_DAY_OF_WEEK_IN_MONTH));
928                case ALIGNED_DAY_OF_WEEK_IN_YEAR: return plusDays(newValue - getLong(ALIGNED_DAY_OF_WEEK_IN_YEAR));
929                case DAY_OF_MONTH: return withDayOfMonth((int) newValue);
930                case DAY_OF_YEAR: return withDayOfYear((int) newValue);
931                case EPOCH_DAY: return LocalDate.ofEpochDay(newValue);
932                case ALIGNED_WEEK_OF_MONTH: return plusWeeks(newValue - getLong(ALIGNED_WEEK_OF_MONTH));
933                case ALIGNED_WEEK_OF_YEAR: return plusWeeks(newValue - getLong(ALIGNED_WEEK_OF_YEAR));
934                case MONTH_OF_YEAR: return withMonth((int) newValue);
935                case EPOCH_MONTH: return plusMonths(newValue - getLong(EPOCH_MONTH));
936                case YEAR_OF_ERA: return withYear((int) (year >= 1 ? newValue : 1 - newValue));
937                case YEAR: return withYear((int) newValue);
938                case ERA: return (getLong(ERA) == newValue ? this : withYear(1 - year));
939            }
940            throw new DateTimeException("Unsupported field: " + field.getName());
941        }
942        return field.adjustInto(this, newValue);
943    }

1363     * Calculates the period between this date and another date in
1364     * terms of the specified unit.
1365     * <p>
1366     * This calculates the period between two dates in terms of a single unit.
1367     * The start and end points are {@code this} and the specified date.
1368     * The result will be negative if the end is before the start.
1369     * The {@code Temporal} passed to this method must be a {@code LocalDate}.
1370     * For example, the period in days between two dates can be calculated
1371     * using {@code startDate.periodUntil(endDate, DAYS)}.
1372     * <p>
1373     * The calculation returns a whole number, representing the number of
1374     * complete units between the two dates.
1375     * For example, the period in months between 2012-06-15 and 2012-08-14
1376     * will only be one month as it is one day short of two months.

1378     * This method operates in association with {@link TemporalUnit#between}.
1379     * The result of this method is a {@code long} representing the amount of
1380     * the specified unit. By contrast, the result of {@code between} is an
1381     * object that can be used directly in addition/subtraction:
1382     * <pre>
1383     *   long period = start.periodUntil(end, MONTHS);   // this method
1384     *   dateTime.plus(MONTHS.between(start, end));      // use in plus/minus
1385     * </pre>
1386     * <p>
1387     * The calculation is implemented in this method for {@link ChronoUnit}.
1388     * The units {@code DAYS}, {@code WEEKS}, {@code MONTHS}, {@code YEARS},
1389     * {@code DECADES}, {@code CENTURIES}, {@code MILLENNIA} and {@code ERAS}
1390     * are supported. Other {@code ChronoUnit} values will throw an exception.
1391     * <p>
1392     * If the unit is not a {@code ChronoUnit}, then the result of this method
1393     * is obtained by invoking {@code TemporalUnit.between(Temporal, Temporal)}
1394     * passing {@code this} as the first argument and the input temporal as
1395     * the second argument.
1396     * <p>
1397     * This instance is immutable and unaffected by this method call.
1398     *

1400     * @param unit  the unit to measure the period in, not null
1401     * @return the amount of the period between this date and the end date
1402     * @throws DateTimeException if the period cannot be calculated
1403     * @throws ArithmeticException if numeric overflow occurs
1404     */
1405    @Override
1406    public long periodUntil(Temporal endDate, TemporalUnit unit) {
1407        if (endDate instanceof LocalDate == false) {
1408            Objects.requireNonNull(endDate, "endDate");
1409            throw new DateTimeException("Unable to calculate period between objects of two different types");
1410        }
1411        LocalDate end = (LocalDate) endDate;
1412        if (unit instanceof ChronoUnit) {
1413            switch ((ChronoUnit) unit) {
1414                case DAYS: return daysUntil(end);
1415                case WEEKS: return daysUntil(end) / 7;
1416                case MONTHS: return monthsUntil(end);
1417                case YEARS: return monthsUntil(end) / 12;
1418                case DECADES: return monthsUntil(end) / 120;
1419                case CENTURIES: return monthsUntil(end) / 1200;
1420                case MILLENNIA: return monthsUntil(end) / 12000;
1421                case ERAS: return end.getLong(ERA) - getLong(ERA);
1422            }
1423            throw new DateTimeException("Unsupported unit: " + unit.getName());
1424        }
1425        return unit.between(this, endDate);
1426    }

1439     * Calculates the period between this date and another date as a {@code Period}.
1440     * <p>
1441     * This calculates the period between two dates in terms of years, months and days.
1442     * The start and end points are {@code this} and the specified date.
1443     * The result will be negative if the end is before the start.
1444     * <p>
1445     * The calculation is performed using the ISO calendar system.
1446     * If necessary, the input date will be converted to ISO.
1447     * <p>
1448     * The start date is included, but the end date is not.
1449     * The period is calculated by removing complete months, then calculating
1450     * the remaining number of days, adjusting to ensure that both have the same sign.
1451     * The number of months is then normalized into years and months based on a 12 month year.
1452     * A month is considered to be complete if the end day-of-month is greater
1453     * than or equal to the start day-of-month.
1454     * For example, from {@code 2010-01-15} to {@code 2011-03-18} is "1 year, 2 months and 3 days".
1455     * <p>
1456     * The result of this method can be a negative period if the end is before the start.
1457     * The negative sign will be the same in each of year, month and day.
1458     * <p>
1459     * There are two equivalent ways of using this method.
1460     * The first is to invoke this method.
1461     * The second is to use {@link Period#between(LocalDate, LocalDate)}:
1462     * <pre>
1463     *   // these two lines are equivalent
1464     *   period = start.periodUntil(end);
1465     *   period = Period.between(start, end);
1466     * </pre>
1467     * The choice should be made based on which makes the code more readable.
1468     *
1469     * @param endDate  the end date, exclusive, which may be in any chronology, not null
1470     * @return the period between this date and the end date, not null
1471     */
1472    @Override
1473    public Period periodUntil(ChronoLocalDate<?> endDate) {
1474        LocalDate end = LocalDate.from(endDate);
1475        long totalMonths = end.getEpochMonth() - this.getEpochMonth();  // safe
1476        int days = end.day - this.day;
1477        if (totalMonths > 0 && days < 0) {
1478            totalMonths--;
1479            LocalDate calcDate = this.plusMonths(totalMonths);
1480            days = (int) (end.toEpochDay() - calcDate.toEpochDay());  // safe
1481        } else if (totalMonths < 0 && days > 0) {
1482            totalMonths++;
1483            days -= end.lengthOfMonth();
1484        }
1485        long years = totalMonths / 12;  // safe
1486        int months = (int) (totalMonths % 12);  // safe
1487        return Period.of(Jdk8Methods.safeToInt(years), months, days);
1488    }

1627    @Override
1628    public long toEpochDay() {
1629        long y = year;
1630        long m = month;
1631        long total = 0;
1632        total += 365 * y;
1633        if (y >= 0) {
1634            total += (y + 3) / 4 - (y + 99) / 100 + (y + 399) / 400;
1635        } else {
1636            total -= y / -4 - y / -100 + y / -400;
1637        }
1638        total += ((367 * m - 362) / 12);
1639        total += day - 1;
1640        if (m > 2) {
1641            total--;
