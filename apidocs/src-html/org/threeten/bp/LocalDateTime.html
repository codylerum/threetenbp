070/**
071 * A date-time without a time-zone in the ISO-8601 calendar system,
072 * such as {@code 2007-12-03T10:15:30}.
073 * <p>
074 * {@code LocalDateTime} is an immutable date-time object that represents a date-time,
075 * often viewed as year-month-day-hour-minute-second. Other date and time fields,
076 * such as day-of-year, day-of-week and week-of-year, can also be accessed.
077 * Time is represented to nanosecond precision.
078 * For example, the value "2nd October 2007 at 13:45.30.123456789" can be
079 * stored in a {@code LocalDateTime}.
080 * <p>
081 * This class does not store or represent a time-zone.
082 * Instead, it is a description of the date, as used for birthdays, combined with
083 * the local time as seen on a wall clock.
084 * It cannot represent an instant on the time-line without additional information
085 * such as an offset or time-zone.
086 * <p>
087 * The ISO-8601 calendar system is the modern civil calendar system used today
088 * in most of the world. It is equivalent to the proleptic Gregorian calendar
089 * system, in which today's rules for leap years are applied for all time.
090 * For most applications written today, the ISO-8601 rules are entirely suitable.
091 * However, any application that makes use of historical dates, and requires them
092 * to be accurate will find the ISO-8601 approach unsuitable.
093 *
094 * <h3>Specification for implementors</h3>
095 * This class is immutable and thread-safe.
096 */
097public final class LocalDateTime
098        extends DefaultInterfaceChronoLocalDateTime<LocalDate>
099        implements Temporal, TemporalAdjuster, ChronoLocalDateTime<LocalDate>, Serializable {

332     * Obtains an instance of {@code LocalDateTime} from an {@code Instant} and zone ID.
333     * <p>
334     * This creates a local date-time based on the specified instant.
335     * First, the offset from UTC/Greenwich is obtained using the zone ID and instant,

345        Objects.requireNonNull(instant, "instant");
346        Objects.requireNonNull(zone, "zone");
347        ZoneRules rules = zone.getRules();
348        ZoneOffset offset = rules.getOffset(instant);
349        return ofEpochSecond(instant.getEpochSecond(), instant.getNano(), offset);
350    }

352    /**
353     * Obtains an instance of {@code LocalDateTime} using seconds from the
354     * epoch of 1970-01-01T00:00:00Z.
355     * <p>
356     * This allows the {@link ChronoField#INSTANT_SECONDS epoch-second} field
357     * to be converted to a local date-time. This is primarily intended for
358     * low-level conversions rather than general application usage.
359     *
360     * @param epochSecond  the number of seconds from the epoch of 1970-01-01T00:00:00Z
361     * @param nanoOfSecond  the nanosecond within the second, from 0 to 999,999,999
362     * @param offset  the zone offset, not null
363     * @return the local date-time, not null
364     * @throws DateTimeException if the result exceeds the supported range
365     */
366    public static LocalDateTime ofEpochSecond(long epochSecond, int nanoOfSecond, ZoneOffset offset) {
367        Objects.requireNonNull(offset, "offset");
368        long localSecond = epochSecond + offset.getTotalSeconds();  // overflow caught later
369        long localEpochDay = Jdk8Methods.floorDiv(localSecond, SECONDS_PER_DAY);
370        int secsOfDay = Jdk8Methods.floorMod(localSecond, SECONDS_PER_DAY);
371        LocalDate date = LocalDate.ofEpochDay(localEpochDay);
372        LocalTime time = LocalTime.ofSecondOfDay(secsOfDay, nanoOfSecond);
373        return new LocalDateTime(date, time);
374    }

378     * Obtains an instance of {@code LocalDateTime} from a temporal object.
379     * <p>
380     * A {@code TemporalAccessor} represents some form of date and time information.
381     * This factory converts the arbitrary temporal object to an instance of {@code LocalDateTime}.
382     * <p>
383     * The conversion extracts and combines {@code LocalDate} and {@code LocalTime}.
384     * <p>
385     * This method matches the signature of the functional interface {@link TemporalQuery}
386     * allowing it to be used as a query via method reference, {@code LocalDateTime::from}.
387     *
388     * @param temporal  the temporal object to convert, not null
389     * @return the local date-time, not null
390     * @throws DateTimeException if unable to convert to a {@code LocalDateTime}
391     */
392    public static LocalDateTime from(TemporalAccessor temporal) {
393        if (temporal instanceof LocalDateTime) {
394            return (LocalDateTime) temporal;
395        } else if (temporal instanceof ZonedDateTime) {
396            return ((ZonedDateTime) temporal).toLocalDateTime();
397        }
398        try {
399            LocalDate date = LocalDate.from(temporal);
400            LocalTime time = LocalTime.from(temporal);
401            return new LocalDateTime(date, time);
402        } catch (DateTimeException ex) {
403            throw new DateTimeException("Unable to obtain LocalDateTime from TemporalAccessor: " + temporal.getClass(), ex);
404        }
405    }

478     * <li>{@code MICRO_OF_DAY}

485     * <li>{@code HOUR_OF_AMPM}
486     * <li>{@code CLOCK_HOUR_OF_AMPM}
487     * <li>{@code HOUR_OF_DAY}
488     * <li>{@code CLOCK_HOUR_OF_DAY}
489     * <li>{@code AMPM_OF_DAY}
490     * <li>{@code DAY_OF_WEEK}
491     * <li>{@code ALIGNED_DAY_OF_WEEK_IN_MONTH}
492     * <li>{@code ALIGNED_DAY_OF_WEEK_IN_YEAR}
493     * <li>{@code DAY_OF_MONTH}
494     * <li>{@code DAY_OF_YEAR}
495     * <li>{@code EPOCH_DAY}
496     * <li>{@code ALIGNED_WEEK_OF_MONTH}
497     * <li>{@code ALIGNED_WEEK_OF_YEAR}
498     * <li>{@code MONTH_OF_YEAR}
499     * <li>{@code EPOCH_MONTH}
500     * <li>{@code YEAR_OF_ERA}
501     * <li>{@code YEAR}
502     * <li>{@code ERA}
503     * </ul>
504     * All other {@code ChronoField} instances will return false.
505     * <p>
506     * If the field is not a {@code ChronoField}, then the result of this method
507     * is obtained by invoking {@code TemporalField.isSupportedBy(TemporalAccessor)}
508     * passing {@code this} as the argument.
509     * Whether the field is supported is determined by the field.
510     *
511     * @param field  the field to check, null returns false
512     * @return true if the field is supported on this date-time, false if not
513     */
514    @Override
515    public boolean isSupported(TemporalField field) {
516        if (field instanceof ChronoField) {
517            ChronoField f = (ChronoField) field;
518            return f.isDateField() || f.isTimeField();
519        }
520        return field != null && field.isSupportedBy(this);
521    }

523    /**
524     * Gets the range of valid values for the specified field.
525     * <p>
526     * The range object expresses the minimum and maximum valid values for a field.
527     * This date-time is used to enhance the accuracy of the returned range.
528     * If it is not possible to return the range, because the field is not supported
529     * or for some other reason, an exception is thrown.
530     * <p>
531     * If the field is a {@link ChronoField} then the query is implemented here.
532     * The {@link #isSupported(TemporalField) supported fields} will return
533     * appropriate range instances.
534     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
535     * <p>
536     * If the field is not a {@code ChronoField}, then the result of this method
537     * is obtained by invoking {@code TemporalField.rangeRefinedBy(TemporalAccessor)}
538     * passing {@code this} as the argument.
539     * Whether the range can be obtained is determined by the field.
540     *
541     * @param field  the field to query the range for, not null
542     * @return the range of valid values for the field, not null
543     * @throws DateTimeException if the range for the field cannot be obtained
544     */
545    @Override
546    public ValueRange range(TemporalField field) {

741     * Returns an adjusted copy of this date-time.
742     * <p>
743     * This returns a new {@code LocalDateTime}, based on this one, with the date-time adjusted.
744     * The adjustment takes place using the specified adjuster strategy object.
745     * Read the documentation of the adjuster to understand what adjustment will be made.
746     * <p>
747     * A simple adjuster might simply set the one of the fields, such as the year field.
748     * A more complex adjuster might set the date to the last day of the month.
749     * A selection of common adjustments is provided in {@link TemporalAdjusters}.
750     * These include finding the "last day of the month" and "next Wednesday".
751     * Key date-time classes also implement the {@code TemporalAdjuster} interface,
752     * such as {@link Month} and {@link MonthDay MonthDay}.
753     * The adjuster is responsible for handling special cases, such as the varying
754     * lengths of month and leap years.
755     * <p>
756     * For example this code returns a date on the last day of July:
757     * <pre>
758     *  import static org.threeten.bp.Month.*;
759     *  import static org.threeten.bp.temporal.Adjusters.*;
760     *
761     *  result = localDateTime.with(JULY).with(lastDayOfMonth());
762     * </pre>
763     * <p>
764     * The classes {@link LocalDate} and {@link LocalTime} implement {@code TemporalAdjuster},
765     * thus this method can be used to change the date, time or offset:
766     * <pre>
767     *  result = localDateTime.with(date);
768     *  result = localDateTime.with(time);
769     * </pre>
770     * <p>
771     * The result of this method is obtained by invoking the
772     * {@link TemporalAdjuster#adjustInto(Temporal)} method on the
773     * specified adjuster passing {@code this} as the argument.
774     * <p>
775     * This instance is immutable and unaffected by this method call.
776     *
777     * @param adjuster the adjuster to use, not null
778     * @return a {@code LocalDateTime} based on {@code this} with the adjustment made, not null
779     * @throws DateTimeException if the adjustment cannot be made
780     * @throws ArithmeticException if numeric overflow occurs
781     */
782    @Override
783    public LocalDateTime with(TemporalAdjuster adjuster) {
784        // optimizations
785        if (adjuster instanceof LocalDate) {
786            return with((LocalDate) adjuster, time);
787        } else if (adjuster instanceof LocalTime) {
788            return with(date, (LocalTime) adjuster);
789        } else if (adjuster instanceof LocalDateTime) {
790            return (LocalDateTime) adjuster;
791        }
792        return (LocalDateTime) adjuster.adjustInto(this);
793    }

796     * Returns a copy of this date-time with the specified field set to a new value.
797     * <p>
798     * This returns a new {@code LocalDateTime}, based on this one, with the value
799     * for the specified field changed.
800     * This can be used to change any supported field, such as the year, month or day-of-month.
801     * If it is not possible to set the value, because the field is not supported or for
802     * some other reason, an exception is thrown.
803     * <p>
804     * In some cases, changing the specified field can cause the resulting date-time to become invalid,
805     * such as changing the month from 31st January to February would make the day-of-month invalid.
806     * In cases like this, the field is responsible for resolving the date. Typically it will choose
807     * the previous valid date, which would be the last valid day of February in this example.
808     * <p>
809     * If the field is a {@link ChronoField} then the adjustment is implemented here.
810     * The {@link #isSupported(TemporalField) supported fields} will behave as per
811     * the matching method on {@link LocalDate#with(TemporalField, long) LocalDate}
812     * or {@link LocalTime#with(TemporalField, long) LocalTime}.
813     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
814     * <p>
815     * If the field is not a {@code ChronoField}, then the result of this method
816     * is obtained by invoking {@code TemporalField.adjustInto(Temporal, long)}
817     * passing {@code this} as the argument. In this case, the field determines
818     * whether and how to adjust the instant.
819     * <p>

1366    /**
1367     * Returns a copy of this {@code LocalDateTime} with the specified period added.
1368     * <p>
1369     * This instance is immutable and unaffected by this method call.
1370     *
1371     * @param newDate  the new date to base the calculation on, not null
1372     * @param hours  the hours to add, may be negative
1373     * @param minutes the minutes to add, may be negative
1374     * @param seconds the seconds to add, may be negative
1375     * @param nanos the nanos to add, may be negative
1376     * @param sign  the sign to determine add or subtract
1377     * @return the combined result, not null
1378     */
1379    private LocalDateTime plusWithOverflow(LocalDate newDate, long hours, long minutes, long seconds, long nanos, int sign) {
1380        // 9223372036854775808 long, 2147483648 int
1381        if ((hours | minutes | seconds | nanos) == 0) {
1382            return with(newDate, time);
1383        }
1384        long totDays = nanos / NANOS_PER_DAY +             //   max/24*60*60*1B
1385                seconds / SECONDS_PER_DAY +                //   max/24*60*60
1386                minutes / MINUTES_PER_DAY +                //   max/24*60
1387                hours / HOURS_PER_DAY;                     //   max/24
1388        totDays *= sign;                                   // total max*0.4237...
1389        long totNanos = nanos % NANOS_PER_DAY +                    //   max  86400000000000
1390                (seconds % SECONDS_PER_DAY) * NANOS_PER_SECOND +   //   max  86400000000000
1391                (minutes % MINUTES_PER_DAY) * NANOS_PER_MINUTE +   //   max  86400000000000
1392                (hours % HOURS_PER_DAY) * NANOS_PER_HOUR;          //   max  86400000000000
1393        long curNoD = time.toNanoOfDay();                       //   max  86400000000000
1394        totNanos = totNanos * sign + curNoD;                    // total 432000000000000
1395        totDays += Jdk8Methods.floorDiv(totNanos, NANOS_PER_DAY);
1396        long newNoD = Jdk8Methods.floorMod(totNanos, NANOS_PER_DAY);
1397        LocalTime newTime = (newNoD == curNoD ? time : LocalTime.ofNanoOfDay(newNoD));
1398        return with(newDate.plusDays(totDays), newTime);
1399    }

1455    /**
1456     * Calculates the period between this date-time and another date-time in
1457     * terms of the specified unit.
1458     * <p>
1459     * This calculates the period between two date-times in terms of a single unit.
1460     * The start and end points are {@code this} and the specified date-time.
1461     * The result will be negative if the end is before the start.
1462     * The {@code Temporal} passed to this method must be a {@code LocalDateTime}.
1463     * For example, the period in days between two date-times can be calculated
1464     * using {@code startDateTime.periodUntil(endDateTime, DAYS)}.
1465     * <p>
1466     * The calculation returns a whole number, representing the number of
1467     * complete units between the two date-times.
1468     * For example, the period in months between 2012-06-15T00:00 and 2012-08-14T23:59
1469     * will only be one month as it is one minute short of two months.
1470     * <p>
1471     * This method operates in association with {@link TemporalUnit#between}.
1472     * The result of this method is a {@code long} representing the amount of
1473     * the specified unit. By contrast, the result of {@code between} is an
1474     * object that can be used directly in addition/subtraction:
1475     * <pre>
1476     *   long period = start.periodUntil(end, MONTHS);   // this method
1477     *   dateTime.plus(MONTHS.between(start, end));      // use in plus/minus
1478     * </pre>
1479     * <p>
1480     * The calculation is implemented in this method for {@link ChronoUnit}.
1481     * The units {@code NANOS}, {@code MICROS}, {@code MILLIS}, {@code SECONDS},
1482     * {@code MINUTES}, {@code HOURS} and {@code HALF_DAYS}, {@code DAYS},
1483     * {@code WEEKS}, {@code MONTHS}, {@code YEARS}, {@code DECADES},
1484     * {@code CENTURIES}, {@code MILLENNIA} and {@code ERAS} are supported.
1485     * Other {@code ChronoUnit} values will throw an exception.
1486     * <p>
1487     * If the unit is not a {@code ChronoUnit}, then the result of this method
1488     * is obtained by invoking {@code TemporalUnit.between(Temporal, Temporal)}
1489     * passing {@code this} as the first argument and the input temporal as
1490     * the second argument.
1491     * <p>
1492     * This instance is immutable and unaffected by this method call.
1493     *
1494     * @param endDateTime  the end date-time, which must be a {@code LocalDateTime}, not null
1495     * @param unit  the unit to measure the period in, not null
1496     * @return the amount of the period between this date-time and the end date-time
1497     * @throws DateTimeException if the period cannot be calculated
1498     * @throws ArithmeticException if numeric overflow occurs
1499     */
1500    @Override
1501    public long periodUntil(Temporal endDateTime, TemporalUnit unit) {
1502        if (endDateTime instanceof LocalDateTime == false) {
1503            Objects.requireNonNull(endDateTime, "endDateTime");
1504            throw new DateTimeException("Unable to calculate period between objects of two different types");
1505        }
1506        LocalDateTime end = (LocalDateTime) endDateTime;
1507        if (unit instanceof ChronoUnit) {
1508            ChronoUnit f = (ChronoUnit) unit;
1509            if (f.isTimeUnit()) {
1510                long amount = date.daysUntil(end.date);
1511                switch (f) {
1512                    case NANOS: amount = Jdk8Methods.safeMultiply(amount, NANOS_PER_DAY); break;
1513                    case MICROS: amount = Jdk8Methods.safeMultiply(amount, MICROS_PER_DAY); break;
1514                    case MILLIS: amount = Jdk8Methods.safeMultiply(amount, MILLIS_PER_DAY); break;
1515                    case SECONDS: amount = Jdk8Methods.safeMultiply(amount, SECONDS_PER_DAY); break;
1516                    case MINUTES: amount = Jdk8Methods.safeMultiply(amount, MINUTES_PER_DAY); break;
1517                    case HOURS: amount = Jdk8Methods.safeMultiply(amount, HOURS_PER_DAY); break;
1518                    case HALF_DAYS: amount = Jdk8Methods.safeMultiply(amount, 2); break;
1519                }
1520                return Jdk8Methods.safeAdd(amount, time.periodUntil(end.time, unit));
1521            }
1522            LocalDate endDate = end.date;
1523            if (end.time.isBefore(time)) {
1524                endDate = endDate.minusDays(1);
1525            }
1526            return date.periodUntil(endDate, unit);
1527        }
1528        return unit.between(this, endDateTime);
1529    }
