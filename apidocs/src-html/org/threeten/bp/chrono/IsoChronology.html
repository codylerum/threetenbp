051/**
052 * The ISO calendar system.
053 * <p>
054 * This chronology defines the rules of the ISO calendar system.
055 * This calendar system is based on the ISO-8601 standard, which is the
056 * <i>de facto</i> world calendar.
057 * <p>
058 * The fields are defined as follows:
059 * <p><ul>
060 * <li>era - There are two eras, 'Current Era' (CE) and 'Before Current Era' (BCE).
061 * <li>year-of-era - The year-of-era is the same as the proleptic-year for the current CE era.
062 *  For the BCE era before the ISO epoch the year increases from 1 upwards as time goes backwards.
063 * <li>proleptic-year - The proleptic year is the same as the year-of-era for the
064 *  current era. For the previous era, years have zero, then negative values.
065 * <li>month-of-year - There are 12 months in an ISO year, numbered from 1 to 12.
066 * <li>day-of-month - There are between 28 and 31 days in each of the ISO month, numbered from 1 to 31.
067 *  Months 4, 6, 9 and 11 have 30 days, Months 1, 3, 5, 7, 8, 10 and 12 have 31 days.
068 *  Month 2 has 28 days, or 29 in a leap year.
069 * <li>day-of-year - There are 365 days in a standard ISO year and 366 in a leap year.
070 *  The days are numbered from 1 to 365 or 1 to 366.
071 * <li>leap-year - Leap years occur every 4 years, except where the year is divisble by 100 and not divisble by 400.
072 * </ul><p>
073 *
074 * <h3>Specification for implementors</h3>
075 * This class is immutable and thread-safe.
076 */
077public final class IsoChronology extends Chronology implements Serializable {

151    /**
152     * Obtains an ISO local date from the era, year-of-era, month-of-year
153     * and day-of-month fields.
154     *
155     * @param era  the ISO era, not null
156     * @param yearOfEra  the ISO year-of-era
157     * @param month  the ISO month-of-year
158     * @param dayOfMonth  the ISO day-of-month
159     * @return the ISO local date, not null
160     * @throws DateTimeException if unable to create the date
161     */
162    @Override  // override with covariant return type
163    public LocalDate date(Era era, int yearOfEra, int month, int dayOfMonth) {
164        return date(prolepticYear(era, yearOfEra), month, dayOfMonth);
165    }

272    /**
273     * Obtains the current ISO local date from the system clock in the default time-zone.
274     * <p>
275     * This will query the {@link Clock#systemDefaultZone() system clock} in the default
276     * time-zone to obtain the current date.
277     * <p>
278     * Using this method will prevent the ability to use an alternate clock for testing
279     * because the clock is hard-coded.
280     *
281     * @return the current ISO local date using the system clock and default time-zone, not null
282     * @throws DateTimeException if unable to create the date
283     */
284    @Override  // override with covariant return type
285    public LocalDate dateNow() {
286        return dateNow(Clock.systemDefaultZone());
287    }

289    /**
290     * Obtains the current ISO local date from the system clock in the specified time-zone.
291     * <p>
292     * This will query the {@link Clock#system(ZoneId) system clock} to obtain the current date.
293     * Specifying the time-zone avoids dependence on the default time-zone.
294     * <p>
295     * Using this method will prevent the ability to use an alternate clock for testing
296     * because the clock is hard-coded.
297     *
298     * @return the current ISO local date using the system clock, not null
299     * @throws DateTimeException if unable to create the date
300     */
301    @Override  // override with covariant return type
302    public LocalDate dateNow(ZoneId zone) {
303        return dateNow(Clock.system(zone));
304    }
305
306    /**
307     * Obtains the current ISO local date from the specified clock.
308     * <p>
309     * This will query the specified clock to obtain the current date - today.
310     * Using this method allows the use of an alternate clock for testing.
311     * The alternate clock may be introduced using {@link Clock dependency injection}.
312     *
313     * @param clock  the clock to use, not null
314     * @return the current ISO local date, not null
315     * @throws DateTimeException if unable to create the date
316     */
317    @Override  // override with covariant return type
318    public LocalDate dateNow(Clock clock) {
319        Objects.requireNonNull(clock, "clock");
320        return date(LocalDate.now(clock));
321    }

323    //-----------------------------------------------------------------------
324    /**
325     * Checks if the year is a leap year, according to the ISO proleptic
326     * calendar system rules.
327     * <p>
328     * This method applies the current rules for leap years across the whole time-line.
329     * In general, a year is a leap year if it is divisible by four without
330     * remainder. However, years divisible by 100, are not leap years, with
331     * the exception of years divisible by 400 which are.
332     * <p>
333     * For example, 1904 is a leap year it is divisible by 4.
334     * 1900 was not a leap year as it is divisible by 100, however 2000 was a
335     * leap year as it is divisible by 400.
336     * <p>
337     * The calculation is proleptic - applying the same rules into the far future and far past.
338     * This is historically inaccurate, but is correct for the ISO-8601 standard.
339     *
340     * @param prolepticYear  the ISO proleptic year to check
341     * @return true if the year is leap, false otherwise
342     */
343    @Override
344    public boolean isLeapYear(long prolepticYear) {
345        return ((prolepticYear & 3) == 0) && ((prolepticYear % 100) != 0 || (prolepticYear % 400) == 0);
346    }
347
348    @Override
349    public int prolepticYear(Era era, int yearOfEra) {
350        if (era instanceof IsoEra == false) {
351            throw new DateTimeException("Era must be IsoEra");
352        }
353        return (era == IsoEra.CE ? yearOfEra : 1 - yearOfEra);
354    }
355
356    @Override
357    public Era eraOf(int eraValue) {
358        return IsoEra.of(eraValue);
359    }
360
361    @Override
362    public List<Era> eras() {
363        return Arrays.<Era>asList(IsoEra.values());
364    }

366    //-----------------------------------------------------------------------
367    @Override
368    public ValueRange range(ChronoField field) {
369        return field.range();
370    }
