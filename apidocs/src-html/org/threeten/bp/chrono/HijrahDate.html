066/**
067 * A date in the Hijrah calendar system.
068 * <p>
069 * This implements {@code ChronoLocalDate} for the {@link HijrahChronology Hijrah calendar}.
070 * <p>
071 * The Hijrah calendar has a different total of days in a year than
072 * Gregorian calendar, and a month is based on the period of a complete
073 * revolution of the moon around the earth (as between successive new moons).
074 * The calendar cycles becomes longer and unstable, and sometimes a manual
075 * adjustment (for entering deviation) is necessary for correctness
076 * because of the complex algorithm.
077 * <p>
078 * HijrahDate supports the manual adjustment feature by providing a configuration
079 * file. The configuration file contains the adjustment (deviation) data with following format.
080 * <pre>
081 *   StartYear/StartMonth(0-based)-EndYear/EndMonth(0-based):Deviation day (1, 2, -1, or -2)
082 *   Line separator or ";" is used for the separator of each deviation data.</pre>
083 *   Here is the example.
084 * <pre>
085 *     1429/0-1429/1:1
086 *     1429/2-1429/7:1;1429/6-1429/11:1
087 *     1429/11-9999/11:1</pre>
088 * The default location of the configuration file is:
089 * <pre>
090 *   $CLASSPATH/org/threeten/bp/chrono</pre>
091 * And the default file name is:
092 * <pre>
093 *   hijrah_deviation.cfg</pre>
094 * The default location and file name can be overriden by setting
095 * following two Java's system property.
096 * <pre>
097 *   Location: org.threeten.bp.i18n.HijrahDate.deviationConfigDir
098 *   File name: org.threeten.bp.i18n.HijrahDate.deviationConfigFile</pre>
099 *
100 * <h3>Specification for implementors</h3>
101 * This class is immutable and thread-safe.
102 */

415     * Obtains an instance of {@code HijrahDate} from the Hijrah era year,
416     * month-of-year and day-of-month. This uses the Hijrah era.
417     *
418     * @param prolepticYear  the proleptic year to represent in the Hijrah
419     * @param monthOfYear  the month-of-year to represent, from 1 to 12
420     * @param dayOfMonth  the day-of-month to represent, from 1 to 30
421     * @return the Hijrah date, never null
422     * @throws IllegalCalendarFieldValueException if the value of any field is out of range
423     * @throws InvalidCalendarFieldException if the day-of-month is invalid for the month-year
424     */
425    static HijrahDate of(int prolepticYear, int monthOfYear, int dayOfMonth) {
426        return (prolepticYear >= 1) ?
427            HijrahDate.of(HijrahEra.AH, prolepticYear, monthOfYear, dayOfMonth) :
428            HijrahDate.of(HijrahEra.BEFORE_AH, 1 - prolepticYear, monthOfYear, dayOfMonth);
429    }

431    /**
432     * Obtains an instance of {@code HijrahDate} from the era, year-of-era
433     * month-of-year and day-of-month.
434     *
435     * @param era  the era to represent, not null
436     * @param yearOfEra  the year-of-era to represent, from 1 to 9999
437     * @param monthOfYear  the month-of-year to represent, from 1 to 12
438     * @param dayOfMonth  the day-of-month to represent, from 1 to 31
439     * @return the Hijrah date, never null
440     * @throws IllegalCalendarFieldValueException if the value of any field is out of range
441     * @throws InvalidCalendarFieldException if the day-of-month is invalid for the month-year
442     */
443    static HijrahDate of(HijrahEra era, int yearOfEra, int monthOfYear, int dayOfMonth) {
444        Objects.requireNonNull(era, "era");
445        checkValidYearOfEra(yearOfEra);
446        checkValidMonth(monthOfYear);
447        checkValidDayOfMonth(dayOfMonth);
448        long gregorianDays = getGregorianEpochDay(era.prolepticYear(yearOfEra), monthOfYear, dayOfMonth);
449        return new HijrahDate(gregorianDays);
450    }

457         if (yearOfEra < MIN_VALUE_OF_ERA  ||
458                 yearOfEra > MAX_VALUE_OF_ERA) {
459             throw new DateTimeException("Invalid year of Hijrah Era");

467         }
468    }

500    /**
501     * Constructs an instance with the specified date.
502     *
503     * @param gregorianDay  the number of days from 0001/01/01 (Gregorian), caller calculated
504     */
505    private HijrahDate(long gregorianDay) {
506        int[] dateInfo = getHijrahDateInfo(gregorianDay);
507
508        checkValidYearOfEra(dateInfo[1]);
509        checkValidMonth(dateInfo[2]);
510        checkValidDayOfMonth(dateInfo[3]);
511        checkValidDayOfYear(dateInfo[4]);
512
513        this.era = HijrahEra.of(dateInfo[0]);
514        this.yearOfEra = dateInfo[1];
515        this.monthOfYear = dateInfo[2];
516        this.dayOfMonth = dateInfo[3];
517        this.dayOfYear = dateInfo[4];
518        this.dayOfWeek = DayOfWeek.of(dateInfo[5]);
519        this.gregorianEpochDay = gregorianDay;
520        this.isLeapYear = isLeapYear(this.yearOfEra);
521    }

538    @Override
539    public ValueRange range(TemporalField field) {
540        if (field instanceof ChronoField) {
541            if (isSupported(field)) {
542                ChronoField f = (ChronoField) field;
543                switch (f) {
544                    case DAY_OF_MONTH: return ValueRange.of(1, lengthOfMonth());
545                    case DAY_OF_YEAR: return ValueRange.of(1, lengthOfYear());
546                    case ALIGNED_WEEK_OF_MONTH: return ValueRange.of(1, 5);  // TODO
547                    case YEAR_OF_ERA: return ValueRange.of(1, 1000);  // TODO
548                }
549                return getChronology().range(f);
550            }
551            throw new DateTimeException("Unsupported field: " + field.getName());
552        }
553        return field.rangeRefinedBy(this);
554    }

556    @Override
557    public long getLong(TemporalField field) {
558        if (field instanceof ChronoField) {
559            switch ((ChronoField) field) {
560                case DAY_OF_WEEK: return dayOfWeek.getValue();
561                case ALIGNED_DAY_OF_WEEK_IN_MONTH: return ((dayOfWeek.getValue() - 1) % 7) + 1;
562                case ALIGNED_DAY_OF_WEEK_IN_YEAR: return ((dayOfYear - 1) % 7) + 1;
563                case DAY_OF_MONTH: return this.dayOfMonth;
564                case DAY_OF_YEAR: return this.dayOfYear;
565                case EPOCH_DAY: return toEpochDay();
566                case ALIGNED_WEEK_OF_MONTH: return ((dayOfMonth - 1) / 7) + 1;
567                case ALIGNED_WEEK_OF_YEAR: return ((dayOfYear - 1) / 7) + 1;
568                case MONTH_OF_YEAR: return monthOfYear;
569                case YEAR_OF_ERA: return yearOfEra;
570                case YEAR: return yearOfEra;
571                case ERA: return era.getValue();
572            }
573            throw new DateTimeException("Unsupported field: " + field.getName());
574        }
575        return field.getFrom(this);
576    }

578    @Override
579    public HijrahDate with(TemporalField field, long newValue) {
580        if (field instanceof ChronoField) {
581            ChronoField f = (ChronoField) field;
582            f.checkValidValue(newValue);        // TODO: validate value
583            int nvalue = (int) newValue;
584            switch (f) {
585                case DAY_OF_WEEK: return plusDays(newValue - dayOfWeek.getValue());
586                case ALIGNED_DAY_OF_WEEK_IN_MONTH: return plusDays(newValue - getLong(ALIGNED_DAY_OF_WEEK_IN_MONTH));
587                case ALIGNED_DAY_OF_WEEK_IN_YEAR: return plusDays(newValue - getLong(ALIGNED_DAY_OF_WEEK_IN_YEAR));
588                case DAY_OF_MONTH: return resolvePreviousValid(yearOfEra, monthOfYear, nvalue);
589                case DAY_OF_YEAR: return resolvePreviousValid(yearOfEra, ((nvalue - 1) / 30) + 1, ((nvalue - 1) % 30) + 1);
590                case EPOCH_DAY: return new HijrahDate(nvalue);
591                case ALIGNED_WEEK_OF_MONTH: return plusDays((newValue - getLong(ALIGNED_WEEK_OF_MONTH)) * 7);
592                case ALIGNED_WEEK_OF_YEAR: return plusDays((newValue - getLong(ALIGNED_WEEK_OF_YEAR)) * 7);
593                case MONTH_OF_YEAR: return resolvePreviousValid(yearOfEra, nvalue, dayOfMonth);
594                case YEAR_OF_ERA: return resolvePreviousValid(yearOfEra >= 1 ? nvalue : 1 - nvalue, monthOfYear, dayOfMonth);
595                case YEAR: return resolvePreviousValid(nvalue, monthOfYear, dayOfMonth);
596                case ERA: return resolvePreviousValid(1 - yearOfEra, monthOfYear, dayOfMonth);
597            }
598            throw new DateTimeException("Unsupported field: " + field.getName());
599        }
600        return field.adjustInto(this, newValue);
601    }

603    private static HijrahDate resolvePreviousValid(int yearOfEra, int month, int day) {
604        int monthDays = getMonthDays(month - 1, yearOfEra);
605        if (day > monthDays) {
606            day = monthDays;
607        }
608        return HijrahDate.of(yearOfEra, month, day);
609    }

645        if (months == 0) {
646            return this;
647        }
648        int newMonth = this.monthOfYear - 1;
649        newMonth = newMonth + (int)months;
650        int years = newMonth / 12;

655        }
656        int newYear = Jdk8Methods.safeAdd(this.yearOfEra, years);
657        return HijrahDate.of(this.era, newYear, newMonth + 1, this.dayOfMonth);

666    /**
667     * Returns the int array containing the following field from the julian day.
668     *
669     * int[0] = ERA
670     * int[1] = YEAR
671     * int[2] = MONTH
672     * int[3] = DATE
673     * int[4] = DAY_OF_YEAR
674     * int[5] = DAY_OF_WEEK
675     *
676     * @param julianDay  a julian day.
677     */
678    private static int[] getHijrahDateInfo(long gregorianDays) {
679        int era, year, month, date, dayOfWeek, dayOfYear;
680
681        int cycleNumber, yearInCycle, dayOfCycle;
682
683        long epochDay = gregorianDays - HIJRAH_JAN_1_1_GREGORIAN_DAY;
684
685        if (epochDay >= 0) {
686            cycleNumber = getCycleNumber(epochDay); // 0 - 99.
687            dayOfCycle = getDayOfCycle(epochDay, cycleNumber); // 0 - 10631.
688            yearInCycle = getYearInCycle(cycleNumber, dayOfCycle); // 0 - 29.
689            dayOfYear = getDayOfYear(cycleNumber, dayOfCycle, yearInCycle);
690            // 0 - 354/355
691            year = cycleNumber * 30 + yearInCycle + 1; // 1-based year.
692            month = getMonthOfYear(dayOfYear, year); // 0-based month-of-year
693            date = getDayOfMonth(dayOfYear, month, year); // 0-based date
694            ++date; // Convert from 0-based to 1-based
695            era = HijrahEra.AH.getValue();
696        } else {
697            cycleNumber = (int) epochDay / 10631; // 0 or negative number.
698            dayOfCycle = (int) epochDay % 10631; // -10630 - 0.
699            if (dayOfCycle == 0) {
700                dayOfCycle = -10631;
701                cycleNumber++;
702            }
703            yearInCycle = getYearInCycle(cycleNumber, dayOfCycle); // 0 - 29.
704            dayOfYear = getDayOfYear(cycleNumber, dayOfCycle, yearInCycle);
705            year = cycleNumber * 30 - yearInCycle; // negative number.
706            year = 1 - year;
707            dayOfYear = (isLeapYear(year) ? (dayOfYear + 355)
708                    : (dayOfYear + 354));
709            month = getMonthOfYear(dayOfYear, year);
710            date = getDayOfMonth(dayOfYear, month, year);
711            ++date; // Convert from 0-based to 1-based
712            era = HijrahEra.BEFORE_AH.getValue();
713        }
714        // Hijrah day zero is a Friday
715        dayOfWeek = (int) ((epochDay + 5) % 7);
716        dayOfWeek += (dayOfWeek <= 0) ? 7 : 0;
717
718        int dateInfo[] = new int[6];
719        dateInfo[0] = era;
720        dateInfo[1] = year;
721        dateInfo[2] = month + 1; // change to 1-based.
722        dateInfo[3] = date;
723        dateInfo[4] = dayOfYear + 1; // change to 1-based.
724        dateInfo[5] = dayOfWeek;
725        return dateInfo;
726    }

736    private static long getGregorianEpochDay(int prolepticYear, int monthOfYear, int dayOfMonth) {

739        day += dayOfMonth;

743    /**
744     * Returns the Gregorian epoch day from the proleptic year
745     * @param prolepticYear the proleptic year
746     * @return the Epoch day
747     */
748    private static long yearToGregorianEpochDay(int prolepticYear) {
749
750        int cycleNumber = (prolepticYear - 1) / 30; // 0-based.
751        int yearInCycle = (prolepticYear - 1) % 30; // 0-based.
752
753        int dayInCycle = getAdjustedCycle(cycleNumber)[Math.abs(yearInCycle)]
754                .intValue();
755
756        if (yearInCycle < 0) {
757            dayInCycle = -dayInCycle;
758        }
759
760        Long cycleDays;
761
762        try {
763            cycleDays = ADJUSTED_CYCLES[cycleNumber];
764        } catch (ArrayIndexOutOfBoundsException e) {
765            cycleDays = null;
766        }
767
768        if (cycleDays == null) {
769            cycleDays = new Long(cycleNumber * 10631);
770        }
771
772        return (cycleDays.longValue() + dayInCycle + HIJRAH_JAN_1_1_GREGORIAN_DAY - 1);
773    }

776     * Returns the 30 year cycle number from the epoch day.
777     *
778     * @param epochDay  an epoch day
779     * @return a cycle number
780     */
781    private static int getCycleNumber(long epochDay) {
782        Long[] days = ADJUSTED_CYCLES;
783        int cycleNumber;
784        try {
785            for (int i = 0; i < days.length; i++) {
786                if (epochDay < days[i].longValue()) {
787                    return i - 1;
788                }
789            }
790            cycleNumber = (int) epochDay / 10631;
791        } catch (ArrayIndexOutOfBoundsException e) {
792            cycleNumber = (int) epochDay / 10631;
793        }
794        return cycleNumber;
795    }

819     * Returns the year in cycle from the cycle number and day of cycle.
820     *
821     * @param cycleNumber  a cycle number
822     * @param dayOfCycle  day of cycle
823     * @return a year in cycle
824     */
825    private static int getYearInCycle(int cycleNumber, long dayOfCycle) {
826        Integer[] cycles = getAdjustedCycle(cycleNumber);
827        if (dayOfCycle == 0) {
828            return 0;
829        }
830
831        if (dayOfCycle > 0) {
832            for (int i = 0; i < cycles.length; i++) {
833                if (dayOfCycle < cycles[i].intValue()) {
834                    return i - 1;
835                }
836            }
837            return 29;
838        } else {
839            dayOfCycle = -dayOfCycle;
840            for (int i = 0; i < cycles.length; i++) {
841                if (dayOfCycle <= cycles[i].intValue()) {
842                    return i - 1;
843                }
844            }
845            return 29;
846        }
847    }

1103    /**
1104     * Adds deviation definition. The year and month sepcifed should be the
1105     * caluculated Hijrah year and month. The month is 0 based. e.g. 8 for
1106     * Ramadan (9th month) Addition of anything minus deviation days is
1107     * calculated negatively in the case the user wants to subtract days from
1108     * the calendar. For example, adding -1 days will subtract one day from the
1109     * current date. Please note that this behavior is different from the
1110     * addDeviaiton method.
1111     *
1112     * @param startYear  start year
1113     * @param startMonth  start month
1114     * @param endYear  end year
1115     * @param endMonth  end month
1116     * @param offset  offset
1117     */
1118    private static void addDeviationAsHijrah(int startYear,
1119            int startMonth, int endYear, int endMonth, int offset) {
1120
1121        if (startYear < 1) {
1122            throw new IllegalArgumentException("startYear < 1");
1123        }
1124        if (endYear < 1) {
1125            throw new IllegalArgumentException("endYear < 1");
1126        }
1127        if (startMonth < 0 || startMonth > 11) {
1128            throw new IllegalArgumentException(
1129                    "startMonth < 0 || startMonth > 11");
1130        }
1131        if (endMonth < 0 || endMonth > 11) {
1132            throw new IllegalArgumentException("endMonth < 0 || endMonth > 11");
1133        }
1134        if (endYear > 9999) {
1135            throw new IllegalArgumentException("endYear > 9999");
1136        }
1137        if (endYear < startYear) {
1138            throw new IllegalArgumentException("startYear > endYear");
1139        }
1140        if (endYear == startYear && endMonth < startMonth) {
1141            throw new IllegalArgumentException(
1142                    "startYear == endYear && endMonth < startMonth");
1143        }
1144
1145        // Adjusting start year.
1146        boolean isStartYLeap = isLeapYear(startYear);
1147
1148        // Adjusting the number of month.
1149        Integer[] orgStartMonthNums = ADJUSTED_MONTH_DAYS.get(new Integer(
1150                startYear));
1151        if (orgStartMonthNums == null) {
1152            if (isStartYLeap) {
1153                orgStartMonthNums = new Integer[LEAP_NUM_DAYS.length];
1154                for (int l = 0; l < LEAP_NUM_DAYS.length; l++) {
1155                    orgStartMonthNums[l] = new Integer(LEAP_NUM_DAYS[l]);
1156                }
1157            } else {
1158                orgStartMonthNums = new Integer[NUM_DAYS.length];
1159                for (int l = 0; l < NUM_DAYS.length; l++) {
1160                    orgStartMonthNums[l] = new Integer(NUM_DAYS[l]);
1161                }
1162            }
1163        }
1164
1165        Integer[] newStartMonthNums = new Integer[orgStartMonthNums.length];
1166
1167        for (int month = 0; month < 12; month++) {
1168            if (month > startMonth) {
1169                newStartMonthNums[month] = new Integer(orgStartMonthNums[month]
1170                        .intValue()
1171                        - offset);
1172            } else {
1173                newStartMonthNums[month] = new Integer(orgStartMonthNums[month]
1174                        .intValue());
1175            }
1176        }
1177
1178        ADJUSTED_MONTH_DAYS.put(new Integer(startYear), newStartMonthNums);
1179
1180        // Adjusting the days of month.
1181
1182        Integer[] orgStartMonthLengths = ADJUSTED_MONTH_LENGTHS.get(new Integer(
1183                startYear));
1184        if (orgStartMonthLengths == null) {
1185            if (isStartYLeap) {
1186                orgStartMonthLengths = new Integer[LEAP_MONTH_LENGTH.length];
1187                for (int l = 0; l < LEAP_MONTH_LENGTH.length; l++) {
1188                    orgStartMonthLengths[l] = new Integer(LEAP_MONTH_LENGTH[l]);
1189                }
1190            } else {
1191                orgStartMonthLengths = new Integer[MONTH_LENGTH.length];
1192                for (int l = 0; l < MONTH_LENGTH.length; l++) {
1193                    orgStartMonthLengths[l] = new Integer(MONTH_LENGTH[l]);
1194                }
1195            }
1196        }
1197
1198        Integer[] newStartMonthLengths = new Integer[orgStartMonthLengths.length];
1199
1200        for (int month = 0; month < 12; month++) {
1201            if (month == startMonth) {
1202                newStartMonthLengths[month] = new Integer(
1203                        orgStartMonthLengths[month].intValue() - offset);
1204            } else {
1205                newStartMonthLengths[month] = new Integer(
1206                        orgStartMonthLengths[month].intValue());
1207            }
1208        }
1209
1210        ADJUSTED_MONTH_LENGTHS.put(new Integer(startYear), newStartMonthLengths);
1211
1212        if (startYear != endYear) {
1213            // System.out.println("over year");
1214            // Adjusting starting 30 year cycle.
1215            int sCycleNumber = (startYear - 1) / 30;
1216            int sYearInCycle = (startYear - 1) % 30; // 0-based.
1217            Integer[] startCycles = ADJUSTED_CYCLE_YEARS.get(new Integer(
1218                    sCycleNumber));
1219            if (startCycles == null) {
1220                startCycles = new Integer[CYCLEYEAR_START_DATE.length];
1221                for (int j = 0; j < startCycles.length; j++) {
1222                    startCycles[j] = new Integer(CYCLEYEAR_START_DATE[j]);
1223                }
1224            }
1225
1226            for (int j = sYearInCycle + 1; j < CYCLEYEAR_START_DATE.length; j++) {
1227                startCycles[j] = new Integer(startCycles[j].intValue() - offset);
1228            }
1229
1230            // System.out.println(sCycleNumber + ":" + sYearInCycle);
1231            ADJUSTED_CYCLE_YEARS.put(new Integer(sCycleNumber), startCycles);
1232
1233            int sYearInMaxY = (startYear - 1) / 30;
1234            int sEndInMaxY = (endYear - 1) / 30;
1235
1236            if (sYearInMaxY != sEndInMaxY) {
1237                // System.out.println("over 30");
1238                // Adjusting starting 30 * MAX_ADJUSTED_CYCLE year cycle.
1239                // System.out.println(sYearInMaxY);
1240
1241                for (int j = sYearInMaxY + 1; j < ADJUSTED_CYCLES.length; j++) {
1242                    ADJUSTED_CYCLES[j] = new Long(ADJUSTED_CYCLES[j].longValue()
1243                            - offset);
1244                }
1245
1246                // Adjusting ending 30 * MAX_ADJUSTED_CYCLE year cycles.
1247                for (int j = sEndInMaxY + 1; j < ADJUSTED_CYCLES.length; j++) {
1248                    ADJUSTED_CYCLES[j] = new Long(ADJUSTED_CYCLES[j].longValue()
1249                            + offset);
1250                }
1251            }
1252
1253            // Adjusting ending 30 year cycle.
1254            int eCycleNumber = (endYear - 1) / 30;
1255            int sEndInCycle = (endYear - 1) % 30; // 0-based.
1256            Integer[] endCycles = ADJUSTED_CYCLE_YEARS.get(new Integer(
1257                    eCycleNumber));
1258            if (endCycles == null) {
1259                endCycles = new Integer[CYCLEYEAR_START_DATE.length];
1260                for (int j = 0; j < endCycles.length; j++) {
1261                    endCycles[j] = new Integer(CYCLEYEAR_START_DATE[j]);
1262                }
1263            }
1264            for (int j = sEndInCycle + 1; j < CYCLEYEAR_START_DATE.length; j++) {
1265                endCycles[j] = new Integer(endCycles[j].intValue() + offset);
1266            }
1267            ADJUSTED_CYCLE_YEARS.put(new Integer(eCycleNumber), endCycles);
1268        }
1269
1270        // Adjusting ending year.
1271        boolean isEndYLeap = isLeapYear(endYear);
1272
1273        Integer[] orgEndMonthDays = ADJUSTED_MONTH_DAYS.get(new Integer(endYear));
1274
1275        if (orgEndMonthDays == null) {
1276            if (isEndYLeap) {
1277                orgEndMonthDays = new Integer[LEAP_NUM_DAYS.length];
1278                for (int l = 0; l < LEAP_NUM_DAYS.length; l++) {
1279                    orgEndMonthDays[l] = new Integer(LEAP_NUM_DAYS[l]);
1280                }
1281            } else {
1282                orgEndMonthDays = new Integer[NUM_DAYS.length];
1283                for (int l = 0; l < NUM_DAYS.length; l++) {
1284                    orgEndMonthDays[l] = new Integer(NUM_DAYS[l]);
1285                }
1286            }
1287        }
1288
1289        Integer[] newEndMonthDays = new Integer[orgEndMonthDays.length];
1290
1291        for (int month = 0; month < 12; month++) {
1292            if (month > endMonth) {
1293                newEndMonthDays[month] = new Integer(orgEndMonthDays[month]
1294                        .intValue()
1295                        + offset);
1296            } else {
1297                newEndMonthDays[month] = new Integer(orgEndMonthDays[month]
1298                        .intValue());
1299            }
1300        }
1301
1302        ADJUSTED_MONTH_DAYS.put(new Integer(endYear), newEndMonthDays);
1303
1304        // Adjusting the days of month.
1305        Integer[] orgEndMonthLengths = ADJUSTED_MONTH_LENGTHS.get(new Integer(
1306                endYear));
1307
1308        if (orgEndMonthLengths == null) {
1309            if (isEndYLeap) {
1310                orgEndMonthLengths = new Integer[LEAP_MONTH_LENGTH.length];
1311                for (int l = 0; l < LEAP_MONTH_LENGTH.length; l++) {
1312                    orgEndMonthLengths[l] = new Integer(LEAP_MONTH_LENGTH[l]);
1313                }
1314            } else {
1315                orgEndMonthLengths = new Integer[MONTH_LENGTH.length];
1316                for (int l = 0; l < MONTH_LENGTH.length; l++) {
1317                    orgEndMonthLengths[l] = new Integer(MONTH_LENGTH[l]);
1318                }
1319            }
1320        }
1321
1322        Integer[] newEndMonthLengths = new Integer[orgEndMonthLengths.length];
1323
1324        for (int month = 0; month < 12; month++) {
1325            if (month == endMonth) {
1326                newEndMonthLengths[month] = new Integer(
1327                        orgEndMonthLengths[month].intValue() + offset);
1328            } else {
1329                newEndMonthLengths[month] = new Integer(
1330                        orgEndMonthLengths[month].intValue());
1331            }
1332        }
1333
1334        ADJUSTED_MONTH_LENGTHS.put(new Integer(endYear), newEndMonthLengths);
1335
1336        Integer[] startMonthLengths = ADJUSTED_MONTH_LENGTHS.get(new Integer(
1337                startYear));
1338        Integer[] endMonthLengths = ADJUSTED_MONTH_LENGTHS.get(new Integer(
1339                endYear));
1340        Integer[] startMonthDays = ADJUSTED_MONTH_DAYS
1341                .get(new Integer(startYear));
1342        Integer[] endMonthDays = ADJUSTED_MONTH_DAYS.get(new Integer(endYear));
1343
1344        int startMonthLength = startMonthLengths[startMonth].intValue();
1345        int endMonthLength = endMonthLengths[endMonth].intValue();
1346        int startMonthDay = startMonthDays[11].intValue()
1347                + startMonthLengths[11].intValue();
1348        int endMonthDay = endMonthDays[11].intValue()
1349                + endMonthLengths[11].intValue();
1350
1351        int maxMonthLength = ADJUSTED_MAX_VALUES[POSITION_DAY_OF_MONTH]
1352                .intValue();
1353        int leastMaxMonthLength = ADJUSTED_LEAST_MAX_VALUES[POSITION_DAY_OF_MONTH]
1354                .intValue();
1355
1356        if (maxMonthLength < startMonthLength) {
1357            maxMonthLength = startMonthLength;
1358        }
1359        if (maxMonthLength < endMonthLength) {
1360            maxMonthLength = endMonthLength;
1361        }
1362        ADJUSTED_MAX_VALUES[POSITION_DAY_OF_MONTH] = new Integer(maxMonthLength);
1363
1364        if (leastMaxMonthLength > startMonthLength) {
1365            leastMaxMonthLength = startMonthLength;
1366        }
1367        if (leastMaxMonthLength > endMonthLength) {
1368            leastMaxMonthLength = endMonthLength;
1369        }
1370        ADJUSTED_LEAST_MAX_VALUES[POSITION_DAY_OF_MONTH] = new Integer(
1371                leastMaxMonthLength);
1372
1373        int maxMonthDay = ADJUSTED_MAX_VALUES[POSITION_DAY_OF_YEAR].intValue();
1374        int leastMaxMonthDay = ADJUSTED_LEAST_MAX_VALUES[POSITION_DAY_OF_YEAR]
1375                .intValue();
1376
1377        if (maxMonthDay < startMonthDay) {
1378            maxMonthDay = startMonthDay;
1379        }
1380        if (maxMonthDay < endMonthDay) {
1381            maxMonthDay = endMonthDay;
1382        }
1383
1384        ADJUSTED_MAX_VALUES[POSITION_DAY_OF_YEAR] = new Integer(maxMonthDay);
1385
1386        if (leastMaxMonthDay > startMonthDay) {
1387            leastMaxMonthDay = startMonthDay;
1388        }
1389        if (leastMaxMonthDay > endMonthDay) {
1390            leastMaxMonthDay = endMonthDay;
1391        }
1392        ADJUSTED_LEAST_MAX_VALUES[POSITION_DAY_OF_YEAR] = new Integer(
1393                leastMaxMonthDay);
1394    }

1396    /**
1397     * Read hijrah_deviation.cfg file. The config file contains the deviation data with
1398     * following format.
1399     *
1400     * StartYear/StartMonth(0-based)-EndYear/EndMonth(0-based):Deviation day (1,
1401     * 2, -1, or -2)
1402     *
1403     * Line separator or ";" is used for the separator of each deviation data.
1404     *
1405     * Here is the example.
1406     *
1407     * 1429/0-1429/1:1
1408     * 1429/2-1429/7:1;1429/6-1429/11:1
1409     * 1429/11-9999/11:1
1410     *
1411     * @throws IOException for zip/jar file handling exception.
1412     * @throws ParseException if the format of the configuration file is wrong.
1413     */
1414    private static void readDeviationConfig() throws IOException, ParseException {
1415        InputStream is = getConfigFileInputStream();
1416        if (is != null) {
1417            try (BufferedReader br = new BufferedReader(new InputStreamReader(is))) {
1418                String line = "";
1419                int num = 0;
1420                while ((line = br.readLine()) != null) {
1421                    num++;
1422                    line = line.trim();
1423                    parseLine(line, num);
1424                }
1425            }
1426        }
1427    }

1429    /**
1430     * Parse each deviation element.
1431     *
1432     * @param line  a line to parse
1433     * @param num  line number
1434     * @throws ParseException if line has incorrect format.
1435     */
1436    private static void parseLine(String line, int num) throws ParseException {
1437        StringTokenizer st = new StringTokenizer(line, ";");
1438        while (st.hasMoreTokens()) {
1439            String deviationElement = st.nextToken();
1440            int offsetIndex = deviationElement.indexOf(':');
1441            if (offsetIndex != -1) {
1442                String offsetString = deviationElement.substring(
1443                        offsetIndex + 1, deviationElement.length());
1444                int offset;
1445                try {
1446                    offset = Integer.parseInt(offsetString);
1447                } catch (NumberFormatException ex) {
1448                    throw new ParseException(
1449                            "Offset is not properly set at line " + num + ".",
1450                            num);
1451                }
1452                int separatorIndex = deviationElement.indexOf('-');
1453                if (separatorIndex != -1) {
1454                    String startDateStg = deviationElement.substring(0,
1455                            separatorIndex);
1456                    String endDateStg = deviationElement.substring(
1457                            separatorIndex + 1, offsetIndex);
1458                    int startDateYearSepIndex = startDateStg.indexOf('/');
1459                    int endDateYearSepIndex = endDateStg.indexOf('/');
1460                    int startYear = -1;
1461                    int endYear = -1;
1462                    int startMonth = -1;
1463                    int endMonth = -1;
1464                    if (startDateYearSepIndex != -1) {
1465                        String startYearStg = startDateStg.substring(0,
1466                                startDateYearSepIndex);
1467                        String startMonthStg = startDateStg.substring(
1468                                startDateYearSepIndex + 1, startDateStg
1469                                        .length());
1470                        try {
1471                            startYear = Integer.parseInt(startYearStg);
1472                        } catch (NumberFormatException ex) {
1473                            throw new ParseException(
1474                                    "Start year is not properly set at line "
1475                                            + num + ".", num);
1476                        }
1477                        try {
1478                            startMonth = Integer.parseInt(startMonthStg);
1479                        } catch (NumberFormatException ex) {
1480                            throw new ParseException(
1481                                    "Start month is not properly set at line "
1482                                            + num + ".", num);
1483                        }
1484                    } else {
1485                        throw new ParseException(
1486                                "Start year/month has incorrect format at line "
1487                                        + num + ".", num);
1488                    }
1489                    if (endDateYearSepIndex != -1) {
1490                        String endYearStg = endDateStg.substring(0,
1491                                endDateYearSepIndex);
1492                        String endMonthStg = endDateStg.substring(
1493                                endDateYearSepIndex + 1, endDateStg.length());
1494                        try {
1495                            endYear = Integer.parseInt(endYearStg);
1496                        } catch (NumberFormatException ex) {
1497                            throw new ParseException(
1498                                    "End year is not properly set at line "
1499                                            + num + ".", num);
1500                        }
1501                        try {
1502                            endMonth = Integer.parseInt(endMonthStg);
1503                        } catch (NumberFormatException ex) {
1504                            throw new ParseException(
1505                                    "End month is not properly set at line "
1506                                            + num + ".", num);
1507                        }
1508                    } else {
1509                        throw new ParseException(
1510                                "End year/month has incorrect format at line "
1511                                        + num + ".", num);
1512                    }
1513                    if (startYear != -1 && startMonth != -1 && endYear != -1
1514                            && endMonth != -1) {
1515                        addDeviationAsHijrah(startYear, startMonth, endYear,
1516                                endMonth, offset);
1517                    } else {
1518                        throw new ParseException("Unknown error at line " + num
1519                                + ".", num);
1520                    }
1521                } else {
1522                    throw new ParseException(
1523                            "Start and end year/month has incorrect format at line "
1524                                    + num + ".", num);
1525                }
1526            } else {
1527                throw new ParseException("Offset has incorrect format at line "
1528                        + num + ".", num);
1529            }
1530        }
1531    }

1533    /**
1534     * Return InputStream for deviation configuration file.
1535     * The default location of the deviation file is:
1536     * <pre>
1537     *   $CLASSPATH/org/threeten/bp/chrono
1538     * </pre>
1539     * And the default file name is:
1540     * <pre>
1541     *   hijrah_deviation.cfg
1542     * </pre>
1543     * The default location and file name can be overriden by setting
1544     * following two Java's system property.
1545     * <pre>
1546     *   Location: org.threeten.bp.i18n.HijrahDate.deviationConfigDir
1547     *   File name: org.threeten.bp.i18n.HijrahDate.deviationConfigFile
1548     * </pre>
1549     * Regarding the file format, see readDeviationConfig() method for details.
1550     *
1551     * @return InputStream for file reading exception.
1552     * @throws IOException for zip/jar file handling exception.
1553     */
1554    private static InputStream getConfigFileInputStream() throws IOException {
1555
1556        String fileName = System
1557                .getProperty("org.threeten.bp.i18n.HijrahDate.deviationConfigFile");
1558
1559        if (fileName == null) {
1560            fileName = DEFAULT_CONFIG_FILENAME;
1561        }
1562
1563        String dir = System
1564                .getProperty("org.threeten.bp.i18n.HijrahDate.deviationConfigDir");
1565
1566        if (dir != null) {
1567            if (!(dir.length() == 0 && dir.endsWith(System
1568                    .getProperty("file.separator")))) {
1569                dir = dir + System.getProperty("file.separator");
1570            }
1571            File file = new File(dir + FILE_SEP + fileName);
1572            if (file.exists()) {
1573                try {
1574                    return new FileInputStream(file);
1575                } catch (IOException ioe) {
1576                    throw ioe;
1577                }
1578            } else {
1579                return null;
1580            }
1581        } else {
1582            String classPath = System.getProperty("java.class.path");
1583            StringTokenizer st = new StringTokenizer(classPath, PATH_SEP);
1584            while (st.hasMoreTokens()) {
1585                String path = st.nextToken();
1586                File file = new File(path);
1587                if (file.exists()) {
1588                    if (file.isDirectory()) {
1589                        File f = new File(
1590                                path + FILE_SEP + DEFAULT_CONFIG_PATH, fileName);
1591                        if (f.exists()) {
1592                            try {
1593                                return new FileInputStream(path + FILE_SEP
1594                                        + DEFAULT_CONFIG_PATH + FILE_SEP
1595                                        + fileName);
1596                            } catch (IOException ioe) {
1597                                throw ioe;
1598                            }
1599                        }
1600                    } else {
1601                        ZipFile zip;
1602                        try {
1603                            zip = new ZipFile(file);
1604                        } catch (IOException ioe) {
1605                            zip = null;
1606                        }
1607
1608                        if (zip != null) {
1609                            String targetFile = DEFAULT_CONFIG_PATH + FILE_SEP
1610                                    + fileName;
1611                            ZipEntry entry = zip.getEntry(targetFile);
1612
1613                            if (entry == null) {
1614                                if (FILE_SEP == '/') {
1615                                    targetFile = targetFile.replace('/', '\\');
1616                                } else if (FILE_SEP == '\\') {
1617                                    targetFile = targetFile.replace('\\', '/');
1618                                }
1619                                entry = zip.getEntry(targetFile);
1620                            }
1621
1622                            if (entry != null) {
1623                                try {
1624                                    return zip.getInputStream(entry);
1625                                } catch (IOException ioe) {
1626                                    throw ioe;
1627                                }
1628                            }
1629                        }
1630                    }
1631                }
1632            }
1633            return null;
1634        }
1635    }
