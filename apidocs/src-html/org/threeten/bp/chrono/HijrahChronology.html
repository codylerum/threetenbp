052/**
053 * The Hijrah calendar system.
054 * <p>
055 * This chronology defines the rules of the Hijrah calendar system.
056 * <p>
057 * The implementation follows the Freeman-Grenville algorithm (*1) and has following features.
058 * <p><ul>
059 * <li>A year has 12 months.</li>
060 * <li>Over a cycle of 30 years there are 11 leap years.</li>
061 * <li>There are 30 days in month number 1, 3, 5, 7, 9, and 11,
062 * and 29 days in month number 2, 4, 6, 8, 10, and 12.</li>
063 * <li>In a leap year month 12 has 30 days.</li>
064 * <li>In a 30 year cycle, year 2, 5, 7, 10, 13, 16, 18, 21, 24,
065 * 26, and 29 are leap years.</li>
066 * <li>Total of 10631 days in a 30 years cycle.</li>
067 * </ul><p>
068 * <P>
069 * The table shows the features described above.
070 * <blockquote>
071 * <table border="1">
072 *   <tbody>
073 *     <tr>
074 *       <th># of month</th>
075 *       <th>Name of month</th>
076 *       <th>Number of days</th>
077 *     </tr>
078 *     <tr>
079 *       <td>1</td>
080 *       <td>Muharram</td>
081 *       <td>30</td>
082 *     </tr>
083 *     <tr>
084 *       <td>2</td>
085 *       <td>Safar</td>
086 *       <td>29</td>
087 *     </tr>
088 *     <tr>
089 *       <td>3</td>
090 *       <td>Rabi'al-Awwal</td>
091 *       <td>30</td>
092 *     </tr>
093 *     <tr>
094 *       <td>4</td>
095 *       <td>Rabi'ath-Thani</td>
096 *       <td>29</td>
097 *     </tr>
098 *     <tr>
099 *       <td>5</td>
100 *       <td>Jumada l-Ula</td>
101 *       <td>30</td>
102 *     </tr>
103 *     <tr>
104 *       <td>6</td>
105 *       <td>Jumada t-Tania</td>
106 *       <td>29</td>
107 *     </tr>
108 *     <tr>
109 *       <td>7</td>
110 *       <td>Rajab</td>
111 *       <td>30</td>
112 *     </tr>
113 *     <tr>
114 *       <td>8</td>
115 *       <td>Sha`ban</td>
116 *       <td>29</td>
117 *     </tr>
118 *     <tr>
119 *       <td>9</td>
120 *       <td>Ramadan</td>
121 *       <td>30</td>
122 *     </tr>
123 *     <tr>
124 *       <td>10</td>
125 *       <td>Shawwal</td>
126 *       <td>29</td>
127 *     </tr>
128 *     <tr>
129 *       <td>11</td>
130 *       <td>Dhu 'l-Qa`da</td>
131 *       <td>30</td>
132 *     </tr>
133 *     <tr>
134 *       <td>12</td>
135 *       <td>Dhu 'l-Hijja</td>
136 *       <td>29, but 30 days in years 2, 5, 7, 10,<br>
137 * 13, 16, 18, 21, 24, 26, and 29</td>
138 *     </tr>
139 *   </tbody>
140 * </table>
141 * </blockquote>
142 * <p>
143 * (*1) The algorithm is taken from the book,
144 * The Muslim and Christian Calendars by G.S.P. Freeman-Grenville.
145 * <p>
146 *
147 * <h3>Specification for implementors</h3>
148 * This class is immutable and thread-safe.
149 */
150public final class HijrahChronology extends Chronology implements Serializable {
151
152    /**
153     * Singleton instance of the Hijrah chronology.
154     */
155    public static final HijrahChronology INSTANCE = new HijrahChronology();
156
157    /**
158     * The singleton instance for the era before the current one - Before Hijrah -
159     * which has the value 0.
160     */
161    public static final Era ERA_BEFORE_AH = HijrahEra.BEFORE_AH;
162    /**
163     * The singleton instance for the current era - Hijrah - which has the value 1.
164     */
165    public static final Era ERA_AH = HijrahEra.AH;
166    /**
167     * Serialization version.
168     */
169    private static final long serialVersionUID = 3127340209035924785L;
170    /**
171     * Narrow names for eras.
172     */
173    private static final HashMap<String, String[]> ERA_NARROW_NAMES = new HashMap<>();
174    /**
175     * Short names for eras.
176     */
177    private static final HashMap<String, String[]> ERA_SHORT_NAMES = new HashMap<>();
178    /**
179     * Full names for eras.
180     */
181    private static final HashMap<String, String[]> ERA_FULL_NAMES = new HashMap<>();
182    /**
183     * Fallback language for the era names.
184     */
185    private static final String FALLBACK_LANGUAGE = "en";
186
187    /**
188     * Language that has the era names.

190    //private static final String TARGET_LANGUAGE = "ar";
191    /**
192     * Name data.
193     */
194    static {
195        ERA_NARROW_NAMES.put(FALLBACK_LANGUAGE, new String[]{"BH", "HE"});
196        ERA_SHORT_NAMES.put(FALLBACK_LANGUAGE, new String[]{"B.H.", "H.E."});
197        ERA_FULL_NAMES.put(FALLBACK_LANGUAGE, new String[]{"Before Hijrah", "Hijrah Era"});
198    }

200    /**
201     * Restrictive constructor.
202     */
203    private HijrahChronology() {
204    }
205
206    /**
207     * Resolve singleton.
208     *
209     * @return the singleton instance, not null
210     */
211    private Object readResolve() {
212        return INSTANCE;
213    }
214
215    //-----------------------------------------------------------------------
216    /**
217     * Gets the ID of the chronology - 'Hijrah'.
218     * <p>
219     * The ID uniquely identifies the {@code Chrono}.
220     * It can be used to lookup the {@code Chrono} using {@link #of(String)}.
221     *
222     * @return the chronology ID - 'Hijrah'
223     * @see #getCalendarType()
224     */
225    @Override
226    public String getId() {
227        return "Hijrah";
228    }
229
230    /**
231     * Gets the calendar type of the underlying calendar system - 'islamicc'.
232     * <p>
233     * The calendar type is an identifier defined by the
234     * <em>Unicode Locale Data Markup Language (LDML)</em> specification.
235     * It can be used to lookup the {@code Chrono} using {@link #of(String)}.
236     * It can also be used as part of a locale, accessible via
237     * {@link Locale#getUnicodeLocaleType(String)} with the key 'ca'.
238     *
239     * @return the calendar system type - 'islamicc'
240     * @see #getId()
241     */
242    @Override
243    public String getCalendarType() {
244        return "islamicc";
245    }

247    //-----------------------------------------------------------------------
248    @Override  // override with covariant return type
249    public HijrahDate date(Era era, int yearOfEra, int month, int dayOfMonth) {
250        return (HijrahDate) super.date(era, yearOfEra, month, dayOfMonth);
251    }
252
253    @Override  // override with covariant return type
254    public HijrahDate date(int prolepticYear, int month, int dayOfMonth) {
255        return HijrahDate.of(prolepticYear, month, dayOfMonth);
256    }
257
258    @Override  // override with covariant return type
259    public HijrahDate dateYearDay(Era era, int yearOfEra, int dayOfYear) {
260        return (HijrahDate) super.dateYearDay(era, yearOfEra, dayOfYear);
261    }
262
263    @Override  // override with covariant return type
264    public HijrahDate dateYearDay(int prolepticYear, int dayOfYear) {
265        return HijrahDate.of(prolepticYear, 1, 1).plusDays(dayOfYear - 1);  // TODO better
266    }
267
268    //-----------------------------------------------------------------------
269    @Override  // override with covariant return type
270    public HijrahDate date(TemporalAccessor temporal) {
271        if (temporal instanceof HijrahDate) {
272            return (HijrahDate) temporal;
273        }
274        return HijrahDate.ofEpochDay(temporal.getLong(EPOCH_DAY));
275    }
276
277    @Override  // override with covariant return type
278    public ChronoLocalDateTime<HijrahDate> localDateTime(TemporalAccessor temporal) {
279        return (ChronoLocalDateTime<HijrahDate>) super.localDateTime(temporal);
280    }
281
282    @Override  // override with covariant return type
283    public ChronoZonedDateTime<HijrahDate> zonedDateTime(TemporalAccessor temporal) {
284        return (ChronoZonedDateTime<HijrahDate>) super.zonedDateTime(temporal);
285    }
286
287    @Override  // override with covariant return type
288    public ChronoZonedDateTime<HijrahDate> zonedDateTime(Instant instant, ZoneId zone) {
289        return (ChronoZonedDateTime<HijrahDate>) super.zonedDateTime(instant, zone);
290    }
291
292    //-----------------------------------------------------------------------
293    @Override  // override with covariant return type
294    public HijrahDate dateNow() {
295        return (HijrahDate) super.dateNow();
296    }
297
298    @Override  // override with covariant return type
299    public HijrahDate dateNow(ZoneId zone) {
300        return (HijrahDate) super.dateNow(zone);
301    }
302
303    @Override  // override with covariant return type
304    public HijrahDate dateNow(Clock clock) {
305        Objects.requireNonNull(clock, "clock");
306        return (HijrahDate) super.dateNow(clock);
307    }
308
309    //-----------------------------------------------------------------------
310    @Override
311    public boolean isLeapYear(long prolepticYear) {
312        return HijrahDate.isLeapYear(prolepticYear);
313    }
314
315    @Override
316    public int prolepticYear(Era era, int yearOfEra) {
317        if (era instanceof HijrahEra == false) {
318            throw new DateTimeException("Era must be HijrahEra");
319        }
320        return (era == HijrahEra.AH ? yearOfEra : 1 - yearOfEra);
321    }
322
323    @Override
324    public Era eraOf(int eraValue) {
325        switch (eraValue) {
326            case 0:
327                return HijrahEra.BEFORE_AH;
328            case 1:
329                return HijrahEra.AH;
330            default:
331                throw new DateTimeException("invalid Hijrah era");
332        }
333    }
334
335    @Override
336    public List<Era> eras() {
337        return Arrays.<Era>asList(HijrahEra.values());
338    }
339
340    //-----------------------------------------------------------------------
341    @Override
342    public ValueRange range(ChronoField field) {
343        return field.range();
344    }
