051/**
052 * The Minguo calendar system.
053 * <p>
054 * This chronology defines the rules of the Minguo calendar system.
055 * This calendar system is primarily used in the Republic of China, often known as Taiwan.
056 * Dates are aligned such that {@code 0001-01-01 (Minguo)} is {@code 1912-01-01 (ISO)}.
057 * <p>
058 * The fields are defined as follows:
059 * <p><ul>
060 * <li>era - There are two eras, the current 'Republic' (ERA_ROC) and the previous era (ERA_BEFORE_ROC).
061 * <li>year-of-era - The year-of-era for the current era increases uniformly from the epoch at year one.
062 *  For the previous era the year increases from one as time goes backwards.
063 *  The value for the current era is equal to the ISO proleptic-year minus 1911.
064 * <li>proleptic-year - The proleptic year is the same as the year-of-era for the
065 *  current era. For the previous era, years have zero, then negative values.
066 *  The value is equal to the ISO proleptic-year minus 1911.
067 * <li>month-of-year - The Minguo month-of-year exactly matches ISO.
068 * <li>day-of-month - The Minguo day-of-month exactly matches ISO.
069 * <li>day-of-year - The Minguo day-of-year exactly matches ISO.
070 * <li>leap-year - The Minguo leap-year pattern exactly matches ISO, such that the two calendars
071 *  are never out of step.
072 * </ul><p>
073 *
074 * <h3>Specification for implementors</h3>
075 * This class is immutable and thread-safe.
076 */
077public final class MinguoChronology extends Chronology implements Serializable {
078
079    /**
080     * Singleton instance for the Minguo chronology.
081     */
082    public static final MinguoChronology INSTANCE = new MinguoChronology();
083
084    /**
085     * The singleton instance for the era ROC.
086     */
087    public static final Era ERA_ROC = MinguoEra.ROC;
088
089    /**
090     * The singleton instance for the era BEFORE_ROC.
091     */
092    public static final Era ERA_BEFORE_ROC = MinguoEra.BEFORE_ROC;
093
094    /**
095     * Serialization version.
096     */
097    private static final long serialVersionUID = 1039765215346859963L;
098    /**
099     * The difference in years between ISO and Minguo.
100     */
101    static final int YEARS_DIFFERENCE = 1911;
102
103    /**
104     * Restricted constructor.
105     */
106    private MinguoChronology() {
107    }
108
109    /**
110     * Resolve singleton.
111     *
112     * @return the singleton instance, not null
113     */
114    private Object readResolve() {
115        return INSTANCE;
116    }
117
118    //-----------------------------------------------------------------------
119    /**
120     * Gets the ID of the chronology - 'Minguo'.
121     * <p>
122     * The ID uniquely identifies the {@code Chrono}.
123     * It can be used to lookup the {@code Chrono} using {@link #of(String)}.
124     *
125     * @return the chronology ID - 'Minguo'
126     * @see #getCalendarType()
127     */
128    @Override
129    public String getId() {
130        return "Minguo";
131    }
132
133    /**
134     * Gets the calendar type of the underlying calendar system - 'roc'.
135     * <p>
136     * The calendar type is an identifier defined by the
137     * <em>Unicode Locale Data Markup Language (LDML)</em> specification.
138     * It can be used to lookup the {@code Chrono} using {@link #of(String)}.
139     * It can also be used as part of a locale, accessible via
140     * {@link Locale#getUnicodeLocaleType(String)} with the key 'ca'.
141     *
142     * @return the calendar system type - 'roc'
143     * @see #getId()
144     */
145    @Override
146    public String getCalendarType() {
147        return "roc";
148    }
149
150    //-----------------------------------------------------------------------
151    @Override  // override with covariant return type
152    public MinguoDate date(Era era, int yearOfEra, int month, int dayOfMonth) {
153        return (MinguoDate) super.date(era, yearOfEra, month, dayOfMonth);
154    }
155
156    @Override  // override with covariant return type
157    public MinguoDate date(int prolepticYear, int month, int dayOfMonth) {
158        return new MinguoDate(LocalDate.of(prolepticYear + YEARS_DIFFERENCE, month, dayOfMonth));
159    }
160
161    @Override  // override with covariant return type
162    public MinguoDate dateYearDay(Era era, int yearOfEra, int dayOfYear) {
163        return (MinguoDate) super.dateYearDay(era, yearOfEra, dayOfYear);
164    }
165
166    @Override  // override with covariant return type
167    public MinguoDate dateYearDay(int prolepticYear, int dayOfYear) {
168        return new MinguoDate(LocalDate.ofYearDay(prolepticYear + YEARS_DIFFERENCE, dayOfYear));
169    }
170
171    //-----------------------------------------------------------------------
172    @Override  // override with covariant return type
173    public MinguoDate date(TemporalAccessor temporal) {
174        if (temporal instanceof MinguoDate) {
175            return (MinguoDate) temporal;
176        }
177        return new MinguoDate(LocalDate.from(temporal));
178    }
179
180    @Override  // override with covariant return type
181    public ChronoLocalDateTime<MinguoDate> localDateTime(TemporalAccessor temporal) {
182        return (ChronoLocalDateTime<MinguoDate>) super.localDateTime(temporal);
183    }
184
185    @Override  // override with covariant return type
186    public ChronoZonedDateTime<MinguoDate> zonedDateTime(TemporalAccessor temporal) {
187        return (ChronoZonedDateTime<MinguoDate>) super.zonedDateTime(temporal);
188    }
189
190    @Override  // override with covariant return type
191    public ChronoZonedDateTime<MinguoDate> zonedDateTime(Instant instant, ZoneId zone) {
192        return (ChronoZonedDateTime<MinguoDate>) super.zonedDateTime(instant, zone);
193    }
194
195    //-----------------------------------------------------------------------
196    @Override  // override with covariant return type
197    public MinguoDate dateNow() {
198        return (MinguoDate) super.dateNow();
199    }
200
201    @Override  // override with covariant return type
202    public MinguoDate dateNow(ZoneId zone) {
203        return (MinguoDate) super.dateNow(zone);
204    }
205
206    @Override  // override with covariant return type
207    public MinguoDate dateNow(Clock clock) {
208        Objects.requireNonNull(clock, "clock");
209        return (MinguoDate) super.dateNow(clock);
210    }
211
212    //-----------------------------------------------------------------------
213    /**
214     * Checks if the specified year is a leap year.
215     * <p>
216     * Minguo leap years occur exactly in line with ISO leap years.
217     * This method does not validate the year passed in, and only has a
218     * well-defined result for years in the supported range.
219     *
220     * @param prolepticYear  the proleptic-year to check, not validated for range
221     * @return true if the year is a leap year
222     */
223    @Override
224    public boolean isLeapYear(long prolepticYear) {
225        return IsoChronology.INSTANCE.isLeapYear(prolepticYear + YEARS_DIFFERENCE);
226    }
227
228    @Override
229    public int prolepticYear(Era era, int yearOfEra) {
230        if (era instanceof MinguoEra == false) {
231            throw new DateTimeException("Era must be MinguoEra");
232        }
233        return (era == MinguoEra.ROC ? yearOfEra : 1 - yearOfEra);
234    }
235
236    @Override
237    public Era eraOf(int eraValue) {
238        return MinguoEra.of(eraValue);
239    }
240
241    @Override
242    public List<Era> eras() {
243        return Arrays.<Era>asList(MinguoEra.values());
244    }
245
246    //-----------------------------------------------------------------------
247    @Override
248    public ValueRange range(ChronoField field) {
249        switch (field) {
250            case YEAR_OF_ERA: {
251                ValueRange range = YEAR.range();
252                return ValueRange.of(1, range.getMaximum() - YEARS_DIFFERENCE, -range.getMinimum() + 1 + YEARS_DIFFERENCE);
253            }
254            case YEAR: {
255                ValueRange range = YEAR.range();
256                return ValueRange.of(range.getMinimum() - YEARS_DIFFERENCE, range.getMaximum() - YEARS_DIFFERENCE);
257            }
258        }
259        return field.range();
260    }
