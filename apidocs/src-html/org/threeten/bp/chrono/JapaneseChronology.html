056/**
057 * The Japanese Imperial calendar system.
058 * <p>
059 * This chronology defines the rules of the Japanese Imperial calendar system.
060 * This calendar system is primarily used in Japan.
061 * The Japanese Imperial calendar system is the same as the ISO calendar system
062 * apart from the era-based year numbering.
063 * <p>
064 * Only Meiji (1865-04-07 - 1868-09-07) and later eras are supported.
065 * Older eras are handled as an unknown era where the year-of-era is the ISO year.
066 *
067 * <h3>Specification for implementors</h3>
068 * This class is immutable and thread-safe.
069 */
070public final class JapaneseChronology extends Chronology implements Serializable {

269     * well-defined result for years in the supported range.

272     * @return true if the year is a leap year

276        return IsoChronology.INSTANCE.isLeapYear(prolepticYear);
277    }

279    @Override
280    public int prolepticYear(Era era, int yearOfEra) {
281        if (era instanceof JapaneseEra == false) {
282            throw new DateTimeException("Era must be JapaneseEra");
283        }
284        JapaneseEra jera = (JapaneseEra) era;
285        int gregorianYear = jera.getPrivateEra().getSinceDate().getYear() + yearOfEra - 1;
286        if (yearOfEra == 1) {
287            return gregorianYear;
288        }
289        LocalGregorianCalendar.Date jdate = JCAL.newCalendarDate(null);
290        jdate.setEra(jera.getPrivateEra()).setDate(yearOfEra, 1, 1);
291        JCAL.normalize(jdate);
292        if (jdate.getNormalizedYear() == gregorianYear) {
293            return gregorianYear;
294        }
295        throw new DateTimeException("invalid yearOfEra value");
296    }

320    //-----------------------------------------------------------------------
321    @Override
322    public ValueRange range(ChronoField field) {
323        switch (field) {
324            case DAY_OF_MONTH:
325            case DAY_OF_WEEK:
326            case MICRO_OF_DAY:
327            case MICRO_OF_SECOND:
328            case HOUR_OF_DAY:
329            case HOUR_OF_AMPM:
330            case MINUTE_OF_DAY:
331            case MINUTE_OF_HOUR:
332            case SECOND_OF_DAY:
333            case SECOND_OF_MINUTE:
334            case MILLI_OF_DAY:
335            case MILLI_OF_SECOND:
336            case NANO_OF_DAY:
337            case NANO_OF_SECOND:
338            case CLOCK_HOUR_OF_DAY:
339            case CLOCK_HOUR_OF_AMPM:
340            case EPOCH_DAY:
341            case EPOCH_MONTH:
342                return field.range();
343        }
344        Calendar jcal = Calendar.getInstance(LOCALE);
345        int fieldIndex;
346        switch (field) {
347            case ERA:
348                return ValueRange.of(jcal.getMinimum(Calendar.ERA) - JapaneseEra.ERA_OFFSET,
349                                             jcal.getMaximum(Calendar.ERA) - JapaneseEra.ERA_OFFSET);
350            case YEAR:
351            case YEAR_OF_ERA:
352                return ValueRange.of(Year.MIN_VALUE, jcal.getGreatestMinimum(Calendar.YEAR),
353                                             jcal.getLeastMaximum(Calendar.YEAR), Year.MAX_VALUE);
354            case MONTH_OF_YEAR:
355                return ValueRange.of(jcal.getMinimum(Calendar.MONTH) + 1, jcal.getGreatestMinimum(Calendar.MONTH) + 1,
356                                             jcal.getLeastMaximum(Calendar.MONTH) + 1, jcal.getMaximum(Calendar.MONTH) + 1);
357            case DAY_OF_YEAR:
358                fieldIndex = Calendar.DAY_OF_YEAR;
359                break;
360            default:
361                 // TODO: review the remaining fields
362                throw new UnsupportedOperationException("Unimplementable field: " + field);
363        }
364        return ValueRange.of(jcal.getMinimum(fieldIndex), jcal.getGreatestMinimum(fieldIndex),
365                                     jcal.getLeastMaximum(fieldIndex), jcal.getMaximum(fieldIndex));
366    }
