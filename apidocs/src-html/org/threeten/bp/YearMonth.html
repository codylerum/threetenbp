069/**
070 * A year-month in the ISO-8601 calendar system, such as {@code 2007-12}.
071 * <p>
072 * {@code YearMonth} is an immutable date-time object that represents the combination
073 * of a year and month. Any field that can be derived from a year and month, such as
074 * quarter-of-year, can be obtained.
075 * <p>
076 * This class does not store or represent a day, time or time-zone.
077 * For example, the value "October 2007" can be stored in a {@code YearMonth}.
078 * <p>
079 * The ISO-8601 calendar system is the modern civil calendar system used today
080 * in most of the world. It is equivalent to the proleptic Gregorian calendar
081 * system, in which today's rules for leap years are applied for all time.
082 * For most applications written today, the ISO-8601 rules are entirely suitable.
083 * However, any application that makes use of historical dates, and requires them
084 * to be accurate will find the ISO-8601 approach unsuitable.
085 *
086 * <h3>Specification for implementors</h3>
087 * This class is immutable and thread-safe.
088 */
089public final class YearMonth
090        extends DefaultInterfaceTemporalAccessor
091        implements Temporal, TemporalAdjuster, Comparable<YearMonth>, Serializable {

192    /**
193     * Obtains an instance of {@code YearMonth} from a temporal object.
194     * <p>
195     * A {@code TemporalAccessor} represents some form of date and time information.
196     * This factory converts the arbitrary temporal object to an instance of {@code YearMonth}.
197     * <p>
198     * The conversion extracts the {@link ChronoField#YEAR YEAR} and
199     * {@link ChronoField#MONTH_OF_YEAR MONTH_OF_YEAR} fields.
200     * The extraction is only permitted if the temporal object has an ISO
201     * chronology, or can be converted to a {@code LocalDate}.
202     * <p>
203     * This method matches the signature of the functional interface {@link TemporalQuery}
204     * allowing it to be used in queries via method reference, {@code YearMonth::from}.
205     *
206     * @param temporal  the temporal object to convert, not null
207     * @return the year-month, not null
208     * @throws DateTimeException if unable to convert to a {@code YearMonth}
209     */
210    public static YearMonth from(TemporalAccessor temporal) {
211        if (temporal instanceof YearMonth) {
212            return (YearMonth) temporal;
213        }
214        try {
215            if (IsoChronology.INSTANCE.equals(Chronology.from(temporal)) == false) {
216                temporal = LocalDate.from(temporal);
217            }
218            return of(temporal.get(YEAR), temporal.get(MONTH_OF_YEAR));
219        } catch (DateTimeException ex) {
220            throw new DateTimeException("Unable to obtain YearMonth from TemporalAccessor: " + temporal.getClass(), ex);
221        }
222    }

283    /**
284     * Checks if the specified field is supported.
285     * <p>
286     * This checks if this year-month can be queried for the specified field.
287     * If false, then calling the {@link #range(TemporalField) range} and
288     * {@link #get(TemporalField) get} methods will throw an exception.
289     * <p>
290     * If the field is a {@link ChronoField} then the query is implemented here.
291     * The {@link #isSupported(TemporalField) supported fields} will return valid
292     * values based on this date-time.
293     * The supported fields are:
294     * <ul>
295     * <li>{@code MONTH_OF_YEAR}
296     * <li>{@code EPOCH_MONTH}
297     * <li>{@code YEAR_OF_ERA}
298     * <li>{@code YEAR}
299     * <li>{@code ERA}
300     * </ul>
301     * All other {@code ChronoField} instances will return false.
302     * <p>
303     * If the field is not a {@code ChronoField}, then the result of this method
304     * is obtained by invoking {@code TemporalField.isSupportedBy(TemporalAccessor)}
305     * passing {@code this} as the argument.
306     * Whether the field is supported is determined by the field.
307     *
308     * @param field  the field to check, null returns false
309     * @return true if the field is supported on this year-month, false if not
310     */
311    @Override
312    public boolean isSupported(TemporalField field) {
313        if (field instanceof ChronoField) {
314            return field == YEAR || field == MONTH_OF_YEAR ||
315                    field == EPOCH_MONTH || field == YEAR_OF_ERA || field == ERA;
316        }
317        return field != null && field.isSupportedBy(this);
318    }

324     * This year-month is used to enhance the accuracy of the returned range.
325     * If it is not possible to return the range, because the field is not supported

327     * <p>
328     * If the field is a {@link ChronoField} then the query is implemented here.
329     * The {@link #isSupported(TemporalField) supported fields} will return
330     * appropriate range instances.
331     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
332     * <p>
333     * If the field is not a {@code ChronoField}, then the result of this method
334     * is obtained by invoking {@code TemporalField.rangeRefinedBy(TemporalAccessor)}
335     * passing {@code this} as the argument.
336     * Whether the range can be obtained is determined by the field.
337     *
338     * @param field  the field to query the range for, not null
339     * @return the range of valid values for the field, not null

342    @Override

346        }

379    /**
380     * Gets the value of the specified field from this year-month as a {@code long}.
381     * <p>
382     * This queries this year-month for the value for the specified field.
383     * If it is not possible to return the value, because the field is not supported
384     * or for some other reason, an exception is thrown.
385     * <p>
386     * If the field is a {@link ChronoField} then the query is implemented here.
387     * The {@link #isSupported(TemporalField) supported fields} will return valid
388     * values based on this year-month.
389     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
390     * <p>
391     * If the field is not a {@code ChronoField}, then the result of this method
392     * is obtained by invoking {@code TemporalField.getFrom(TemporalAccessor)}
393     * passing {@code this} as the argument. Whether the value can be obtained,
394     * and what the value represents, is determined by the field.
395     *
396     * @param field  the field to get, not null
397     * @return the value for the field
398     * @throws DateTimeException if a value for the field cannot be obtained
399     * @throws ArithmeticException if numeric overflow occurs
400     */
401    @Override
402    public long getLong(TemporalField field) {
403        if (field instanceof ChronoField) {
404            switch ((ChronoField) field) {
405                case MONTH_OF_YEAR: return month;
406                case EPOCH_MONTH: return getEpochMonth();
407                case YEAR_OF_ERA: return (year < 1 ? 1 - year : year);
408                case YEAR: return year;
409                case ERA: return (year < 1 ? 0 : 1);
410            }
411            throw new DateTimeException("Unsupported field: " + field.getName());
412        }
413        return field.getFrom(this);
414    }
415
416    private long getEpochMonth() {
417        return ((year - 1970) * 12L) + (month - 1);
418    }

463    /**
464     * Checks if the year is a leap year, according to the ISO proleptic
465     * calendar system rules.
466     * <p>
467     * This method applies the current rules for leap years across the whole time-line.
468     * In general, a year is a leap year if it is divisible by four without
469     * remainder. However, years divisible by 100, are not leap years, with
470     * the exception of years divisible by 400 which are.
471     * <p>
472     * For example, 1904 is a leap year it is divisible by 4.
473     * 1900 was not a leap year as it is divisible by 100, however 2000 was a
474     * leap year as it is divisible by 400.
475     * <p>
476     * The calculation is proleptic - applying the same rules into the far future and far past.
477     * This is historically inaccurate, but is correct for the ISO-8601 standard.
478     *
479     * @return true if the year is leap, false otherwise
480     */
481    public boolean isLeapYear() {
482        return IsoChronology.INSTANCE.isLeapYear(year);
483    }

549    /**
550     * Returns a copy of this year-month with the specified field set to a new value.
551     * <p>
552     * This returns a new {@code YearMonth}, based on this one, with the value
553     * for the specified field changed.
554     * This can be used to change any supported field, such as the year or month.
555     * If it is not possible to set the value, because the field is not supported or for
556     * some other reason, an exception is thrown.
557     * <p>
558     * If the field is a {@link ChronoField} then the adjustment is implemented here.
559     * The supported fields behave as follows:
560     * <ul>
561     * <li>{@code MONTH_OF_YEAR} -
562     *  Returns a {@code YearMonth} with the specified month-of-year.
563     *  The year will be unchanged.
564     * <li>{@code EPOCH_MONTH} -
565     *  Returns a {@code YearMonth} with the specified epoch-month.
566     *  This completely replaces the year and month of this object.
567     * <li>{@code YEAR_OF_ERA} -
568     *  Returns a {@code YearMonth} with the specified year-of-era
569     *  The month and era will be unchanged.
570     * <li>{@code YEAR} -
571     *  Returns a {@code YearMonth} with the specified year.
572     *  The month will be unchanged.
573     * <li>{@code ERA} -
574     *  Returns a {@code YearMonth} with the specified era.
575     *  The month and year-of-era will be unchanged.
576     * </ul>
577     * <p>
578     * In all cases, if the new value is outside the valid range of values for the field
579     * then a {@code DateTimeException} will be thrown.
580     * <p>
581     * All other {@code ChronoField} instances will throw a {@code DateTimeException}.
582     * <p>
583     * If the field is not a {@code ChronoField}, then the result of this method
584     * is obtained by invoking {@code TemporalField.adjustInto(Temporal, long)}
585     * passing {@code this} as the argument. In this case, the field determines
586     * whether and how to adjust the instant.
587     * <p>
588     * This instance is immutable and unaffected by this method call.
589     *
590     * @param field  the field to set in the result, not null
591     * @param newValue  the new value of the field in the result
592     * @return a {@code YearMonth} based on {@code this} with the specified field set, not null
593     * @throws DateTimeException if the field cannot be set
594     * @throws ArithmeticException if numeric overflow occurs
595     */
596    @Override
597    public YearMonth with(TemporalField field, long newValue) {
598        if (field instanceof ChronoField) {
599            ChronoField f = (ChronoField) field;
600            f.checkValidValue(newValue);
601            switch (f) {
602                case MONTH_OF_YEAR: return withMonth((int) newValue);
603                case EPOCH_MONTH: return plusMonths(newValue - getLong(EPOCH_MONTH));
604                case YEAR_OF_ERA: return withYear((int) (year < 1 ? 1 - newValue : newValue));
605                case YEAR: return withYear((int) newValue);
606                case ERA: return (getLong(ERA) == newValue ? this : withYear(1 - year));
607            }
608            throw new DateTimeException("Unsupported field: " + field.getName());
609        }
610        return field.adjustInto(this, newValue);
611    }

665     * {@inheritDoc}
666     * @throws DateTimeException {@inheritDoc}
667     * @throws ArithmeticException {@inheritDoc}

669    @Override
670    public YearMonth plus(long amountToAdd, TemporalUnit unit) {
671        if (unit instanceof ChronoUnit) {
672            switch ((ChronoUnit) unit) {
673                case MONTHS: return plusMonths(amountToAdd);
674                case YEARS: return plusYears(amountToAdd);
675                case DECADES: return plusYears(Jdk8Methods.safeMultiply(amountToAdd, 10));
676                case CENTURIES: return plusYears(Jdk8Methods.safeMultiply(amountToAdd, 100));
677                case MILLENNIA: return plusYears(Jdk8Methods.safeMultiply(amountToAdd, 1000));
678                case ERAS: return with(ERA, Jdk8Methods.safeAdd(getLong(ERA), amountToAdd));
679            }
680            throw new DateTimeException("Unsupported unit: " + unit.getName());
681        }
682        return unit.addTo(this, amountToAdd);

703     * Returns a copy of this year-month with the specified period in months added.
704     * <p>
705     * This instance is immutable and unaffected by this method call.
706     *
707     * @param monthsToAdd  the months to add, may be negative
708     * @return a {@code YearMonth} based on this year-month with the months added, not null
709     * @throws DateTimeException if the result exceeds the supported range
710     */
711    public YearMonth plusMonths(long monthsToAdd) {
712        if (monthsToAdd == 0) {
713            return this;
714        }
715        long monthCount = year * 12L + (month - 1);
716        long calcMonths = monthCount + monthsToAdd;  // safe overflow
717        int newYear = YEAR.checkValidIntValue(Jdk8Methods.floorDiv(calcMonths, 12));
718        int newMonth = Jdk8Methods.floorMod(calcMonths, 12) + 1;
719        return with(newYear, newMonth);
720    }

810    /**
811     * Adjusts the specified temporal object to have this year-month.
812     * <p>
813     * This returns a temporal object of the same observable type as the input
814     * with the year and month changed to be the same as this.
815     * <p>
816     * The adjustment is equivalent to using {@link Temporal#with(TemporalField, long)}
817     * passing {@link ChronoField#EPOCH_MONTH} as the field.
818     * If the specified temporal object does not use the ISO calendar system then
819     * a {@code DateTimeException} is thrown.
820     * <p>
821     * In most cases, it is clearer to reverse the calling pattern by using
822     * {@link Temporal#with(TemporalAdjuster)}:
823     * <pre>
824     *   // these two lines are equivalent, but the second approach is recommended
825     *   temporal = thisYearMonth.adjustInto(temporal);
826     *   temporal = temporal.with(thisYearMonth);
827     * </pre>
828     * <p>
829     * This instance is immutable and unaffected by this method call.
830     *
831     * @param temporal  the target object to be adjusted, not null
832     * @return the adjusted object, not null
833     * @throws DateTimeException if unable to make the adjustment
834     * @throws ArithmeticException if numeric overflow occurs
835     */
836    @Override
837    public Temporal adjustInto(Temporal temporal) {
838        if (Chronology.from(temporal).equals(IsoChronology.INSTANCE) == false) {
839            throw new DateTimeException("Adjustment only supported on ISO date-time");
840        }
841        return temporal.with(EPOCH_MONTH, getEpochMonth());

844    /**
845     * Calculates the period between this year-month and another year-month in
846     * terms of the specified unit.
847     * <p>
848     * This calculates the period between two year-months in terms of a single unit.
849     * The start and end points are {@code this} and the specified year-month.
850     * The result will be negative if the end is before the start.
851     * The {@code Temporal} passed to this method must be a {@code YearMonth}.
852     * For example, the period in years between two year-months can be calculated
853     * using {@code startYearMonth.periodUntil(endYearMonth, YEARS)}.
854     * <p>
855     * The calculation returns a whole number, representing the number of
856     * complete units between the two year-months.
857     * For example, the period in decades between 2012-06 and 2032-05
858     * will only be one decade as it is one month short of two decades.
859     * <p>
860     * This method operates in association with {@link TemporalUnit#between}.
861     * The result of this method is a {@code long} representing the amount of
862     * the specified unit. By contrast, the result of {@code between} is an
863     * object that can be used directly in addition/subtraction:
864     * <pre>
865     *   long period = start.periodUntil(end, YEARS);   // this method
866     *   dateTime.plus(YEARS.between(start, end));      // use in plus/minus
867     * </pre>
868     * <p>
869     * The calculation is implemented in this method for {@link ChronoUnit}.
870     * The units {@code MONTHS}, {@code YEARS}, {@code DECADES},
871     * {@code CENTURIES}, {@code MILLENNIA} and {@code ERAS} are supported.
872     * Other {@code ChronoUnit} values will throw an exception.
873     * <p>
874     * If the unit is not a {@code ChronoUnit}, then the result of this method
875     * is obtained by invoking {@code TemporalUnit.between(Temporal, Temporal)}
876     * passing {@code this} as the first argument and the input temporal as
877     * the second argument.
878     * <p>
879     * This instance is immutable and unaffected by this method call.
880     *
881     * @param endYearMonth  the end year-month, which must be a {@code YearMonth}, not null
882     * @param unit  the unit to measure the period in, not null
883     * @return the amount of the period between this year-month and the end year-month
884     * @throws DateTimeException if the period cannot be calculated
885     * @throws ArithmeticException if numeric overflow occurs
886     */
887    @Override
888    public long periodUntil(Temporal endYearMonth, TemporalUnit unit) {
889        if (endYearMonth instanceof YearMonth == false) {
890            Objects.requireNonNull(endYearMonth, "endYearMonth");
891            throw new DateTimeException("Unable to calculate period between objects of two different types");
892        }
893        YearMonth end = (YearMonth) endYearMonth;
894        if (unit instanceof ChronoUnit) {
895            long monthsUntil = end.getEpochMonth() - getEpochMonth();  // no overflow
896            switch ((ChronoUnit) unit) {
897                case MONTHS: return monthsUntil;
898                case YEARS: return monthsUntil / 12;
899                case DECADES: return monthsUntil / 120;
900                case CENTURIES: return monthsUntil / 1200;
901                case MILLENNIA: return monthsUntil / 12000;
902                case ERAS: return end.getLong(ERA) - getLong(ERA);
903            }
904            throw new DateTimeException("Unsupported unit: " + unit.getName());
905        }
906        return unit.between(this, endYearMonth);
907    }
